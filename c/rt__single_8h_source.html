114 static void read_table(header_t *header) {
115  table = MALLOC(
116  sizeof(uint32_t) + (sizeof(entry_t) * header->table_size));
117  if (table == NULL) {
118  log_error("failed to allocate memory for routing tables");
119  malloc_extras_terminate(EXIT_FAIL);
120  }
121  // Copy the size of the table
122  table->size = header->table_size;
123 
124  // Copy in the routing table entries
125  spin1_memcpy(table->entries, header->entries,
126  sizeof(entry_t) * table->size);
127 }

133 bool load_routing_table(uint32_t app_id) {
134  // Try to allocate sufficient room for the routing table.
135  uint32_t entry_id = rtr_alloc_id(table->size, app_id);
136  if (entry_id == 0) {
137  log_error("Unable to allocate routing table of size %u\n", table->size);
138  return FALSE;
139  }
140 
141  // Load entries into the table (provided the allocation succeeded).
142  // Note that although the allocation included the specified
143  // application ID we also need to include it as the most significant
144  // byte in the route (see `sark_hw.c`).
145  for (uint32_t i = 0; i < table->size; i++) {
146  entry_t entry = table->entries[i];
147  uint32_t route = entry.route | (app_id << 24);
148  uint success = rtr_mc_set(
149  entry_id + i, entry.key_mask.key, entry.key_mask.mask, route);
150  if (success == RTR_MC_SET_FAILED) {
151  log_warning("failed to set a router table entry at index %d",
152  entry_id + i);
153  }
154  }
155 
156  // Indicate we were able to allocate routing table entries.
157  return TRUE;
158 }
