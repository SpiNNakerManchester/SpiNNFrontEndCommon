149 static inline key_mask_t key_mask_merge(key_mask_t a, key_mask_t b) {
150  uint32_t new_xs = ~(a.key ^ b.key);
151  key_mask_t c;
152  c.mask = a.mask & b.mask & new_xs;
153  c.key = (a.key | b.key) & c.mask;
154 
155  return c;
