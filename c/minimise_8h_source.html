152 bool load_routing_table(uint32_t app_id) {
153  // Try to allocate sufficient room for the routing table.
154  uint32_t entry_id = rtr_alloc_id(table->size, app_id);
155  if (entry_id == 0) {
156  log_info("Unable to allocate routing table of size %u\n", table->size);
157  return FALSE;
158  }
159 
160  // Load entries into the table (provided the allocation succeeded).
161  // Note that although the allocation included the specified
162  // application ID we also need to include it as the most significant
163  // byte in the route (see `sark_hw.c`).
164  for (uint32_t i = 0; i < table->size; i++) {
165  entry_t entry = table->entries[i];
166  uint32_t route = entry.route | (app_id << 24);
167  uint success = rtr_mc_set(
168  entry_id + i, entry.keymask.key, entry.keymask.mask, route);
169  if (success == RTR_MC_SET_FAILED) {
170  log_error(
171  "failed to set a router table entry at index %d",
172  entry_id + i);
173  }
174  }
175 
176  // Indicate we were able to allocate routing table entries.
177  return TRUE;
178 }

200 void compress_start(uint unused0, uint unused1) {
201  use(unused0);
202  use(unused1);
203  uint32_t size_original;
204 
205  log_info("Starting on chip router compressor");
206 
207  // Prepare to minimise the routing tables
208  log_debug("looking for header using tag %u app_id %u", 1, sark_app_id());
209  header_t *header = (header_t *) sark_tag_ptr(1, sark_app_id());
210  log_debug("reading data from 0x%08x", (uint32_t) header);
211  print_header(header);
212 
213  // set the flag to something non-useful
214  sark.vcpu->user0 = 20;
215 
216  // Load the routing table
217  log_debug("start reading table");
218  read_table(header);
219  log_debug("finished reading table");
220 
221  // Store intermediate sizes for later reporting (if we fail to minimise)
222  size_original = routing_table_sdram_get_n_entries();
223 
224  // Try to load the table
225  log_debug("check if compression is needed and compress if needed");
226  if (header->compress_only_when_needed == 1) {
227  if (load_routing_table(header->app_id)) {
228  cleanup_and_exit(header);
229  } else {
230  // Otherwise remove default routes.
231  log_debug("remove default routes from minimiser");
232  remove_default_routes_minimise(table);
233  if (load_routing_table(header->app_id)) {
234  cleanup_and_exit(header);
235  } else {
236  //Oops we need the defaults back before trying compression
237  log_debug("free the tables entries");
238  FREE(table->entries);
239  read_table(header);
240  }
241  }
242  }
243 
244  // Get the target length of the routing table
245  log_debug("acquire target length");
246  uint32_t target_length = 0;
247  if (header->compress_as_much_as_possible == 0) {
248  target_length = rtr_alloc_max();
249  }
250  log_info("target length of %d", target_length);
251 
252  // Perform the minimisation
253  log_debug("minimise");
254  minimise(target_length);
255  log_debug("done minimise");
256 
257  // report size to the host for provenance aspects
258  log_info("has compressed the router table to %d entries",
259  routing_table_sdram_get_n_entries());
260 
261  // Try to load the routing table
262  log_debug("try loading tables");
263  if (load_routing_table(header->app_id)) {
264  cleanup_and_exit(header);
265  } else {
266  // Otherwise give up and exit with an error
267  log_error("Failed to minimise routing table to fit %u entries. "
268  "(Original table: %u after compression: %u).",
269  rtr_alloc_max(), size_original,
270  routing_table_sdram_get_n_entries());
271 
272  // Free the block of SDRAM used to load the routing table.
273  log_debug("free sdram blocks which held router tables");
274  FREE((void *) header);
275 
276  // set the failed flag and exit
277  malloc_extras_terminate(EXIT_FAIL);
278  }
279 }
