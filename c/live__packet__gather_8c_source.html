178static inline void write_word(void *base, uint32_t index, uint32_t value) {
179 uint16_t *ary = base;
180 uint32_t idx = index * 2;
181 ary[idx++] = CLAMP16(value);
182 ary[idx] = CLAMP16(value >> 16);
183}

189static inline void write_short(void *base, uint32_t index, uint32_t value) {
190 uint16_t *ary = base;
191 ary[index] = CLAMP16(value);
192}

197 uint8_t event_count = buffer_index;

207static void flush_events(void) {
208 // Send the event message only if there is data
209 if ((buffer_index > 0) && (
210 (config.packets_per_timestamp == 0) ||
211 (packets_sent < config.packets_per_timestamp))) {
212 // Get the event count depending on if there is a payload or not
213 uint8_t event_count = get_event_count();
214
215 // insert appropriate header
216 sdp_msg_aer_header[0] = eieio_constant_header | CLAMP8(event_count);
217
218 g_event_message.length =
219 sizeof(sdp_hdr_t) + sdp_msg_aer_header_len +
220 event_count * event_size;
221
222 // Add the timestamp if required
223 if (sdp_msg_aer_payload_prefix && config.payload_timestamp) {
224 if (!HAVE_WIDE_LOAD(config.packet_type)) {
225 write_short(sdp_msg_aer_payload_prefix, 0, time);
226 } else {
227 write_word(sdp_msg_aer_payload_prefix, 0, time);
228 }
229 }
230
231 spin1_send_sdp_msg(&g_event_message, 1);
232 packets_sent++;
233 provenance_data.number_of_sent_messages++;
234 }
235
236 // reset counter
237 buffer_index = 0;
238}

257static void timer_callback(UNUSED uint unused0, UNUSED uint unused1) {
258 // flush the spike message and sent it over the Ethernet
259 flush_events();
260
261 // increase time variable to keep track of current timestep
262 time++;
263 log_debug("Timer tick %u", time);
264
265 // Reset the count of packets sent in the current timestep
266 packets_sent = 0;
267
268 // check if the simulation has run to completion
269 if (simulation_is_finished()) {
270 simulation_handle_pause_resume(NULL);
271
272 // Subtract 1 from the time so this tick gets done again on the next
273 // run
274 time--;
275
276 simulation_ready_to_read();
277 }
278}

292static void process_incoming_event(uint key) {
293 log_debug("Processing key %x", key);
294
295 // process the received spike
296 if (!HAVE_WIDE_LOAD(config.packet_type)) {
297 // 16 bit packet
298 write_short(sdp_msg_aer_data, buffer_index++,
299 key >> config.key_right_shift);
300
301 // if there is a payload to be added
302 if (HAVE_PAYLOAD(config.packet_type) && !config.payload_timestamp) {
303 write_short(sdp_msg_aer_data, buffer_index++, 0);
304 } else if (HAVE_PAYLOAD(config.packet_type) && config.payload_timestamp) {
305 write_short(sdp_msg_aer_data, buffer_index++, time);
306 }
307 } else {
308 // 32 bit packet
309 write_word(sdp_msg_aer_data, buffer_index++, key);
310
311 // if there is a payload to be added
312 if (HAVE_PAYLOAD(config.packet_type) && !config.payload_timestamp) {
313 write_word(sdp_msg_aer_data, buffer_index++, 0);
314 } else if (HAVE_PAYLOAD(config.packet_type) && config.payload_timestamp) {
315 write_word(sdp_msg_aer_data, buffer_index++, time);
316 }
317 }
318 provenance_data.number_of_gathered_events++;
319}

369static void incoming_event_process_callback(
370 UNUSED uint unused0, UNUSED uint unused1) {
371 do {
372 uint32_t key, payload;
373
374 if (circular_buffer_get_next(without_payload_buffer, &key)) {
375 process_incoming_event(key);
376 } else if (circular_buffer_get_next(with_payload_buffer, &key)
377 && circular_buffer_get_next(with_payload_buffer, &payload)) {
378 process_incoming_event_payload(key, payload);
379 } else {
380 processing_events = false;
381 break;
382 }
383
384 // send packet if enough data is stored
385 flush_events_if_full();
386 } while (processing_events);
387}

396static void incoming_event_callback(uint key, UNUSED uint unused) {
397 log_debug("Received key %x", key);
398
399 if (circular_buffer_add(without_payload_buffer, key)) {
400 if (!processing_events) {
401 processing_events = true;
402 spin1_trigger_user_event(0, 0);
403 }
404 } else {
405 provenance_data.number_of_overflows_no_payload++;
406 }
407}

416static void incoming_event_payload_callback(uint key, uint payload) {
417 log_debug("Received key %x, payload %x", key, payload);
418
419 if (circular_buffer_add(with_payload_buffer, key)) {
420 circular_buffer_add(with_payload_buffer, payload);
421 if (!processing_events) {
422 processing_events = true;
423 spin1_trigger_user_event(0, 0);
424 }
425 } else {
426 provenance_data.number_of_overflows_with_payload++;

435static void read_parameters(struct lpg_config *sdram_config) {
436 // P bit
437 config.apply_prefix = sdram_config->apply_prefix;
438 // Prefix data
439 config.prefix = sdram_config->prefix;
440 // F bit (for the receiver)
441 config.prefix_type = sdram_config->prefix_type;
442 // Type bits
443 config.packet_type = sdram_config->packet_type;
444 // Right packet shift (for the sender)
445 config.key_right_shift = sdram_config->key_right_shift;
446 // T bit
447 config.payload_timestamp = sdram_config->payload_timestamp;
448 // D bit
449 config.payload_apply_prefix = sdram_config->payload_apply_prefix;
450 // Payload prefix data (for the receiver)
451 config.payload_prefix = sdram_config->payload_prefix;
452 // Right payload shift (for the sender)
453 config.payload_right_shift = sdram_config->payload_right_shift;
454 config.sdp_tag = sdram_config->sdp_tag;
455 config.sdp_dest = sdram_config->sdp_dest;
456 config.packets_per_timestamp = sdram_config->packets_per_timestamp;
457
458 log_info("apply_prefix: %d", config.apply_prefix);
459 log_info("prefix: %08x", config.prefix);
460 log_info("prefix_type: %d", config.prefix_type);
461 log_info("packet_type: %d", config.packet_type);
462 log_info("key_right_shift: %d", config.key_right_shift);
463 log_info("payload_timestamp: %d", config.payload_timestamp);

469 log_info("packets_per_timestamp: %d", config.packets_per_timestamp);

476static bool initialize(uint32_t *timer_period) {
477 // Get the address this core's DTCM data starts at from SRAM
478 data_specification_metadata_t *ds_regions =
479 data_specification_get_data_address();
480
481 // Read the header
482 if (!data_specification_read_header(ds_regions)) {
483 return false;
484 }
485
486 // Get the timing details and set up the simulation interface
487 if (!simulation_initialise(
488 data_specification_get_region(SYSTEM_REGION, ds_regions),
489 APPLICATION_NAME_HASH, timer_period, &simulation_ticks,
490 &infinite_run, &time, SDP, DMA)) {
491 return false;
492 }
493 simulation_set_provenance_function(
494 record_provenance_data,
495 data_specification_get_region(PROVENANCE_REGION, ds_regions));
496
497 // Fix simulation ticks to be one extra timer period to soak up last events
498 if (infinite_run != TRUE) {
499 simulation_ticks++;
500 }
501
502 // Read the parameters
503 read_parameters(
504 data_specification_get_region(CONFIGURATION_REGION, ds_regions));
505 return true;
506}

510static bool configure_sdp_msg(void) {
511 log_info("configure_sdp_msg");
512
513 switch (config.packet_type) {
514 case NO_PAYLOAD_16:
515 event_size = 2;
516 break;
517 case PAYLOAD_16:
518 event_size = 4;
519 break;
520 case NO_PAYLOAD_32:
521 event_size = 4;
522 break;
523 case PAYLOAD_32:
524 event_size = 8;
525 break;
526 default:
527 log_error("unknown packet type: %d", config.packet_type);
528 return false;
529 }
530
531 // initialise SDP header
532 g_event_message.tag = config.sdp_tag;
533 // No reply required
534 g_event_message.flags = 0x07;
535 // Chip 0,0
536 g_event_message.dest_addr = config.sdp_dest;
537 // Dump through Ethernet
538 g_event_message.dest_port = PORT_ETH;
539 // Set up monitoring address and port
540 g_event_message.srce_addr = spin1_get_chip_id();
541 g_event_message.srce_port = (3 << PORT_SHIFT) | spin1_get_core_id();
542
543 // check incompatible options
544 if (config.payload_timestamp && config.payload_apply_prefix
545 && HAVE_PAYLOAD(config.packet_type)) {
546 log_error("Timestamp can either be included as payload prefix or as"
547 "payload to each key, not both");
548 return false;
549 }
550 if (config.payload_timestamp && !config.payload_apply_prefix
551 && !HAVE_PAYLOAD(config.packet_type)) {
552 log_error("Timestamp can either be included as payload prefix or as"
553 "payload to each key, but current configuration does not"
554 "specify either of these");
555 return false;
556 }
557
558 // initialise AER header
559 // pointer to data space
560 sdp_msg_aer_header = &g_event_message.cmd_rc;
561
562 eieio_constant_header = 0;
563 eieio_constant_header |= config.apply_prefix << APPLY_PREFIX;
564 eieio_constant_header |= config.prefix_type << PREFIX_UPPER;
565 eieio_constant_header |= config.payload_apply_prefix << APPLY_PAYLOAD_PREFIX;
566 eieio_constant_header |= config.payload_timestamp << PAYLOAD_IS_TIMESTAMP;
567 eieio_constant_header |= config.packet_type << PACKET_TYPE;
568
569 // pointers for AER packet header, prefix and data
570 // Point to the half-word after main header half-word
571 sdp_msg_aer_data = sdp_msg_aer_header + 1;
572 if (config.apply_prefix) {
573 // pointer to key prefix, so data is one half-word further ahead
574 write_short(sdp_msg_aer_header, 1, config.prefix);
575 sdp_msg_aer_data++;
576 }
577
578 if (config.payload_apply_prefix) {
579 // pointer to payload prefix
580 sdp_msg_aer_payload_prefix = sdp_msg_aer_data;
581
582 if (!HAVE_WIDE_LOAD(config.packet_type)) {
583 //16 bit payload prefix; advance data position by one half word
584 sdp_msg_aer_data++;
585 if (!config.payload_timestamp) {
586 // add payload prefix as required - not a timestamp
587 write_short(sdp_msg_aer_payload_prefix, 0, config.payload_prefix);
588 }
589 } else {
590 //32 bit payload prefix; advance data position by two half words
591 sdp_msg_aer_data += 2;
592 if (!config.payload_timestamp) {
593 // add payload prefix as required - not a timestamp
594 write_word(sdp_msg_aer_payload_prefix, 0, config.payload_prefix);
595 }
596 }
597 }
598
599 // compute header length in bytes
600 sdp_msg_aer_header_len =
601 (sdp_msg_aer_data - sdp_msg_aer_header) * sizeof(uint16_t);
602
603 log_debug("sdp_msg_aer_header: %08x", sdp_msg_aer_header);
604 log_debug("sdp_msg_aer_payload_prefix: %08x", sdp_msg_aer_payload_prefix);
605 log_debug("sdp_msg_aer_data: %08x", sdp_msg_aer_data);
606 log_debug("sdp_msg_aer_header_len: %d", sdp_msg_aer_header_len);
607
608 packets_sent = 0;
609 buffer_index = 0;
610
611 return true;
612}

615void c_main(void) {
616 // Configure system
617 uint32_t timer_period = 0;
618 if (!initialize(&timer_period)) {
619 log_error("Error in initialisation - exiting!");
620 rt_error(RTE_SWERR);
621 }
622
623 // Configure SDP message
624 if (!configure_sdp_msg()) {
625 rt_error(RTE_SWERR);
626 }
627
628 // Set up circular buffers for multicast message reception
629 without_payload_buffer = circular_buffer_initialize(BUFFER_CAPACITY);
630 with_payload_buffer = circular_buffer_initialize(BUFFER_CAPACITY * 2);
631
632 // Set timer_callback
633 spin1_set_timer_tick(timer_period);
634
635 // Register callbacks
636 spin1_callback_on(MC_PACKET_RECEIVED, incoming_event_callback, MC_PACKET);
637 spin1_callback_on(
638 MCPL_PACKET_RECEIVED, incoming_event_payload_callback, MC_PACKET);
639 spin1_callback_on(USER_EVENT, incoming_event_process_callback, USER);
640 spin1_callback_on(TIMER_TICK, timer_callback, TIMER);
641
642 // Start the time at "-1" so that the first tick will be 0
643 time = UINT32_MAX;
644 simulation_run();
645}
