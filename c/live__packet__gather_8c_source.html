203 static void flush_events(void) {
204  // Send the event message only if there is data
205  if ((buffer_index > 0) && (
206  (config.packets_per_timestamp == 0) ||
207  (packets_sent < config.packets_per_timestamp))) {
208  // Get the event count depending on if there is a payload or not
209  uint8_t event_count = get_event_count();
210 
211  // insert appropriate header
212  sdp_msg_aer_header[0] = eieio_constant_header | CLAMP8(event_count);
213 
214  g_event_message.length =
215  sizeof(sdp_hdr_t) + sdp_msg_aer_header_len +
216  event_count * event_size;
217 
218  // Add the timestamp if required
219  if (sdp_msg_aer_payload_prefix && config.payload_timestamp) {
220  if (!HAVE_WIDE_LOAD(config.packet_type)) {
221  write_short(sdp_msg_aer_payload_prefix, 0, time);
222  } else {
223  write_word(sdp_msg_aer_payload_prefix, 0, time);
224  }
225  }
226 
227  spin1_send_sdp_msg(&g_event_message, 1);
228  packets_sent++;
229  }
230 
231  // reset counter
232  buffer_index = 0;
233 }

252 static void timer_callback(UNUSED uint unused0, UNUSED uint unused1) {
253  // flush the spike message and sent it over the Ethernet
254  flush_events();
255 
256  // increase time variable to keep track of current timestep
257  time++;
258  log_debug("Timer tick %u", time);
259 
260  // Reset the count of packets sent in the current timestep
261  packets_sent = 0;
262 
263  // check if the simulation has run to completion
264  if ((infinite_run != TRUE) && (time >= simulation_ticks)) {
265  simulation_handle_pause_resume(NULL);
266 
267  // Subtract 1 from the time so this tick gets done again on the next
268  // run
269  time--;
270 
271  simulation_ready_to_read();
272  }
273 }

287 static void process_incoming_event(uint key) {
288  log_debug("Processing key %x", key);
289 
290  // process the received spike
291  if (!HAVE_WIDE_LOAD(config.packet_type)) {
292  // 16 bit packet
293  write_short(sdp_msg_aer_data, buffer_index++,
294  key >> config.key_right_shift);
295 
296  // if there is a payload to be added
297  if (HAVE_PAYLOAD(config.packet_type) && !config.payload_timestamp) {
298  write_short(sdp_msg_aer_data, buffer_index++, 0);
299  } else if (HAVE_PAYLOAD(config.packet_type) && config.payload_timestamp) {
300  write_short(sdp_msg_aer_data, buffer_index++, time);
301  }
302  } else {
303  // 32 bit packet
304  write_word(sdp_msg_aer_data, buffer_index++, key);
305 
306  // if there is a payload to be added
307  if (HAVE_PAYLOAD(config.packet_type) && !config.payload_timestamp) {
308  write_word(sdp_msg_aer_data, buffer_index++, 0);
309  } else if (HAVE_PAYLOAD(config.packet_type) && config.payload_timestamp) {
310  write_word(sdp_msg_aer_data, buffer_index++, time);
311  }
312  }
313 }

362 static void incoming_event_process_callback(
363  UNUSED uint unused0, UNUSED uint unused1) {
364  do {
365  uint32_t key, payload;
366 
367  if (circular_buffer_get_next(without_payload_buffer, &key)) {
368  process_incoming_event(key);
369  } else if (circular_buffer_get_next(with_payload_buffer, &key)
370  && circular_buffer_get_next(with_payload_buffer, &payload)) {
371  process_incoming_event_payload(key, payload);
372  } else {
373  processing_events = false;
374  break;
375  }
376 
377  // send packet if enough data is stored
378  flush_events_if_full();
379  } while (processing_events);
380 }

469 static bool initialize(uint32_t *timer_period) {
470  // Get the address this core's DTCM data starts at from SRAM
471  data_specification_metadata_t *ds_regions =
472  data_specification_get_data_address();
473 
474  // Read the header
475  if (!data_specification_read_header(ds_regions)) {
476  return false;
477  }
478 
479  // Get the timing details and set up the simulation interface
480  if (!simulation_initialise(
481  data_specification_get_region(SYSTEM_REGION, ds_regions),
482  APPLICATION_NAME_HASH, timer_period, &simulation_ticks,
483  &infinite_run, &time, SDP, DMA)) {
484  return false;
485  }
486  simulation_set_provenance_function(
487  record_provenance_data,
488  data_specification_get_region(PROVENANCE_REGION, ds_regions));
489 
490  // Fix simulation ticks to be one extra timer period to soak up last events
491  if (infinite_run != TRUE) {
492  simulation_ticks++;
493  }
494 
495  // Read the parameters
496  read_parameters(
497  data_specification_get_region(CONFIGURATION_REGION, ds_regions));
498  return true;
499 }

503 static bool configure_sdp_msg(void) {
504  log_info("configure_sdp_msg");
505 
506  switch (config.packet_type) {
507  case NO_PAYLOAD_16:
508  event_size = 2;
509  break;
510  case PAYLOAD_16:
511  event_size = 4;
512  break;
513  case NO_PAYLOAD_32:
514  event_size = 4;
515  break;
516  case PAYLOAD_32:
517  event_size = 8;
518  break;
519  default:
520  log_error("unknown packet type: %d", config.packet_type);
521  return false;
522  }
523 
524  // initialise SDP header
525  g_event_message.tag = config.sdp_tag;
526  // No reply required
527  g_event_message.flags = 0x07;
528  // Chip 0,0
529  g_event_message.dest_addr = config.sdp_dest;
530  // Dump through Ethernet
531  g_event_message.dest_port = PORT_ETH;
532  // Set up monitoring address and port
533  g_event_message.srce_addr = spin1_get_chip_id();
534  g_event_message.srce_port = (3 << PORT_SHIFT) | spin1_get_core_id();
535 
536  // check incompatible options
537  if (config.payload_timestamp && config.payload_apply_prefix
538  && HAVE_PAYLOAD(config.packet_type)) {
539  log_error("Timestamp can either be included as payload prefix or as"
540  "payload to each key, not both");
541  return false;
542  }
543  if (config.payload_timestamp && !config.payload_apply_prefix
544  && !HAVE_PAYLOAD(config.packet_type)) {
545  log_error("Timestamp can either be included as payload prefix or as"
546  "payload to each key, but current configuration does not"
547  "specify either of these");
548  return false;
549  }
550 
551  // initialise AER header
552  // pointer to data space
553  sdp_msg_aer_header = &g_event_message.cmd_rc;
554 
555  eieio_constant_header = 0;
556  eieio_constant_header |= config.apply_prefix << APPLY_PREFIX;
557  eieio_constant_header |= config.prefix_type << PREFIX_UPPER;
558  eieio_constant_header |= config.payload_apply_prefix << APPLY_PAYLOAD_PREFIX;
559  eieio_constant_header |= config.payload_timestamp << PAYLOAD_IS_TIMESTAMP;
560  eieio_constant_header |= config.packet_type << PACKET_TYPE;
561 
562  // pointers for AER packet header, prefix and data
563  // Point to the half-word after main header half-word
564  sdp_msg_aer_data = sdp_msg_aer_header + 1;
565  if (config.apply_prefix) {
566  // pointer to key prefix, so data is one half-word further ahead
567  write_short(sdp_msg_aer_header, 1, config.prefix);
568  sdp_msg_aer_data++;
569  }
570 
571  if (config.payload_apply_prefix) {
572  // pointer to payload prefix
573  sdp_msg_aer_payload_prefix = sdp_msg_aer_data;
574 
575  if (!HAVE_WIDE_LOAD(config.packet_type)) {
576  //16 bit payload prefix; advance data position by one half word
577  sdp_msg_aer_data++;
578  if (!config.payload_timestamp) {
579  // add payload prefix as required - not a timestamp
580  write_short(sdp_msg_aer_payload_prefix, 0, config.payload_prefix);
581  }
582  } else {
583  //32 bit payload prefix; advance data position by two half words
584  sdp_msg_aer_data += 2;
585  if (!config.payload_timestamp) {
586  // add payload prefix as required - not a timestamp
587  write_word(sdp_msg_aer_payload_prefix, 0, config.payload_prefix);
588  }
589  }
590  }
591 
592  // compute header length in bytes
593  sdp_msg_aer_header_len =
594  (sdp_msg_aer_data - sdp_msg_aer_header) * sizeof(uint16_t);
595 
596  log_debug("sdp_msg_aer_header: %08x", sdp_msg_aer_header);
597  log_debug("sdp_msg_aer_payload_prefix: %08x", sdp_msg_aer_payload_prefix);
598  log_debug("sdp_msg_aer_data: %08x", sdp_msg_aer_data);
599  log_debug("sdp_msg_aer_header_len: %d", sdp_msg_aer_header_len);
600 
601  packets_sent = 0;
602  buffer_index = 0;
603 
604  return true;
605 }

608 void c_main(void) {
609  // Configure system
610  uint32_t timer_period = 0;
611  if (!initialize(&timer_period)) {
612  log_error("Error in initialisation - exiting!");
613  rt_error(RTE_SWERR);
614  }
615 
616  // Configure SDP message
617  if (!configure_sdp_msg()) {
618  rt_error(RTE_SWERR);
619  }
620 
621  // Set up circular buffers for multicast message reception
622  without_payload_buffer = circular_buffer_initialize(BUFFER_CAPACITY);
623  with_payload_buffer = circular_buffer_initialize(BUFFER_CAPACITY * 2);
624 
625  // Set timer_callback
626  spin1_set_timer_tick(timer_period);
627 
628  // Register callbacks
629  spin1_callback_on(MC_PACKET_RECEIVED, incoming_event_callback, MC_PACKET);
630  spin1_callback_on(
631  MCPL_PACKET_RECEIVED, incoming_event_payload_callback, MC_PACKET);
632  spin1_callback_on(USER_EVENT, incoming_event_process_callback, USER);
633  spin1_callback_on(TIMER_TICK, timer_callback, TIMER);
634 
635  // Start the time at "-1" so that the first tick will be 0
636  time = UINT32_MAX;
637  simulation_run();
638 }
