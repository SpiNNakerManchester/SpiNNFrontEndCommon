41 entry_t* routing_table_get_entry_marked(uint32_t entry_id_to_find, int marker) {
42  uint32_t table_id = entry_id_to_find >> TABLE_SHIFT;
43  if (table_id >= multi_table.n_sub_tables) {
44  log_error("Id %d to big for %d tables marker %d",
45  entry_id_to_find, multi_table.n_sub_tables, marker);
46  malloc_extras_terminate(RTE_SWERR);
47  }
48  uint32_t local_id = entry_id_to_find & LOCAL_ID_ADD;
49  if (local_id >= multi_table.sub_tables[table_id]->size) {
50  log_error("Id %d has local_id %d which is too big for "
51  "table of size %d marker %d",
52  entry_id_to_find, local_id,
53  multi_table.sub_tables[table_id]->size, marker);
54  malloc_extras_terminate(RTE_SWERR);
55  }
56  return &multi_table.sub_tables[table_id]->entries[local_id];
57 }

70 entry_t* routing_table_append_get_entry(void) {
71  // check that we're not hitting the max entries supported by the table
72  if (multi_table.n_entries == (int) multi_table.max_entries) {
73  log_error(
74  "there is no more space in this multi-table for this entry.");
75  malloc_extras_terminate(RTE_SWERR);
76  }
77 
78  // locate right table index
79  uint32_t table_id = multi_table.n_entries >> TABLE_SHIFT;
80  if (table_id >= multi_table.n_sub_tables) {
81  log_error("Id %d to big for %d tables",
82  multi_table.n_entries, multi_table.n_sub_tables);
83  malloc_extras_terminate(RTE_SWERR);
84  }
85 
86  // locate entry index
87  uint32_t local_id = multi_table.n_entries & LOCAL_ID_ADD;
88  if (local_id != multi_table.sub_tables[table_id]->size) {
89  log_error("Id %d has local_id %d which is big for %d table",
90  multi_table.n_entries, local_id,
91  multi_table.sub_tables[table_id]->size);
92  malloc_extras_terminate(RTE_SWERR);
93  }
94 
95  // update trackers.
96  multi_table.n_entries++;
97  multi_table.sub_tables[table_id]->size++;
98  return &multi_table.sub_tables[table_id]->entries[local_id];
99 }
