74void start_compression_process(void) {
75 log_debug("in compression phase");
76
77 // restart timer (also puts us in running state)
78 spin1_resume(SYNC_NOWAIT);
79
80#ifdef DEBUG_COMPRESSOR
81 if (comms_sdram->mid_point >= 100) {
82 log_warning("HACK fail at 100 plus bitfeilds!");
83 comms_sdram->compressor_state = FAILED_TO_COMPRESS;
84 return;
85 }
86#endif
87
88#ifdef DEBUG_COMPRESSOR
89 if ((comms_sdram->mid_point > 0) || (!hack_malloc_failed)) {
90 log_warning("HACK malloc fail!");
91 hack_malloc_failed = true;
92 comms_sdram->compressor_state = FAILED_MALLOC;
93 return;
94 }
95#endif
96
97 // run compression
98 bool success = run_compressor(
99 compress_as_much_as_possible, &failed_by_malloc, &stop_compressing);
100
101 // turn off timer and set us into pause state
102 spin1_pause();
103
104 // Decode whether we succeeded or failed.
105 int max_length = rtr_alloc_max();
106 if (success && (routing_table_get_n_entries() <= max_length)) {
107 log_info("Passed minimise_run() with success code: %d", success);
108 routing_tables_save(comms_sdram->routing_tables);
109 comms_sdram->compressor_state = SUCCESSFUL_COMPRESSION;
110 return;
111 }
112
113 // Not a success, could be one of 4 failure states
114 log_info("Failed minimise_run() with success code: %d", success);
115 if (failed_by_malloc) { // malloc failed somewhere
116 log_debug("failed malloc response");
117 comms_sdram->compressor_state = FAILED_MALLOC;
118 } else if (comms_sdram->sorter_instruction != RUN) { // control killed it
119 log_debug("force fail response");
120 comms_sdram->compressor_state = FORCED_BY_COMPRESSOR_CONTROL;
121 log_debug("send ack");
122 } else if (stop_compressing) { // ran out of time
123 log_debug("time fail response");
124 comms_sdram->compressor_state = RAN_OUT_OF_TIME;
125 } else { // after finishing compression, still could not fit into table.
126 log_debug("failed by space response");
127 comms_sdram->compressor_state = FAILED_TO_COMPRESS;
128 }
129}

146void run_compression_process(void) {
147 if (comms_sdram->mid_point > 0) {
148 log_info("Run with %d tables and %d mid_point out of %d bitfields",
149 comms_sdram->routing_tables->n_sub_tables,
150 comms_sdram->mid_point,
151 comms_sdram->sorted_bit_fields->n_bit_fields);
152 } else {
153 log_info("Run with %d tables and no bitfields",
154 comms_sdram->routing_tables->n_sub_tables);
155 }
156 log_debug("setting up fake heap for sdram usage");
157 malloc_extras_initialise_with_fake_heap(comms_sdram->fake_heap_data);
158 log_debug("set up fake heap for sdram usage");
159
160 // Set all status flags
161 failed_by_malloc = false;
162 stop_compressing = false;
163 counter = 0;
164
165 setup_routing_tables();
166
167 log_debug("starting compression attempt with %d entries",
168 routing_table_get_n_entries());
169
170 // start compression process
171 start_compression_process();
172}

178static inline bool process_run(compressor_states compressor_state) {
179 switch (compressor_state) {
180 case PREPARED:
181 comms_sdram->compressor_state = COMPRESSING;
182 run_compression_process();
183 return true;
184 case COMPRESSING:
185 // Should not be back in this loop before result set
186 return false;
187 case FAILED_MALLOC:
188 case FORCED_BY_COMPRESSOR_CONTROL:
189 case SUCCESSFUL_COMPRESSION:
190 case FAILED_TO_COMPRESS:
191 case RAN_OUT_OF_TIME:
192 // waiting for sorter to pick up result
193 return true;
194 case UNUSED_CORE:
195 // Should never happen
196 return false;
197 }
198 return false;
199}

205static inline bool process_prepare(compressor_states compressor_state) {
206 switch (compressor_state) {
207 case UNUSED_CORE:
208 // First prepare
209 log_info("Prepared for the first time");
210 comms_sdram->compressor_state = PREPARED;
211 return true;
212 case FAILED_MALLOC:
213 case FORCED_BY_COMPRESSOR_CONTROL:
214 case SUCCESSFUL_COMPRESSION:
215 case FAILED_TO_COMPRESS:
216 case RAN_OUT_OF_TIME:
217 // clear previous result
218 log_info("prepared");
219 comms_sdram->compressor_state = PREPARED;
220 return true;
221 case PREPARED:
222 // waiting for sorter to pick up result
223 return true;
224 case COMPRESSING:
225 // Should never happen
226 return false;
227 }
228 return false;
229}

238static inline bool process_force(compressor_states compressor_state) {
239 switch (compressor_state) {
240 case COMPRESSING:
241 // passed to compressor as *sorter_instruction
242 // Do nothing until compressor notices changed
243 return true;
244 case FORCED_BY_COMPRESSOR_CONTROL:
245 // Waiting for sorter to pick up
246 return true;
247 case FAILED_MALLOC:
248 case SUCCESSFUL_COMPRESSION:
249 case FAILED_TO_COMPRESS:
250 case RAN_OUT_OF_TIME:
251 log_info("Force detected so changing result to ack");
252 // The results other than MALLOC no longer matters
253 comms_sdram->compressor_state = FORCED_BY_COMPRESSOR_CONTROL;
254 return true;
255 case PREPARED:
256 case UNUSED_CORE:
257 // Should never happen
258 return false;
259 }
260 return false;
261}

268static void wait_for_instructions(UNUSED uint unused0, UNUSED uint unused1) {
269 // set if combination of user2 and user3 is expected
270 bool users_match = true;
271
272 // cache the states so they dont change inside one loop
273 compressor_states compressor_state = comms_sdram->compressor_state;
274 instructions_to_compressor sorter_state = comms_sdram->sorter_instruction;
275 // When debugging Log if changed
276 if (sorter_state != previous_sorter_state) {
277 previous_sorter_state = sorter_state;
278 log_debug("Sorter state changed sorter: %d compressor %d",
279 sorter_state, compressor_state);
280 }
281 if (compressor_state != previous_compressor_state) {
282 previous_compressor_state = compressor_state;
283 log_debug("Compressor state changed sorter: %d compressor %d",
284 sorter_state, compressor_state);
285 }
286
287 switch (sorter_state) {
288 case PREPARE:
289 users_match = process_prepare(compressor_state);
290 break;
291 case RUN:
292 users_match = process_run(compressor_state);
293 break;
294 case FORCE_TO_STOP:
295 users_match = process_force(compressor_state);
296 break;
297 case NOT_COMPRESSOR:
298 // For some reason compressor sees this state too
299 case TO_BE_PREPARED:
300 users_match = (compressor_state == UNUSED_CORE);
301 break;
302 case DO_NOT_USE:
303 log_warning("DO_NOT_USE detected exiting wait");
304 spin1_pause();
305 return;
306 }
307 if (users_match) {
308 spin1_schedule_callback(
309 wait_for_instructions, 0, 0, COMPRESSION_START_PRIORITY);
310 } else {
311 log_error("Unexpected combination of sorter_state %d and "
312 "compressor_state %d", sorter_state, compressor_state);
313 malloc_extras_terminate(RTE_SWERR);
314 }
315}

322static void timer_callback(UNUSED uint unused0, UNUSED uint unused1) {
323 counter++;
324
325 if (counter >= max_counter) {
326 stop_compressing = true;
327 log_info("passed timer point");
328 spin1_pause();
329 }
330
331 // check that the sorter has told the compressor to finish for any reason
332 if (comms_sdram->sorter_instruction != RUN) {
333 stop_compressing = true;
334 if (comms_sdram->compressor_state == COMPRESSING) {
335 log_info("Sorter cancelled run request");
336 } else if (comms_sdram->sorter_instruction == DO_NOT_USE) {
337 log_info("Compressor no longer to be used");
338 } else {
339 log_warning("timer weirdness %d %d",
340 comms_sdram->sorter_instruction,
341 comms_sdram->compressor_state);
342 }
343 spin1_pause();
344 }

348static void initialise(void) {
349 log_info("Setting up stuff to allow bitfield compressor to occur.");
350
351 log_debug("reading time_for_compression_attempt");
352 vcpu_t *restrict sark_virtual_processor_info = (vcpu_t *) SV_VCPU;
353 vcpu_t *restrict this_vcpu_info =
354 &sark_virtual_processor_info[spin1_get_core_id()];
355
356 uint32_t time_for_compression_attempt = this_vcpu_info->user1;
357 log_info("time_for_compression_attempt = %d",
358 time_for_compression_attempt);
359
360 // 0 = compress_only_when_needed, 1 = compress_as_much_as_possible
361 uint32_t int_value = this_vcpu_info->user2;
362 if (int_value & 1) {
363 compress_as_much_as_possible = true;
364 }
365 log_info("int %d, compress_as_much_as_possible = %d",
366 int_value, compress_as_much_as_possible);
367
368 // Get the pointer for all cores
369 comms_sdram = (comms_sdram_t *) this_vcpu_info->user3;
370
371 // Now move the pointer to the comms for this core
372 comms_sdram += spin1_get_core_id();
373
374 // sort out timer (this is shrank to be called 1000 times, so that we can
375 // check for sorter controls. e.g. is the sorter forces the compressor
376 // to stop early).
377 max_counter = time_for_compression_attempt / TIMER_ITERATIONS;
378 spin1_set_timer_tick(TIMER_ITERATIONS);
379 spin1_callback_on(TIMER_TICK, timer_callback, TIMER_TICK_PRIORITY);
380 log_info("my processor id is %d", spin1_get_core_id());
381}
