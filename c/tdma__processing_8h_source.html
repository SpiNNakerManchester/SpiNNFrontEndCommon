95static inline void tdma_processing_send_packet(
96 uint32_t transmission_key, uint32_t payload,
97 uint32_t with_payload, uint32_t timer_count) {
98 // Spin1 API ticks - to know when the timer wraps
99 extern uint ticks;
100
101 uint32_t timer_value = timer1_control->current_value;
102
103 // Find the next valid phase to send in; might run out of phases, at
104 // which point we will sent immediately. We also should just send
105 // if the timer has already expired completely as then we are really late!
106 while ((ticks == timer_count) && (timer_value < tdma_expected_time)
107 && (tdma_expected_time >= tdma_params.min_expected_time)) {
108 tdma_expected_time -= tdma_params.time_between_sends;
109 }
110
111 n_tdma_behind_times += tdma_expected_time < tdma_params.min_expected_time;
112
113 // Wait until the expected time to send; might already have passed in
114 // which case we just skip this
115 while ((ticks == timer_count) &&
116 (timer1_control->current_value > tdma_expected_time)) {
117 tdma_waits++;
118 }
119
120 // Send the spike
121 uint32_t time = tc[T1_COUNT];
122 if (time < tdma_latest_send) {
123 tdma_latest_send = time;
124 }
125 send_packet(transmission_key, payload, with_payload);
126}
