101 static inline bool bit_set_add(bit_set_t* b, unsigned int i) {
102  if (b->n_elements <= i) {
103  return false;
104  }
105 
106  // Determine the word and bit
107  unsigned int word = i / BITS_IN_A_WORD;
108  unsigned int bit = 1 << (i & 31);
109 
110  // Set the word and bit
111  b->_data[word] |= bit;
112 
113  // Increment the count of set elements
114  b->count++;
115  return true;
116 }

122 static inline bool bit_set_contains(bit_set_t *b, unsigned int i) {

129  uint32_t bit = 1 << (i & 31);

137 static inline bool bit_set_remove(bit_set_t *b, unsigned int i) {
138  if (!bit_set_contains(b, i)) {
139  return false;
140  }
141  // Determine the word and bit
142  unsigned int word = i >> 5;
143  unsigned int bit = 1 << (i & 0x1f);
144 
145  // Decrement the count of set elements
146  b->count--;
147 
148  // Unset the bit of the appropriate word
149  b->_data[word] &= ~bit;
150  return true;
151 }
