63 static inline bool bit_set_init(bit_set_t *b, unsigned int length) {
64  // Malloc space for the data
65  unsigned int n_words = length / BITS_IN_A_WORD;
66  if (length % BITS_IN_A_WORD) {
67  n_words++;
68  }
69 
70  uint32_t *data = (uint32_t *) MALLOC(n_words * sizeof(uint32_t));
71  if (data == NULL) {
72  b->_data = NULL;
73  b->n_elements = 0;
74  b->n_words = 0;
75  return false;
76  }
77 
78  b->_data = data;
79  b->n_words = n_words;
80  b->n_elements = length;
81  bit_set_clear(b);
82  return true;
83 }

98 static inline bool bit_set_add(bit_set_t* b, unsigned int i) {
99  if (b->n_elements <= i) {
100  return false;
101  }
102 
103  // Determine the word and bit
104  unsigned int word = i / BITS_IN_A_WORD;
105  unsigned int bit = 1 << (i & 31);
106 
107  // Set the word and bit
108  b->_data[word] |= bit;
109 
110  // Increment the count of set elements
111  b->count++;
112  return true;
113 }

119 static inline bool bit_set_contains(bit_set_t *b, unsigned int i) {
120  if (b->n_elements <= i) {
121  return false;
122  }
123 
124  // Determine the word and bit
125  unsigned int word = i / BITS_IN_A_WORD;
126  uint32_t bit = 1 << (i & 31);
127  return (bool) (b->_data[word] & bit);
128 }

134 static inline bool bit_set_remove(bit_set_t *b, unsigned int i) {
135  if (!bit_set_contains(b, i)) {
136  return false;
137  }
138  // Determine the word and bit
139  unsigned int word = i >> 5;
140  unsigned int bit = 1 << (i & 0x1f);
141 
142  // Decrement the count of set elements
143  b->count--;
144 
145  // Unset the bit of the appropriate word
146  b->_data[word] &= ~bit;
147  return true;
148 }
