35 void profiler_init(uint32_t* data_region) {
36  log_info("Reading profile setup from 0x%08x", data_region);
37  profiler_state.samples_remaining = data_region[0];
38  profiler_state.count = &data_region[0];
39  profiler_state.output = &data_region[1];
40 
41  log_info("Initialising profiler with storage for %u samples starting at 0x%08x",
42  profiler_state.samples_remaining, profiler_state.output);
43 
44  // If profiler is turned on, start timer 2 with no clock divider
45  if (profiler_state.samples_remaining > 0) {
46  tc[T2_CONTROL] = 0x82;
47  tc[T2_LOAD] = 0;
48  }
49 }

52 void profiler_finalise(void) {
53  uint32_t words_written = (profiler_state.output - profiler_state.count) - 1;
54  *profiler_state.count = words_written;
55  log_info("Profiler wrote %u bytes to 0x%08x",
56  (words_written * 4) + 4, profiler_state.count);
57 }
