143 static void synchronise_start(uint unused0, uint unused1) {
144  while (resume_wait()) {
145  wait_for_interrupt();
146  }
147  sark_cpu_state(CPU_STATE_RUN);
148  stored_start_function();
149 
150  // If we are not using the timer, no-one else resets the event, so do it now
151  // (event comes from sark.h - this is how SARK knows whether to wait for a
152  // SYNC0 or SYNC1 message)
153  if (!uses_timer) {
154  event.wait ^= 1;
155  }
156 }

163 static void simulation_control_scp_callback(uint mailbox, UNUSED uint port) {
164  sdp_msg_t *msg = (sdp_msg_t *) mailbox;
165  uint16_t length = msg->length;
166 
167  switch (msg->cmd_rc) {
168  case CMD_STOP:
169  log_info("Received exit signal. Program complete.");
170 
171  // free the message to stop overload
172  spin1_msg_free(msg);
173 
174  // call any stored exit callbacks
175  if (stored_exit_function != NULL) {
176  log_info("Calling pre-exit function");
177  stored_exit_function();
178  }
179  log_info("Exiting");
180  spin1_exit(0);
181  break;
182 
183  case CMD_RUNTIME:
184  log_info("Setting the runtime of this model to %d starting at %d",
185  msg->arg1, msg->arg3);
186  log_info("Setting the flag of infinite run for this model to %d",
187  msg->arg2);
188 
189  // resetting the simulation time pointer
190  *pointer_to_simulation_time = msg->arg1;
191  *pointer_to_infinite_run = msg->arg2;
192  // We start at time - 1 because the first thing models do is
193  // increment a time counter
194  *pointer_to_current_time = (msg->arg3 - 1);
195 
196  if (stored_resume_function != NULL) {
197  log_info("Calling pre-resume function");
198  stored_resume_function();
199  stored_resume_function = NULL;
200  }
201 
202  if (stored_start_function != NULL) {
203  spin1_schedule_callback(synchronise_start, 0, 0, 1);
204  }
205  if (uses_timer) {
206  log_info("Resuming");
207  spin1_resume(SYNC_WAIT);
208  } else {
209  if (event.wait) {
210  sark_cpu_state(CPU_STATE_SYNC1);
211  } else {
212  sark_cpu_state(CPU_STATE_SYNC0);
213  }
214  }
215 
216 
217  // If we are told to send a response, send it now
218  if (msg->data[0] == 1) {
219  send_ok_response(msg);
220  }
221 
222  // free the message to stop overload
223  spin1_msg_free(msg);
224  break;
225 
226  case PROVENANCE_DATA_GATHERING:
227  log_info("Forced provenance gathering");
228 
229  // force provenance to be executed and then exit
230  execute_provenance_storage();
231 
232  // call any stored exit callbacks
233  if (stored_exit_function != NULL) {
234  log_info("Calling pre-exit function");
235  stored_exit_function();
236  }
237  spin1_msg_free(msg);
238  spin1_exit(1);
239  break;
240 
241  case IOBUF_CLEAR:
242  // run clear iobuf code
243  sark_io_buf_reset();
244 
245  // If we are told to send a response, send it now
246  if (msg->arg3 == 1) {
247  send_ok_response(msg);
248  }
249 
250  // free the message to stop overload
251  spin1_msg_free(msg);
252  break;
253 
254  default:
255  // should never get here
256  log_error("received packet with unknown command code %d",
257  msg->cmd_rc);
258  spin1_msg_free(msg);
259  }
260 }

265 static void simulation_sdp_callback_handler(uint mailbox, uint port) {
266  if (sdp_callback[port] != NULL) {
267  // if a callback is associated with the port, process it
268  sdp_callback[port](mailbox, port);

270  // if no callback is associated, dump the received packet

299 bool simulation_dma_transfer_done_callback_on(uint tag, callback_t callback) {
300  // ensure that tag being allocated is less than max tag
301  if (tag >= MAX_DMA_CALLBACK_TAG) {
302  log_error("Cannot handle tag value above %d, please reduce the tag "
303  "value accordingly.", MAX_DMA_CALLBACK_TAG - 1);
304  return false;
305  }
306 
307  // allocate tag callback if not already allocated
308  if (dma_complete_callbacks[tag] != NULL) {
309  // if allocated already, raise error
310  log_error("Cannot allocate DMA transfer callback on tag %d as its "
311  "already been allocated.", tag);
312  return false;
313  }
314 
315  dma_complete_callbacks[tag] = callback;
316  return true;
317 }

323 bool simulation_initialise(
324  address_t address, uint32_t expected_app_magic_number,
325  uint32_t* timer_period, uint32_t *simulation_ticks_pointer,
326  uint32_t *infinite_run_pointer, uint32_t *time_pointer,
327  int sdp_packet_callback_priority,
328  int dma_transfer_done_callback_priority) {
329  struct simulation_config *config = (void *) address;
330 
331  // handle the timing reading
332  if (config->application_magic_number != expected_app_magic_number) {
333  log_error("Unexpected magic number 0x%08x instead of 0x%08x at 0x%08x",
334  config->application_magic_number, expected_app_magic_number,
335  &config->application_magic_number);
336  return false;
337  }
338 
339  if (sdp_packet_callback_priority < -1) {
340  log_error("The SDP callback priority should be set to a number greater "
341  "than or equal to -1. It is currently set to %d",
342  sdp_packet_callback_priority);
343  return false;
344  }
345 
346  // transfer data to pointers for end user usage
347  *timer_period = config->timer_period;
348 
349  // handle the SDP callback for the simulation
350  pointer_to_simulation_time = simulation_ticks_pointer;
351  pointer_to_infinite_run = infinite_run_pointer;
352  pointer_to_current_time = time_pointer;
353 
354  spin1_callback_on(SDP_PACKET_RX, simulation_sdp_callback_handler,
355  sdp_packet_callback_priority);
356  simulation_sdp_callback_on(config->control_sdp_port,
357  simulation_control_scp_callback);
358  spin1_callback_on(DMA_TRANSFER_DONE, simulation_dma_transfer_done_callback,
359  dma_transfer_done_callback_priority);
360 
361  // if all simulation initialisation complete return true,
362  return true;
363 }
