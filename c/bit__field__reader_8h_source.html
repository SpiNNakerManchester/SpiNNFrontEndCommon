39 static uint32_t detect_redundant_packet_count(
40  filter_info_t *restrict filter_info) {
41  uint32_t n_filtered_packets = 0;
42  uint32_t n_neurons = filter_info->n_atoms;
43  for (uint neuron_id = 0; neuron_id < n_neurons; neuron_id++) {
44  if (!bit_field_test(filter_info->data, neuron_id)) {
45  n_filtered_packets += 1;
46  }
47  }
48  return n_filtered_packets;
49 }

53 static inline void order_bitfields(
54  sorted_bit_fields_t *restrict sorted_bit_fields) {
55  // Semantic sugar to avoid extra lookup all the time
56  int *restrict processor_ids = sorted_bit_fields->processor_ids;
57  int *restrict sort_order = sorted_bit_fields->sort_order;
58  filter_info_t **restrict bit_fields = sorted_bit_fields->bit_fields;
59 
60  // To label each row in sort order
61  for (int sorted_index = 0; sorted_index < sorted_bit_fields->n_bit_fields;
62  sorted_index++) {
63 
64  // Find the processor with highest number of packets coming in
65  int worst_processor = 0;
66  uint32_t highest_neurons = 0;
67  for (int c = 0; c < MAX_PROCESSORS; c++) {
68  if (processor_totals[c] > highest_neurons){
69  worst_processor = c;
70  highest_neurons = processor_totals[c];
71  }
72  }
73 
74  // Label the row pointer to be the header as next
75  int index = processor_heads[worst_processor];
76  sort_order[index] = sorted_index;
77  log_debug("processor %u index %u total %u",
78  worst_processor, index, processor_totals[worst_processor]);
79 
80  // If there is another row with the same processor
81  if ((index < sorted_bit_fields->n_bit_fields - 1) &&
82  (processor_ids[index] == processor_ids[index + 1])) {
83  log_debug("i %u processor %u index %u more %u total %u",
84  sorted_index, worst_processor, index,
85  sorted_bit_fields->n_bit_fields,
86  processor_totals[worst_processor]);
87 
88  // reduce the packet count bu redundancy
89  processor_totals[worst_processor] -=
90  detect_redundant_packet_count(bit_fields[index]);
91 
92  // move the pointer
93  processor_heads[worst_processor] += 1;
94  } else {
95  // otherwise set the counters to ignore this processor
96  processor_totals[worst_processor] = NO_BIT_FIELDS;
97  processor_heads[worst_processor] = DO_NOT_USE;
98 
99  log_debug("i %u processor %u index %u last %u total %u",
100  sorted_index, worst_processor, index,
101  sorted_bit_fields->n_bit_fields,
102  processor_totals[worst_processor]);
103  }
104  }
105 }

111 static inline void sort_by_order(
112  sorted_bit_fields_t *restrict sorted_bit_fields) {
113  // Every time there is a swap at least one of the rows is moved to the
114  // final place.
115  //
116  // There is one check per row in the for loop plus if the first fails
117  // up to one more for each row about to be moved to the correct place.
118 
119  int *restrict processor_ids = sorted_bit_fields->processor_ids;
120  int *restrict sort_order = sorted_bit_fields->sort_order;
121  filter_info_t **restrict bit_fields = sorted_bit_fields->bit_fields;
122 
123  // Check each row in the lists
124  for (int i = 0; i < sorted_bit_fields->n_bit_fields; i++) {
125  // check that the data is in the correct place
126  while (sort_order[i] != i) {
127  int j = sort_order[i];
128  // If not swap the data there into the correct place
129  int temp_processor_id = processor_ids[i];
130  processor_ids[i] = processor_ids[j];
131  processor_ids[j] = temp_processor_id;
132 
133  uint32_t temp_sort_order = sort_order[i];
134  sort_order[i] = sort_order[j];
135  sort_order[j] = temp_sort_order;
136 
137  filter_info_t* bit_field_temp = bit_fields[i];
138  bit_fields[i] = bit_fields[j];
139  bit_fields[j] = bit_field_temp;
140  }
141  }
142 }

201 static inline void fills_in_sorted_bit_fields_and_tracker(
202  region_addresses_t *restrict region_addresses,
203  sorted_bit_fields_t *restrict sorted_bit_fields) {
204  // iterate through a processors bitfield region and add to the bf by
205  // processor struct, whilst updating num of total param.
206  for (int r_id = 0, index = 0; r_id < region_addresses->n_triples; r_id++) {
207  // locate data for malloc memory calcs
208  filter_region_t *restrict filter_region =
209  region_addresses->triples[r_id].filter;
210  int processor = region_addresses->triples[r_id].processor;
211 
212  if (filter_region->n_redundancy_filters == 0) {
213  // no bitfields to point at or sort so total can stay zero
214  break;
215  }
216 
217  // store the index in bitfields list where this processors bitfields
218  // start being read in at. (not sorted)
219  processor_heads[processor] = index;
220 
221  // read in the processors bitfields.
222  for (int bf_id = 0; bf_id < filter_region->n_redundancy_filters;
223  bf_id++, index++) {
224  // update trackers.
225  sorted_bit_fields->processor_ids[index] = processor;
226  sorted_bit_fields->bit_fields[index] =
227  &filter_region->filters[bf_id];
228  processor_totals[processor] +=
229  filter_region->filters[bf_id].n_atoms;
230  }
231 
232  // accum the incoming packets from bitfields which have no redundancy
233  for (int bf_id = filter_region->n_redundancy_filters;
234  bf_id < filter_region->n_filters; bf_id++) {
235  processor_totals[processor] +=
236  filter_region->filters[bf_id].n_atoms;
237  }
238  }
239 }

246 static inline void bit_field_reader_read_in_bit_fields(
247  region_addresses_t *restrict region_addresses,
248  sorted_bit_fields_t *restrict sorted_bit_fields) {
249  //init data tracking structures
250  for (int i = 0; i < MAX_PROCESSORS; i++) {
251  processor_heads[i] = DO_NOT_USE;
252  processor_totals[i] = 0;
253  }
254 
255  // track positions and incoming packet counts.
256  fills_in_sorted_bit_fields_and_tracker(region_addresses, sorted_bit_fields);
257 
258  //TODO safety code to be removed.
259  for (int i = 0; i < MAX_PROCESSORS; i++) {
260  log_debug("i: %d, head: %d count: %d",
261  i, processor_heads[i], processor_totals[i]);
262  }
263 #if 0
264  print_structs(sorted_bit_fields);
265 #endif
266 
267  // sort bitfields so that bit-fields are in order of most impact on worse
268  // affected cores. so that merged bitfields should reduce packet rates
269  // fairly across cores on the chip.
270  order_bitfields(sorted_bit_fields);
271 
272  //TODO safety code to be removed.
273 #if 0
274  print_structs(sorted_bit_fields);
275 #endif
276 
277  // sort the bitfields by key.
278  // NOTE: does not affect previous sort, as this directly affects the index
279  // in the bitfield array, whereas the previous sort affects the sort index
280  // array.
281  sort_by_key(sorted_bit_fields);
282 
283  // useful for debugging
284 #if 0
285  print_structs(sorted_bit_fields);
286 #endif
287 }

294 static inline sorted_bit_fields_t * bit_field_reader_initialise(
295  region_addresses_t *restrict region_addresses) {
296  sorted_bit_fields_t *restrict sorted_bit_fields = MALLOC_SDRAM(
297  sizeof(sorted_bit_fields_t));
298  if (sorted_bit_fields == NULL) {
299  log_error("failed to allocate DTCM for sorted bitfields.");
300  return NULL;
301  }
302 
303  // figure out how many bitfields we need
304  log_debug("n triples of addresses = %d", region_addresses->n_triples);
305  int n_bit_fields = 0;
306  for (int r_id = 0; r_id < region_addresses->n_triples; r_id++) {
307  n_bit_fields +=
308  region_addresses->triples[r_id].filter->n_redundancy_filters;
309  log_info("Core %d has %u bitfields of which %u have redundancy",
310  region_addresses->triples[r_id].processor,
311  region_addresses->triples[r_id].filter->n_filters,
312  region_addresses->triples[r_id].filter->n_redundancy_filters);
313  }
314  sorted_bit_fields->n_bit_fields = n_bit_fields;
315  log_info("Number of bitfields with redundancy found is %u",
316  sorted_bit_fields->n_bit_fields);
317 
318  // if there are no bit-fields just return sorted bitfields.
319  if (n_bit_fields != 0) {
320  // malloc the separate bits of the sorted bitfield struct
321  sorted_bit_fields->bit_fields = MALLOC_SDRAM(
322  n_bit_fields * sizeof(filter_info_t*));
323  if (sorted_bit_fields->bit_fields == NULL) {
324  log_error(
325  "cannot allocate memory for the sorted bitfield addresses");
326  FREE(sorted_bit_fields);
327  return NULL;
328  }
329 
330  sorted_bit_fields->processor_ids = MALLOC_SDRAM(
331  n_bit_fields * sizeof(int));
332  if (sorted_bit_fields->processor_ids == NULL) {
333  log_error("cannot allocate memory for the sorted bitfields with "
334  "processors ids");
335  FREE(sorted_bit_fields->bit_fields);
336  FREE(sorted_bit_fields);
337  return NULL;
338  }
339 
340  sorted_bit_fields->sort_order = MALLOC_SDRAM(
341  n_bit_fields * sizeof(int));
342  if (sorted_bit_fields->sort_order == NULL) {
343  log_error("cannot allocate memory for the sorted bitfields with "
344  "sort_order");
345  FREE(sorted_bit_fields->bit_fields);
346  FREE(sorted_bit_fields->processor_ids);
347  FREE(sorted_bit_fields);
348  return NULL;
349  }
350 
351  // init to -1, else random data (used to make prints cleaner)
352  for (int sorted_index = 0; sorted_index < n_bit_fields;
353  sorted_index++) {
354  sorted_bit_fields->sort_order[sorted_index] = FAILED_TO_FIND;
355  }
356  }
357 
358  // return
359  return sorted_bit_fields;
360 }
