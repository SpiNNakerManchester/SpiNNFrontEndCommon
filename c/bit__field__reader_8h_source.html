35 static uint32_t detect_redundant_packet_count(
36  filter_info_t *restrict filter_info) {
37  uint32_t n_filtered_packets = 0;
38  uint32_t n_neurons = filter_info->n_atoms;
39  for (uint neuron_id = 0; neuron_id < n_neurons; neuron_id++) {
40  if (!bit_field_test(filter_info->data, neuron_id)) {
41  n_filtered_packets += 1;
42  }
43  }
44  return n_filtered_packets;
45 }

49 static inline void order_bitfields(
50  sorted_bit_fields_t *restrict sorted_bit_fields) {
51  // Semantic sugar to avoid extra lookup all the time
52  int *restrict processor_ids = sorted_bit_fields->processor_ids;
53  int *restrict sort_order = sorted_bit_fields->sort_order;
54  filter_info_t **restrict bit_fields = sorted_bit_fields->bit_fields;
55 
56  // To label each row in sort order
57  for (int sorted_index = 0; sorted_index < sorted_bit_fields->n_bit_fields;
58  sorted_index++) {
59 
60  // Find the processor with highest number of packets coming in
61  int worst_processor = 0;
62  uint32_t highest_neurons = 0;
63  for (int c = 0; c < MAX_PROCESSORS; c++) {
64  if (processor_totals[c] > highest_neurons){
65  worst_processor = c;
66  highest_neurons = processor_totals[c];
67  }
68  }
69 
70  // Label the row pointer to be the header as next
71  int index = processor_heads[worst_processor];
72  sort_order[index] = sorted_index;
73  log_debug("processor %u index %u total %u",
74  worst_processor, index, processor_totals[worst_processor]);
75 
76  // If there is another row with the same processor
77  if ((index < sorted_bit_fields->n_bit_fields - 1) &&
78  (processor_ids[index] == processor_ids[index + 1])) {
79  log_debug("i %u processor %u index %u more %u total %u",
80  sorted_index, worst_processor, index,
81  sorted_bit_fields->n_bit_fields,
82  processor_totals[worst_processor]);
83 
84  // reduce the packet count bu redundancy
85  processor_totals[worst_processor] -=
86  detect_redundant_packet_count(bit_fields[index]);
87 
88  // move the pointer
89  processor_heads[worst_processor] += 1;
90  } else {
91  // otherwise set the counters to ignore this processor
92  processor_totals[worst_processor] = NO_BIT_FIELDS;
93  processor_heads[worst_processor] = DO_NOT_USE;
94 
95  log_debug("i %u processor %u index %u last %u total %u",
96  sorted_index, worst_processor, index,
97  sorted_bit_fields->n_bit_fields,
98  processor_totals[worst_processor]);
99  }
100  }
101 }

107 static inline void sort_by_order(
108  sorted_bit_fields_t *restrict sorted_bit_fields) {
109  // Everytime there is a swap at least one of the rows is moved to the
110  // final place.
111  // There is one check per row in the for loop plus if the first fails
112  // up to one more for each row about to be moved to the correct place.
113 
114  int *restrict processor_ids = sorted_bit_fields->processor_ids;
115  int *restrict sort_order = sorted_bit_fields->sort_order;
116  filter_info_t **restrict bit_fields = sorted_bit_fields->bit_fields;
117 
118  // Check each row in the lists
119  for (int i = 0; i < sorted_bit_fields->n_bit_fields; i++) {
120  // check that the data is in the correct place
121  while (sort_order[i] != i) {
122  int j = sort_order[i];
123  // If not swap the data there into the correct place
124  int temp_processor_id = processor_ids[i];
125  processor_ids[i] = processor_ids[j];
126  processor_ids[j] = temp_processor_id;
127 
128  uint32_t temp_sort_order = sort_order[i];
129  sort_order[i] = sort_order[j];
130  sort_order[j] = temp_sort_order;
131 
132  filter_info_t* bit_field_temp = bit_fields[i];
133  bit_fields[i] = bit_fields[j];
134  bit_fields[j] = bit_field_temp;
135  }
136  }

196 static inline void fills_in_sorted_bit_fields_and_tracker(
197  region_addresses_t *restrict region_addresses,
198  sorted_bit_fields_t *restrict sorted_bit_fields) {
199  // iterate through a processors bitfield region and add to the bf by
200  // processor struct, whilst updating n bf total param.
201  for (int r_id = 0, index = 0; r_id < region_addresses->n_triples; r_id++) {
202  // locate data for malloc memory calcs
203  filter_region_t *restrict filter_region =
204  region_addresses->triples[r_id].filter;
205  int processor = region_addresses->triples[r_id].processor;
206 
207  if (filter_region->n_redundancy_filters == 0) {
208  // no bitfields to point at or sort so total can stay zero
209  break;
210  }
211 
212  // store the index in bitfields list where this processors bitfields
213  // start being read in at. (not sorted)
214  processor_heads[processor] = index;
215 
216  // read in the processors bitfields.
217  for (int bf_id = 0; bf_id < filter_region->n_redundancy_filters;
218  bf_id++, index++) {
219  // update trackers.
220  sorted_bit_fields->processor_ids[index] = processor;
221  sorted_bit_fields->bit_fields[index] =
222  &filter_region->filters[bf_id];
223  processor_totals[processor] +=
224  filter_region->filters[bf_id].n_atoms;
225  }
226 
227  // accum the incoming packets from bitfields which have no redundancy
228  for (int bf_id = filter_region->n_redundancy_filters;
229  bf_id < filter_region->n_filters; bf_id++) {
230  processor_totals[processor] +=
231  filter_region->filters[bf_id].n_atoms;
232  }
233  }
234 }

241 static inline void bit_field_reader_read_in_bit_fields(
242  region_addresses_t *restrict region_addresses,
243  sorted_bit_fields_t *restrict sorted_bit_fields) {
244  //init data tracking structures
245  for (int i = 0; i < MAX_PROCESSORS; i++) {
246  processor_heads[i] = DO_NOT_USE;
247  processor_totals[i] = 0;
248  }
249 
250  // track positions and incoming packet counts.
251  fills_in_sorted_bit_fields_and_tracker(region_addresses, sorted_bit_fields);
252 
253  //TODO safety code to be removed.
254  for (int i = 0; i < MAX_PROCESSORS; i++) {
255  log_debug("i: %d, head: %d count: %d",
256  i, processor_heads[i], processor_totals[i]);
257  }
258 #if 0
259  print_structs(sorted_bit_fields);
260 #endif
261 
262  // sort bitfields so that bit-fields are in order of most impact on worse
263  // affected cores. so that merged bitfields should reduce packet rates
264  // fairly across cores on the chip.
265  order_bitfields(sorted_bit_fields);
266 
267  //TODO safety code to be removed.
268 #if 0
269  print_structs(sorted_bit_fields);
270 #endif
271 
272  // sort the bitfields by key.
273  // NOTE: does not affect previous sort, as this directly affects the index
274  // in the bitfield array, whereas the previous sort affects the sort index
275  // array.
276  sort_by_key(sorted_bit_fields);
277 
278  // useful for debugging
279 #if 0
280  print_structs(sorted_bit_fields);
281 #endif
282 }

290 static inline sorted_bit_fields_t * bit_field_reader_initialise(
291  region_addresses_t *restrict region_addresses) {
292  sorted_bit_fields_t *restrict sorted_bit_fields = MALLOC_SDRAM(
293  sizeof(sorted_bit_fields_t));
294  if (sorted_bit_fields == NULL) {
295  log_error("failed to allocate DTCM for sorted bitfields.");
296  return NULL;
297  }
298 
299  // figure out how many bitfields we need
300  log_debug("n triples of addresses = %d", region_addresses->n_triples);
301  int n_bit_fields = 0;
302  for (int r_id = 0; r_id < region_addresses->n_triples; r_id++) {
303  n_bit_fields +=
304  region_addresses->triples[r_id].filter->n_redundancy_filters;
305  log_info("Core %d has %u bitfields of which %u have redundancy",
306  region_addresses->triples[r_id].processor,
307  region_addresses->triples[r_id].filter->n_filters,
308  region_addresses->triples[r_id].filter->n_redundancy_filters);
309  }
310  sorted_bit_fields->n_bit_fields = n_bit_fields;
311  log_info("Number of bitfields with redundancy found is %u",
312  sorted_bit_fields->n_bit_fields);
313 
314  // if there are no bit-fields just return sorted bitfields.
315  if (n_bit_fields != 0) {
316  // malloc the separate bits of the sorted bitfield struct
317  sorted_bit_fields->bit_fields = MALLOC_SDRAM(
318  n_bit_fields * sizeof(filter_info_t*));
319  if (sorted_bit_fields->bit_fields == NULL) {
320  log_error(
321  "cannot allocate memory for the sorted bitfield addresses");
322  FREE(sorted_bit_fields);
323  return NULL;
324  }
325 
326  sorted_bit_fields->processor_ids = MALLOC_SDRAM(
327  n_bit_fields * sizeof(int));
328  if (sorted_bit_fields->processor_ids == NULL) {
329  log_error("cannot allocate memory for the sorted bitfields with "
330  "processors ids");
331  FREE(sorted_bit_fields->bit_fields);
332  FREE(sorted_bit_fields);
333  return NULL;
334  }
335 
336  sorted_bit_fields->sort_order = MALLOC_SDRAM(
337  n_bit_fields * sizeof(int));
338  if (sorted_bit_fields->sort_order == NULL) {
339  log_error("cannot allocate memory for the sorted bitfields with "
340  "sort_order");
341  FREE(sorted_bit_fields->bit_fields);
342  FREE(sorted_bit_fields->processor_ids);
343  FREE(sorted_bit_fields);
344  return NULL;
345  }
346 
347  // init to -1, else random data (used to make prints cleaner)
348  for (int sorted_index = 0; sorted_index < n_bit_fields;
349  sorted_index++) {
350  sorted_bit_fields->sort_order[sorted_index] = FAILED_TO_FIND;
351  }
352  }
353 
354  // return
355  return sorted_bit_fields;
356 }
