38 static uint32_t n_redundant(filter_info_t *restrict filter) {
39  uint32_t n_atoms = filter->n_atoms;
40  uint32_t n_words = get_bit_field_size(n_atoms);
41  return n_atoms - count_bit_field(filter->data, n_words);

46 static inline void order_bitfields(
47  sorted_bit_fields_t *restrict sorted_bit_fields) {
48  // Semantic sugar to avoid extra lookup all the time
49  int *restrict processor_ids = sorted_bit_fields->processor_ids;
50  int *restrict sort_order = sorted_bit_fields->sort_order;
51  filter_info_t **restrict bit_fields = sorted_bit_fields->bit_fields;
52 
53  // To label each row in sort order
54  for (int sorted_index = 0; sorted_index < sorted_bit_fields->n_bit_fields;
55  sorted_index++) {
56 
57  // Find the processor with highest number of packets coming in
58  int worst_processor = 0;
59  uint32_t highest_neurons = 0;
60  for (int c = 0; c < MAX_PROCESSORS; c++) {
61  if (processor_totals[c] > highest_neurons){
62  worst_processor = c;
63  highest_neurons = processor_totals[c];
64  }
65  }
66 
67  // Label the row pointer to be the header as next
68  int index = processor_heads[worst_processor];
69  sort_order[index] = sorted_index;
70  log_debug("processor %u index %u total %u",
71  worst_processor, index, processor_totals[worst_processor]);
72 
73  // If there is another row with the same processor
74  if ((index < sorted_bit_fields->n_bit_fields - 1) &&
75  (processor_ids[index] == processor_ids[index + 1])) {
76  log_debug("i %u processor %u index %u more %u total %u",
77  sorted_index, worst_processor, index,
78  sorted_bit_fields->n_bit_fields,
79  processor_totals[worst_processor]);
80 
81  // reduce the packet count bu redundancy
82  processor_totals[worst_processor] -= n_redundant(bit_fields[index]);
83 
84  // move the pointer
85  processor_heads[worst_processor] += 1;
86  } else {
87  // otherwise set the counters to ignore this processor
88  processor_totals[worst_processor] = NO_BIT_FIELDS;
89  processor_heads[worst_processor] = DO_NOT_USE;
90 
91  log_debug("i %u processor %u index %u last %u total %u",
92  sorted_index, worst_processor, index,
93  sorted_bit_fields->n_bit_fields,
94  processor_totals[worst_processor]);
95  }
96  }
97 }

101 static inline void sort_by_key(
102  sorted_bit_fields_t *restrict sorted_bit_fields) {
103  // Semantic sugar to avoid extra lookup all the time
104  int *restrict processor_ids = sorted_bit_fields->processor_ids;
105  int *restrict sort_order = sorted_bit_fields->sort_order;
106  filter_info_t **restrict bit_fields = sorted_bit_fields->bit_fields;
107  int i, j;
108 
109  for (i = 1; i < sorted_bit_fields->n_bit_fields; i++) {
110  const int temp_processor_id = processor_ids[i];
111  const uint32_t temp_sort_order = sort_order[i];
112  filter_info_t *const bit_field_temp = bit_fields[i];
113  register uint32_t key = bit_field_temp->key;
114 
115  for (j = i; j > 0 && bit_fields[j - 1]->key > key; j--) {
116  processor_ids[j] = processor_ids[j - 1];
117  sort_order[j] = sort_order[j - 1];
118  bit_fields[j] = bit_fields[j - 1];
119  }
120 
121  processor_ids[j] = temp_processor_id;
122  sort_order[j] = temp_sort_order;
123  bit_fields[j] = bit_field_temp;
124  }
125 }

149 static inline void sort_by_redundancy(sorted_bit_fields_t *sorted_bit_fields,
150  uint32_t start, uint32_t end) {
151  // We only need to sort the bit fields, as this assumes it is called
152  // before the index is filled in, and where start and n_items covers items
153  // with the same processor id
154  filter_info_t **bit_fields = sorted_bit_fields->bit_fields;
155  for (uint32_t i = start + 1; i < end; i++) {
156  filter_info_t *temp_bf = bit_fields[i];
157 
158  uint32_t j;
159  for (j = i; j > start && n_redundant(bit_fields[j - 1]) < n_redundant(temp_bf); j--) {
160  bit_fields[j] = bit_fields[j - 1];
161  }
162  bit_fields[j] = temp_bf;
163  }
164 }

171 static inline void fills_in_sorted_bit_fields_and_tracker(
172  region_addresses_t *restrict region_addresses,
173  sorted_bit_fields_t *restrict sorted_bit_fields) {
174  // iterate through a processors bitfield region and add to the bf by
175  // processor struct, whilst updating num of total param.
176  for (int r_id = 0, index = 0; r_id < region_addresses->n_processors; r_id++) {
177  // locate data for malloc memory calcs
178  filter_region_t *restrict filter_region =
179  region_addresses->processors[r_id].filter;
180  int processor = region_addresses->processors[r_id].processor;
181 
182  // store the index in bitfields list where this processors bitfields
183  // start being read in at. (not sorted)
184  processor_heads[processor] = index;
185 
186  // read in the processors bitfields.
187  filter_info_t *filters = filter_region->filters;
188  for (uint32_t bf_id = 0; bf_id < filter_region->n_filters; bf_id++) {
189  // update trackers.
190  if (!filters[bf_id].all_ones) {
191  sorted_bit_fields->processor_ids[index] = processor;
192  sorted_bit_fields->bit_fields[index] = &filters[bf_id];
193  index++;
194  }
195 
196  // also accum the incoming packets from bitfields which have no
197  // redundancy
198  processor_totals[processor] += filters[bf_id].n_atoms;
199  }
200  sort_by_redundancy(sorted_bit_fields, processor_heads[processor], index);
201  }
202 }

209 static inline void bit_field_reader_read_in_bit_fields(
210  region_addresses_t *restrict region_addresses,
211  sorted_bit_fields_t *restrict sorted_bit_fields) {
212  //init data tracking structures
213  for (int i = 0; i < MAX_PROCESSORS; i++) {
214  processor_heads[i] = DO_NOT_USE;
215  processor_totals[i] = 0;
216  }
217 
218  // track positions and incoming packet counts.
219  fills_in_sorted_bit_fields_and_tracker(region_addresses, sorted_bit_fields);
220 
221  //TODO safety code to be removed.
222  for (int i = 0; i < MAX_PROCESSORS; i++) {
223  log_debug("i: %d, head: %d count: %d",
224  i, processor_heads[i], processor_totals[i]);
225  }
226 #if 0
227  print_structs(sorted_bit_fields);
228 #endif
229 
230  // sort bitfields so that bit-fields are in order of most impact on worse
231  // affected cores. so that merged bitfields should reduce packet rates
232  // fairly across cores on the chip.
233  order_bitfields(sorted_bit_fields);
234 
235  //TODO safety code to be removed.
236 #if 0
237  print_structs(sorted_bit_fields);
238 #endif
239 
240  // sort the bitfields by key.
241  // NOTE: does not affect previous sort, as this directly affects the index
242  // in the bitfield array, whereas the previous sort affects the sort index
243  // array.
244  sort_by_key(sorted_bit_fields);
245 
246  // useful for debugging
247 #if 0
248  print_structs(sorted_bit_fields);
249 #endif
250 }

257 static inline sorted_bit_fields_t * bit_field_reader_initialise(
258  region_addresses_t *restrict region_addresses) {
259  sorted_bit_fields_t *restrict sorted_bit_fields = MALLOC_SDRAM(
260  sizeof(sorted_bit_fields_t));
261  if (sorted_bit_fields == NULL) {
262  log_error("failed to allocate DTCM for sorted bitfields.");
263  return NULL;
264  }
265 
266  // figure out how many bitfields we need
267  log_debug("n_processors of addresses = %d", region_addresses->n_processors);
268  int n_bit_fields = 0;
269  for (int r_id = 0; r_id < region_addresses->n_processors; r_id++) {
270  filter_region_t *restrict filter = region_addresses->processors[r_id].filter;
271  uint32_t n_filters = filter->n_filters;
272  filter_info_t *filters = filter->filters;
273  uint32_t n_usable = 0;
274  for (uint32_t f_id = 0; f_id < n_filters; f_id++) {
275  n_usable += !filters[f_id].all_ones;
276  }
277  n_bit_fields += n_usable;
278  log_debug("Core %d has %u bitfields of which %u have redundancy",
279  region_addresses->processors[r_id].processor,
280  filter->n_filters, n_usable);
281  }
282  sorted_bit_fields->n_bit_fields = n_bit_fields;
283  log_info("Number of bitfields with redundancy found is %u",
284  sorted_bit_fields->n_bit_fields);
285 
286  // if there are no bit-fields just return sorted bitfields.
287  if (n_bit_fields != 0) {
288  // malloc the separate bits of the sorted bitfield struct
289  sorted_bit_fields->bit_fields = MALLOC_SDRAM(
290  n_bit_fields * sizeof(filter_info_t*));
291  if (sorted_bit_fields->bit_fields == NULL) {
292  log_error(
293  "cannot allocate memory for the sorted bitfield addresses");
294  FREE(sorted_bit_fields);
295  return NULL;
296  }
297 
298  sorted_bit_fields->processor_ids = MALLOC_SDRAM(
299  n_bit_fields * sizeof(int));
300  if (sorted_bit_fields->processor_ids == NULL) {
301  log_error("cannot allocate memory for the sorted bitfields with "
302  "processors ids");
303  FREE(sorted_bit_fields->bit_fields);
304  FREE(sorted_bit_fields);
305  return NULL;
306  }
307 
308  sorted_bit_fields->sort_order = MALLOC_SDRAM(
309  n_bit_fields * sizeof(int));
310  if (sorted_bit_fields->sort_order == NULL) {
311  log_error("cannot allocate memory for the sorted bitfields with "
312  "sort_order");
313  FREE(sorted_bit_fields->bit_fields);
314  FREE(sorted_bit_fields->processor_ids);
315  FREE(sorted_bit_fields);
316  return NULL;
317  }
318 
319  // init to -1, else random data (used to make prints cleaner)
320  for (int sorted_index = 0; sorted_index < n_bit_fields;
321  sorted_index++) {
322  sorted_bit_fields->sort_order[sorted_index] = FAILED_TO_FIND;
323  }
324  }
325 
326  // return
327  return sorted_bit_fields;
328 }
