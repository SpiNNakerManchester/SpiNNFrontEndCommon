720 static INT_HANDLER reinjection_ready_to_send_callback(void) {
721  // TODO: may need to deal with packet timestamp.
722 
723  // check if router not blocked
724  if (reinjection_can_send_now()) {
725  // access packet queue with FIQ disabled,
726  uint cpsr = cpu_fiq_disable();
727 
728  // if queue not empty bounce packet,
729  if (reinject_pkt_queue.tail != reinject_pkt_queue.head) {
730  // dequeue packet and update queue pointer
731  dumped_packet_t pkt =
732  reinject_pkt_queue.queue[reinject_pkt_queue.head];
733  reinject_pkt_queue.head =
734  (reinject_pkt_queue.head + 1) % PKT_QUEUE_SIZE;

750  reinjection_disable_comms_interrupt();
751  }
752

758 static INT_HANDLER reinjection_dropped_packet_callback(void) {
759  // get packet from router,
760  router_packet_header_t hdr = router_control->dump.header;
761  uint pld = router_control->dump.payload;
762  uint key = router_control->dump.key;
763 
764  // clear dump status and interrupt in router,
765  router_dump_status_t rtr_dstat = router_control->dump.status;
766  router_dump_outputs_t rtr_dump_outputs = router_control->dump.outputs;
767 
768  // only reinject if configured
769 
770  uint packet_type = ((spinnaker_packet_control_byte_t) hdr.control).type;
771  if (((packet_type == SPINNAKER_PACKET_TYPE_MC) && reinject_mc) ||
772  ((packet_type == SPINNAKER_PACKET_TYPE_P2P) && reinject_pp) ||
773  ((packet_type == SPINNAKER_PACKET_TYPE_NN) && reinject_nn) ||
774  ((packet_type == SPINNAKER_PACKET_TYPE_FR) && reinject_fr)) {
775 
776  // check for overflow from router
777  if (rtr_dstat.overflow) {
778  reinject_n_missed_dropped_packets++;
779  } else {
780  // Note that the processor_dump and link_dump flags are sticky
781  // so you can only really count these if you *haven't* missed a
782  // dropped packet - hence this being split out
783 
784  if (rtr_dump_outputs.processor > 0) {
785  // add to the count the number of active bits from this dumped
786  // packet, as this indicates how many processors this packet
787  // was meant to go to.
788  reinject_n_processor_dumped_packets +=
789  __builtin_popcount(rtr_dump_outputs.processor);
790  }
791 
792  if (rtr_dump_outputs.link > 0) {
793  // add to the count the number of active bits from this dumped
794  // packet, as this indicates how many links this packet was
795  // meant to go to.
796  reinject_n_link_dumped_packets +=
797  __builtin_popcount(rtr_dump_outputs.link);
798  }
799  }
800 
801  // Only update this counter if this is a packet to reinject
802  reinject_n_dropped_packets++;
803 
804  // Disable FIQ for queue access
805  uint cpsr = cpu_fiq_disable();
806 
807  // try to insert dumped packet in the queue,
808  uint new_tail = (reinject_pkt_queue.tail + 1) % PKT_QUEUE_SIZE;
809 
810  // check for space in the queue
811  if (new_tail != reinject_pkt_queue.head) {
812  // queue packet,
813  reinject_pkt_queue.queue[reinject_pkt_queue.tail].hdr = hdr;
814  reinject_pkt_queue.queue[reinject_pkt_queue.tail].key = key;
815  reinject_pkt_queue.queue[reinject_pkt_queue.tail].pld = pld;
816 
817  // update queue pointer,
818  reinject_pkt_queue.tail = new_tail;
819  } else {
820  // The queue of packets has overflowed
821  reinject_n_dropped_packet_overflows++;
822  }
823 
824  // restore FIQ after queue access,
825  cpu_int_restore(cpsr);
826  }
827 }

1705 static void data_out_speed_up_command(sdp_msg_pure_data *msg) {

1707

1712  data_out_transaction_id = 0;
1713  publish_transaction_id(data_out_transaction_id);
1714  }
1715

1717  // and worthless
1718  if (message->transaction_id != data_out_transaction_id + 1) {
1719  io_printf(IO_BUF,

1721  "transaction id %d; mine is %d\n",
1722  message->transaction_id, data_out_transaction_id + 1);

1724  }

1728  publish_transaction_id(data_out_transaction_id);
1729 
1730  data_out_stop = false;
1731 
1732  // set SDRAM position and length
1733  data_out_store_address = message->sdram_location;
1734  // state for how many bytes it needs to send, gives approximate
1735  // bandwidth if round number.
1736  uint32_t bytes_to_read_write = message->length;
1737 
1738  data_out_max_seq_num = bytes_to_read_write / SDP_PAYLOAD_BYTES;
1739  uint32_t mod = bytes_to_read_write % SDP_PAYLOAD_BYTES;
1740  data_out_max_seq_num += mod > 0;
1741 
1742  // reset states
1743  data_out_first_transmission = true;
1744  data_out_transmit_dma_pointer = 0;
1745  data_out_position_in_store = 0;
1746  data_out_n_elements_to_read_from_sdram =
1747  bytes_to_read_write / sizeof(uint);
1748 
1749  if (data_out_n_elements_to_read_from_sdram < SDP_PAYLOAD_WORDS) {
1750  data_out_read(DMA_TAG_READ_FOR_TRANSMISSION, 2,
1751  data_out_n_elements_to_read_from_sdram);
1752  } else {
1753  data_out_read(DMA_TAG_READ_FOR_TRANSMISSION, 2, SDP_PAYLOAD_WORDS);
1754  }
1755  return;
1756  }
1757  case SDP_CMD_START_OF_MISSING_SDP_PACKETS:
1758  if (message->transaction_id != data_out_transaction_id) {
1759  io_printf(IO_BUF,
1760  "received data from a different transaction for "
1761  "start of missing. expected %d got %d\n",
1762  data_out_transaction_id, message->transaction_id);
1763  return;
1764  }
1765

1767  if (data_out_n_missing_seq_packets != 0) {
1768  io_printf(IO_BUF, "forcing start of retransmission packet\n");

1770  data_out_missing_seq_num_sdram_address[data_out_n_missing_seq_nums_in_sdram++] =
1771  END_FLAG;
1772  data_out_read_data_position = 0;
1773  data_out_position_for_retransmission = 0;
1774  data_out_in_retransmission_mode = true;
1775  data_out_retransmission_dma_read();

1777  }
1778  // fall through
1779  case SDP_CMD_MORE_MISSING_SDP_PACKETS:
1780  if (message->transaction_id != data_out_transaction_id) {
1781  io_printf(IO_BUF,
1782  "received data from different transaction for more "
1783  "missing; expected %d, got %d\n",
1784  data_out_transaction_id, message->transaction_id);
1785  return;
1786  }
1787 
1788  // reset state, as could be here from multiple attempts
1789  if (!data_out_in_retransmission_mode) {
1790  // put missing sequence numbers into SDRAM
1791  data_out_store_missing_seq_nums(
1792  msg->data,
1793  (msg->length - LENGTH_OF_SDP_HEADER) / sizeof(uint),
1794  message->command == SDP_CMD_START_OF_MISSING_SDP_PACKETS);

1799

1801  data_out_missing_seq_num_sdram_address[
1802  data_out_n_missing_seq_nums_in_sdram++] = END_FLAG;
1803  data_out_read_data_position = 0;
1804  data_out_position_for_retransmission = 0;
1805

1808  data_out_retransmission_dma_read();

1815  io_printf(IO_BUF,
1816  "received data from different transaction for "
1817  "clear; expected %d, got %d\n",
