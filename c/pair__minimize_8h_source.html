127 static inline int compare_routes(uint32_t route_a, uint32_t route_b) {
128  if (route_a == route_b) {
129  return 0;
130  }
131  for (uint i = 0; i < routes_count; i++) {
132  if (routes[i] == route_a) {
133  return -1;
134  }
135  if (routes[i] == route_b) {
136  return 1;
137  }
138  }

140  // set the failed flag and exit
141  malloc_extras_terminate(EXIT_FAIL);
142 
143  return 0;

164 static void sort_routes(void) {
165  uint32_t i, j;
166 
167  for (i = 1; i < routes_count; i++) {
168  // The entry we're going to move is "taken out"
169  uint32_t r_tmp = routes[i];
170  uint32_t rf_tmp = routes_frequency[i];
171 
172  // The entries below it that are larger are shuffled up
173  for (j = i; j > 0 && routes_frequency[j - 1] > rf_tmp; j--) {
174  routes[j] = routes[j - 1];
175  routes_frequency[j] = routes_frequency[j - 1];
176  }
177 
178  // The entry is dropped back into place
179  routes[j] = r_tmp;
180  routes_frequency[j] = rf_tmp;
181  }
182 }

187 static inline bool update_frequency(int index) {
188  uint32_t route = routing_table_get_entry(index)->route;
189  for (uint i = 0; i < routes_count; i++) {
190  if (routes[i] == route) {
191  routes_frequency[i]++;
192  return true;
193  }
194  }
195  routes[routes_count] = route;
196  routes_frequency[routes_count] = 1;
197  routes_count++;
198  if (routes_count >= MAX_NUM_ROUTES) {
199  if (standalone()) {
200  log_error("Too many different routes to compress found %d "
201  "compared to max legal of %d",
202  routes_count, MAX_NUM_ROUTES);
203  }
204  return false;
205  }
206  return true;
207 }
