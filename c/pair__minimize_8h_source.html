109 static inline bool find_merge(int left, int index) {
110  cancel_dmas();
111  const entry_t *entry1 = routing_table_get_entry(left);
112  const entry_t *entry2 = routing_table_get_entry(index);
113  const entry_t merged = merge(entry1, entry2);
114 
115  uint32_t size = routing_table_get_n_entries();
116  uint32_t items_to_go = size - remaining_index;
117  uint32_t next_n_items = transfer_next(remaining_index, items_to_go, 0);
118  uint32_t next_items_to_go = items_to_go - next_n_items;
119  uint32_t next_start = remaining_index + next_n_items;
120  bool dma_in_progress = true;
121  uint32_t read_cache = 0;
122  uint32_t write_cache = 1;
123  while (items_to_go > 0) {
124 
125  // Finish any outstanding transfer
126  if (dma_in_progress) {
127  routing_table_wait_for_last_transfer();
128  dma_in_progress = false;
129  }
130 
131  // Get the details of the last transfer done
132  uint32_t n_items = next_n_items;
133  uint32_t cache = read_cache;
134 
135  // Start the next transfer if needed
136  if (next_items_to_go > 0) {
137  next_n_items = transfer_next(next_start, next_items_to_go, write_cache);
138  next_items_to_go -= next_n_items;
139  next_start += next_n_items;
140  dma_in_progress = true;
141  write_cache = (write_cache + 1) & 0x1;
142  read_cache = (read_cache + 1) & 0x1;
143  }
144 
145  // Check the items now available
146  entry_t *entries = route_cache[cache];
147  for (uint32_t i = 0; i < n_items; i++) {
148  if (key_mask_intersect(entries[i].key_mask, merged.key_mask)) {
149  if (dma_in_progress) {
150  routing_table_wait_for_last_transfer();
151  }
152  return false;
153  }
154  }
155 
156  items_to_go = next_items_to_go;
157  }
158  routing_table_put_entry(&merged, left);
159  return true;
160 }

165 static inline void compress_by_route(int left, int right) {
166  while (left < right) {
167  bool merged = false;
168 
169  for (int index = left + 1; index <= right; index++) {
170  merged = find_merge(left, index);
171  if (merged) {
172  routing_table_copy_entry(index, right--);
173  break;
174  }
175  }
176  if (!merged) {
177  routing_table_copy_entry(write_index++, left++);
178  }
179  }
180  if (left == right) {
181  routing_table_copy_entry(write_index++, left);
182  }
183 }

187 static void sort_routes(void) {
188  uint32_t i, j;
189 
190  for (i = 1; i < routes_count; i++) {
191  // The entry we're going to move is "taken out"
192  uint32_t r_tmp = routes[i];
193  uint32_t rf_tmp = routes_frequency[i];
194 
195  // The entries below it that are larger are shuffled up
196  for (j = i; j > 0 && routes_frequency[j - 1] > rf_tmp; j--) {
197  routes[j] = routes[j - 1];
198  routes_frequency[j] = routes_frequency[j - 1];
199  }
200 
201  // The entry is dropped back into place
202  routes[j] = r_tmp;
203  routes_frequency[j] = rf_tmp;
204  }
205 }

210 static inline bool update_frequency(int index) {
211  uint32_t route = routing_table_get_entry(index)->route;
212  for (uint i = 0; i < routes_count; i++) {
213  if (routes[i] == route) {
214  routes_frequency[i]++;
215  return true;
216  }
217  }
218  routes[routes_count] = route;
219  routes_frequency[routes_count] = 1;
220  routes_count++;
221  if (routes_count >= MAX_NUM_ROUTES) {
222  if (standalone()) {
223  log_error("Too many different routes to compress found %d "
224  "compared to max legal of %d",
225  routes_count, MAX_NUM_ROUTES);
226  }
227  return false;
228  }
229  return true;
230 }

249 void sort_table(void) {
250  if (routes_count == 0) {
251  return;
252  }
253 
254  // Set up a pointer for each of the routes
255  uint16_t route_offset[routes_count];
256  uint16_t route_end[routes_count];
257  uint32_t offset = 0;
258  for (uint32_t i = 0; i < routes_count; i++) {
259  route_offset[i] = offset;
260  offset += routes_frequency[i];
261  route_end[i] = offset - 1;
262  }
263 
264  // Go through and move things into position
265  uint32_t pos = 0;
266  uint32_t pos_index = 0;
267  uint32_t next_index_offset = routes_frequency[0];
268  uint32_t n_entries = routing_table_get_n_entries();
269  while (pos < n_entries) {
270  // Get the entry
271  entry_t entry = *routing_table_get_entry(pos);
272 
273  // Where does the route need to go
274  uint32_t route_index = find_route_index(entry.route);
275 
276  // Where are we reading from?
277  uint32_t read_index = pos_index;
278 
279  // If we are in the right region
280  bool skip = false;
281  if (route_index == read_index) {
282  // If we already put this item here, don't move it
283  if (pos < route_offset[route_index]) {
284  skip = true;
285  }
286  }
287 
288  // Keep swapping things until they are in the right place
289  while (!skip) {
290 
291  // Find the place to put the route in its group
292  uint32_t new_pos = route_offset[route_index];
293  if (new_pos >= n_entries) {
294  log_error("New table position %u out of range!", new_pos);
295  rt_error(RTE_SWERR);
296  }
297 
298  if (new_pos > route_end[route_index]) {
299  log_error("New table position %u of region %u is out of range!",
300  new_pos, route_index);
301  rt_error(RTE_SWERR);
302  }
303  route_offset[route_index] += 1;
304 
305  // Swap out the existing entry with the new one
306  entry_t old_entry = *routing_table_get_entry(new_pos);
307  routing_table_put_entry(&entry, new_pos);
308 
309  // We can quit because we should have moved this one
310  if (new_pos <= pos) {
311  break;
312  }
313  entry = old_entry;
314  read_index = route_index;
315 
316  // Find the index of the item we swapped out so it can be swapped next
317  route_index = find_route_index(entry.route);
318  }
319 
320  // Where are we next
321  pos++;
322  if (pos == next_index_offset) {
323  pos_index += 1;
324  next_index_offset += routes_frequency[pos_index];
325  }
326  }
327 }

354  }

370  }
371

374  tc[T2_CONTROL] = 0x83;
375  sort_table();

377  tc[T2_CONTROL] = 0;

381  return false;

383 
384  write_index = 0;

386  int left = 0;
387 
388  while (left <= max_index) {

391  log_debug("A %u %u %u %u", left, max_index, right, left_route);
392  while ((right < table_size - 1) &&
393  routing_table_get_entry(right+1)->route ==

415  return false;

418  }
