154 static inline void recording_host_data_read(const eieio_msg_t msg) {
155  const host_data_read_packet_header *ptr_hdr =
156  (const host_data_read_packet_header *) msg;
157  uint8_t n_requests = ptr_hdr->request;
158  uint8_t sequence = ptr_hdr->sequence;
159  const host_data_read_packet_data *ptr_data =
160  (const host_data_read_packet_data *) &ptr_hdr[1];
161 
162  if (sequence != sequence_number) {
163  log_debug("dropping packet with sequence no: %d", sequence);
164  return;
165  }
166  sequence_number = (sequence_number + 1) & MAX_SEQUENCE_NO;
167  sequence_ack = false;
168 
169  uint32_t i;
170  for (i = 0; i < n_requests; i++) {
171  uint8_t channel = ptr_data[i].channel;
172  uint32_t space_read = ptr_data[i].space_read;
173 
174  uint32_t temp_value = (uint32_t) (
175  g_recording_channels[channel].current_read + space_read);
176 
177  log_debug("channel %d, updating read pointer by %d bytes, from 0x%08x",
178  channel, space_read, g_recording_channels[channel].current_read);
179  if (temp_value >= (uint32_t) g_recording_channels[channel].end) {
180  uint32_t channel_space_total = (uint32_t) (
181  g_recording_channels[channel].end -
182  g_recording_channels[channel].start);
183  temp_value = temp_value - channel_space_total;
184  log_debug("channel %d, read wrap around", channel);
185  }
186 
187  g_recording_channels[channel].current_read = (uint8_t *) temp_value;
188  g_recording_channels[channel].last_buffer_operation =
189  BUFFER_OPERATION_READ;
190  }
191 }

217 static inline void recording_eieio_packet_handler(
218  const eieio_msg_t msg, uint length) {
219  uint16_t data_hdr_value = msg[0];
220  uint8_t pkt_type = (data_hdr_value >> 14) && 0x03;
221  uint16_t pkt_command = data_hdr_value & (~0xC000);
222 
223  log_debug("received packet of type %d", pkt_type);
224 
225  if (pkt_type == 0x01) {
226  log_debug("recording - parsing a command packet");
227  switch (pkt_command) {
228  case HOST_DATA_READ:
229  log_debug("command: HOST_DATA_READ");
230  recording_host_data_read(msg);
231  break;
232 
233  case HOST_DATA_READ_ACK:
234  log_debug("command: HOST_DATA_READ_ACK");
235  recording_host_data_read_ack(msg);
236  break;
237 
238  default:
239  log_debug("unhandled command ID %d", pkt_command);
240  break;
241  }
242  }
243 
244  log_debug("leaving packet handler");
245 }

250 static uint32_t compute_available_space_in_channel(uint8_t channel) {
251  uint8_t *buffer_region = g_recording_channels[channel].start;
252  uint8_t *end_of_buffer_region = g_recording_channels[channel].end;
253  uint8_t *write_pointer = g_recording_channels[channel].current_write;
254  uint8_t *read_pointer = g_recording_channels[channel].current_read;
255  buffered_operations last_buffer_operation =
256  g_recording_channels[channel].last_buffer_operation;
257 
258  if (read_pointer < write_pointer) {
259  uint32_t final_space =
260  (uint32_t) end_of_buffer_region - (uint32_t) write_pointer;
261  uint32_t initial_space =
262  (uint32_t) read_pointer - (uint32_t) buffer_region;
263  return final_space + initial_space;
264  } else if (write_pointer < read_pointer) {
265  return (uint32_t) read_pointer - (uint32_t) write_pointer;
266  } else if (last_buffer_operation == BUFFER_OPERATION_WRITE) {
267  // If pointers are equal, buffer is full if last operation is write
268  return 0;
269  } else {
270  // If pointers are equal, buffer is empty if last operation is read
271  return end_of_buffer_region - buffer_region;
272  }
273 }

286 static void recording_write_one_chunk(
287  uint8_t channel, void *data, void *write_pointer, uint32_t length,
288  void *finished_write_pointer, recording_complete_callback_t callback) {
289  if (callback != NULL) {
290  // add to DMA complete tracker
291  circular_buffer_add(dma_complete_buffer, (uint32_t) channel);
292  circular_buffer_add(
293  dma_complete_buffer, (uint32_t) finished_write_pointer);
294  circular_buffer_add(dma_complete_buffer, (uint32_t) callback);
295 
296  // set off DMA - if not accepted, wait until another DMA is done
297  while (!spin1_dma_transfer(
298  RECORDING_DMA_COMPLETE_TAG_ID, write_pointer, data, DMA_WRITE,
299  length)) {
300  wait_for_interrupt();
301  }
302  } else {
303  spin1_memcpy(write_pointer, data, length);
304  g_recording_channels[(uint8_t) channel].dma_current_write =
305  (uint8_t *) finished_write_pointer;
306  }

318 static inline bool recording_write_memory(
319  uint8_t channel, void *data, uint32_t length,
320  recording_complete_callback_t callback) {
321  uint8_t *buffer_region = g_recording_channels[channel].start;
322  uint8_t *end_of_buffer_region = g_recording_channels[channel].end;
323  uint8_t *write_pointer = g_recording_channels[channel].current_write;
324  uint8_t *read_pointer = g_recording_channels[channel].current_read;
325  buffered_operations last_buffer_operation =
326  g_recording_channels[channel].last_buffer_operation;
327 
328  log_debug("t = %u, channel = %u, start = 0x%08x, read = 0x%08x,"
329  "write = 0x%08x, end = 0x%08x, operation == read = %u, len = %u",
330  spin1_get_simulation_time(), channel, buffer_region,
331  read_pointer, write_pointer, end_of_buffer_region,
332  last_buffer_operation == BUFFER_OPERATION_READ, length);
333 
334  if ((read_pointer < write_pointer) ||
335  (read_pointer == write_pointer &&
336  last_buffer_operation == BUFFER_OPERATION_READ)) {
337  uint32_t final_space =
338  (uint32_t) end_of_buffer_region - (uint32_t) write_pointer;
339 
340  if (final_space >= length) {
341  log_debug("Packet fits in final space of %u", final_space);
342  recording_write_one_chunk(channel, data, write_pointer, length,
343  write_pointer + length, callback);
344  write_pointer += length;
345  } else {
346  uint32_t total_space = final_space +
347  ((uint32_t) read_pointer - (uint32_t) buffer_region);
348  if (total_space < length) {
349  log_debug("Not enough space in final area (%u bytes)",
350  total_space);
351  return false;
352  }
353 
354  log_debug("Copying first %d bytes to final space of %u",
355  length, final_space);
356 
357  recording_write_one_chunk(channel, data, write_pointer, final_space,
358  buffer_region, NULL);
359 
360  write_pointer = buffer_region;
361  data += final_space;
362 
363  uint32_t final_len = length - final_space;
364  log_debug("Copying remaining %u bytes", final_len);
365 
366  recording_write_one_chunk(channel, data, write_pointer, final_len,
367  write_pointer + final_len, callback);
368 
369  write_pointer += final_len;
370  }
371  } else if (write_pointer < read_pointer) {
372  uint32_t middle_space =
373  (uint32_t) read_pointer - (uint32_t) write_pointer;
374 
375  if (middle_space < length) {
376  log_debug("Not enough space in middle (%u bytes)", middle_space);
377  return false;
378  }
379 
380  log_debug("Packet fits in middle space of %u", middle_space);
381  recording_write_one_chunk(channel, data, write_pointer, length,
382  write_pointer + length, callback);
383  write_pointer += length;
384  } else {
385  log_debug("reached end");
386 
387  log_debug("Buffer already full");
388  return false;
389  }
390 
391  if (write_pointer == end_of_buffer_region) {
392  write_pointer = buffer_region;
393  log_debug("channel %u, write wrap around", channel);
394  }
395  g_recording_channels[channel].current_write = write_pointer;
396  g_recording_channels[channel].last_buffer_operation =
397  BUFFER_OPERATION_WRITE;
398  return true;
399 }

421 static inline void recording_send_buffering_out_trigger_message(
422  bool flush_all) {
423  uint msg_size = 16 + sizeof(read_request_packet_header);
424  uint n_requests = 0;
425 
426  for (uint channel = 0; channel < n_recording_regions; channel++) {
427  uint32_t space_total = (uint32_t) (
428  g_recording_channels[channel].end -
429  g_recording_channels[channel].start);
430  uint32_t space_available = compute_available_space_in_channel(channel);
431 
432  if (has_been_initialised(channel) && (flush_all ||
433  space_total - space_available >= buffer_size_before_trigger)) {
434  uint8_t *buffer_region = g_recording_channels[channel].start;
435  uint8_t *end_of_buffer_region = g_recording_channels[channel].end;
436  uint8_t *write_pointer =
437  g_recording_channels[channel].dma_current_write;
438  uint8_t *read_pointer = g_recording_channels[channel].current_read;
439  buffered_operations last_buffer_operation =
440  g_recording_channels[channel].last_buffer_operation;
441 
442  if (read_pointer < write_pointer) {
443  create_buffer_message(
444  &read_request_data[n_requests++], channel,
445  read_pointer, write_pointer - read_pointer);
446  } else if ((write_pointer < read_pointer) || (
447  write_pointer == read_pointer &&
448  last_buffer_operation == BUFFER_OPERATION_WRITE)) {
449  create_buffer_message(
450  &read_request_data[n_requests++], channel,
451  read_pointer, end_of_buffer_region - read_pointer);
452  create_buffer_message(
453  &read_request_data[n_requests++], channel,
454  buffer_region, write_pointer - buffer_region);
455  } else {
456  /* something somewhere went terribly wrong this should never
457  * happen */
458  log_error("Unknown channel state - channel: %d, start pointer: %d,"
459  " end pointer: %d, read_pointer: %d, write_pointer: %d,"
460  " last operation==READ: %d",
461  channel, buffer_region, end_of_buffer_region,
462  read_pointer, write_pointer,
463  last_buffer_operation == BUFFER_OPERATION_READ);
464  }
465  }
466  }
467 
468  if (n_requests > 0) {
469  // eieio command packet with command ID 8
470  read_request_hdr->eieio_header_command = 0x4008;
471  read_request_hdr->chip_id = spin1_get_chip_id();
472  read_request_data[0].processor_and_request =
473  (spin1_get_core_id() << 3) | n_requests;
474  read_request_data[0].sequence = sequence_number;
475  log_debug("Sending request with sequence %d", sequence_number);
476  msg_size += (n_requests * sizeof(read_request_packet_data));
477  read_request_msg.length = msg_size;
478 
479  spin1_send_sdp_msg(&read_request_msg, 1);
480  }
481 }

501 bool recording_do_record_and_notify(
502  uint8_t channel, void *data, uint32_t size_bytes,
503  recording_complete_callback_t callback) {
504  if (has_been_initialised(channel)) {
505  recording_channel_t *recording_channel = &g_recording_channels[channel];
506  uint32_t space_available = compute_available_space_in_channel(channel);
507 
508  // If there's space to record
509  if (space_available >= size_bytes) {
510  // Copy data into recording channel
511  recording_write_memory(channel, data, size_bytes, callback);
512  return true;
513  }
514 
515  if (!g_recording_channels[channel].missing_info) {
516  log_info("WARNING: recording channel %u out of space", channel);
517  g_recording_channels[channel].missing_info = 1;
518  }
519  }
520 
521  // Call the callback to make sure resources are freed
522  if (callback != NULL) {
523  callback();
524  }
525  return false;
526 }

556 void recording_finalise(void) {
557  uint8_t i;
558 
559  log_debug("Finalising recording channels");
560 
561  // wait till all DMA's have been finished
562  while (circular_buffer_size(dma_complete_buffer) != 0) {
563  wait_for_interrupt();
564  }
565 
566  // update buffer state data write
567  recording_buffer_state_data_write();
568 
569  // Loop through channels
570  for (uint32_t channel = 0; channel < n_recording_regions; channel++) {
571  // If this channel's in use
572  if (has_been_initialised(channel)) {
573  recording_channel_t *recording_channel =
574  &g_recording_channels[channel];
575 
576  /* Calculate the number of bytes that have been written and write

578  if (g_recording_channels[channel].missing_info) {
579  log_info("\tFinalising channel %u - dropped information while"

618 bool recording_initialize(
619  void **recording_data_address, uint32_t *recording_flags) {
620  // Get the data and number of recording regions
621  recording_data_t *recording_data = *recording_data_address;
622  n_recording_regions = recording_data->n_regions;
623 
624  // Recording channel pointers are after the recording data (initially NULL)
625  recording_channel_t **sdram_region_ptrs =
626  (recording_channel_t **) &recording_data[1];
627 
628  // Sizes are after the channel pointers
629  uint32_t *sdram_region_sizes =
630  (uint32_t *) &sdram_region_ptrs[n_recording_regions];
631 
632  // Update the pointer to after the data
633  *recording_data_address = &sdram_region_sizes[n_recording_regions];
634 
635  // build DMA address circular queue
636  dma_complete_buffer = circular_buffer_initialize(DMA_QUEUE_SIZE * 4);
637 
638  // Read in the parameters
639  uint8_t buffering_output_tag = recording_data->tag;
640  uint32_t buffering_destination = recording_data->tag_destination;
641  sdp_port = recording_data->sdp_port;
642  buffer_size_before_trigger = recording_data->buffer_size_before_request;
643  time_between_triggers = recording_data->time_between_triggers;
644  if (time_between_triggers < MIN_TIME_BETWEEN_TRIGGERS) {
645  time_between_triggers = MIN_TIME_BETWEEN_TRIGGERS;
646  }
647  last_sequence_number = &(recording_data->last_sequence_number);
648 
649  log_info("Recording %d regions, using output tag %d, size before trigger %d, "
650  "time between triggers %d",
651  n_recording_regions, buffering_output_tag, buffer_size_before_trigger,
652  time_between_triggers);
653 
654  // Set up the space for holding recording pointers and sizes
655  region_addresses = spin1_malloc(
656  n_recording_regions * sizeof(recording_channel_t *));
657  if (region_addresses == NULL) {
658  log_error("Not enough space to allocate recording addresses");
659  return false;
660  }
661  region_sizes = spin1_malloc(n_recording_regions * sizeof(uint32_t));
662  if (region_sizes == NULL) {
663  log_error("Not enough space to allocate region sizes");
664  return false;
665  }
666 
667  // Set up the recording flags
668  if (recording_flags != NULL) {
669  *recording_flags = 0;
670  }
671 
672  /* Reserve the actual recording regions.
673  *
674  * An extra sizeof(recording_channel_t) bytes are reserved per channel
675  * to store the data after recording */
676  for (uint32_t counter = 0; counter < n_recording_regions; counter++) {
677  uint32_t size = sdram_region_sizes[counter];
678  if (size > 0) {
679  region_sizes[counter] = size;
680  region_addresses[counter] = sark_xalloc(
681  sv->sdram_heap, size + sizeof(recording_channel_t), 0,
682  ALLOC_LOCK + ALLOC_ID + (sark_vec->app_id << 8));
683  if (region_addresses[counter] == NULL) {
684  log_error("Could not allocate recording region %u of %u bytes",
685  counter, size);
686  return false;
687  }
688  sdram_region_ptrs[counter] = region_addresses[counter];
689  log_info("recording address is %x", region_addresses[counter]);
690  *recording_flags = (*recording_flags | (1 << counter));
691  } else {
692  sdram_region_ptrs[counter] = NULL;
693  region_addresses[counter] = 0;
694  region_sizes[counter] = 0;
695  }
696  }
697  g_recording_channels =
698  spin1_malloc(n_recording_regions * sizeof(recording_channel_t));
699  if (!g_recording_channels) {
700  log_error("Not enough space to create recording channels");
701  return false;
702  }
703  log_debug("Allocated recording channels to 0x%08x", g_recording_channels);
704 
705  // Set up the channels and write the initial state data
706  recording_reset();
707 
708  // Set up the buffer message
709  read_request_hdr = (read_request_packet_header *) &read_request_msg.cmd_rc;
710  read_request_data = (read_request_packet_data *) &read_request_hdr[1];
711  read_request_msg.flags = 0x7;
712  read_request_msg.tag = buffering_output_tag;
713  read_request_msg.dest_port = 0xFF;

728 void recording_reset(void) {
729  // Go through the regions and set up the data
730  for (uint32_t i = 0; i < n_recording_regions; i++) {
731  uint32_t region_size = region_sizes[i];
732  log_debug("region size %d", region_size);
733  if (region_size > 0) {
734  recording_channel_t *region_ptr = region_addresses[i];
735 
736  log_debug("%d is size of buffer state in words",
737  sizeof(recording_channel_t) / sizeof(address_t));
738 
739  uint8_t *region_data_address = (uint8_t *) &region_ptr[1];
740 
741  // store pointers to the start, current position and end of this
742  g_recording_channels[i].start = region_data_address;
743  g_recording_channels[i].current_write = region_data_address;
744  g_recording_channels[i].dma_current_write = region_data_address;
745  g_recording_channels[i].current_read = region_data_address;
746  g_recording_channels[i].end = region_data_address + region_size;
747  g_recording_channels[i].last_buffer_operation =
748  BUFFER_OPERATION_READ;
749  g_recording_channels[i].region_id = i;
750  g_recording_channels[i].missing_info = 0;
751 
752  log_info("Recording channel %u configured to use %u byte memory block"
753  " starting at 0x%08x",
754  i, region_size, g_recording_channels[i].start);
755  } else {
756  g_recording_channels[i].start = NULL;
757  g_recording_channels[i].current_write = NULL;
758  g_recording_channels[i].dma_current_write = NULL;
759  g_recording_channels[i].current_read = NULL;
760  g_recording_channels[i].end = NULL;
761  g_recording_channels[i].last_buffer_operation =
762  BUFFER_OPERATION_READ;
763  g_recording_channels[i].region_id = i;
764  g_recording_channels[i].missing_info = 0;
765 
766  log_info("Recording channel %u left uninitialised", i);
767  }
768  }
769  recording_buffer_state_data_write();
770  sequence_number = 0;
771  sequence_ack = false;
