40 void compress_start(UNUSED uint unused0, UNUSED uint unused1) {
41  log_info("Starting on chip router compressor");
42 
43  // Prepare to minimise the routing tables
44  log_debug("looking for header using tag %u app_id %u", 1, sark_app_id());
45  header_t *header = (header_t *) sark_tag_ptr(1, sark_app_id());
46  log_debug("reading data from 0x%08x", (uint32_t) header);
47  print_header(header);
48 
49  // set the flag to something non-useful
50  sark.vcpu->user0 = 20;
51 
52  // Load the routing table
53  log_debug("start reading table");
54  read_table(header);
55  log_debug("finished reading table");
56 
57  // Store intermediate sizes for later reporting (if we fail to minimise)
58  uint32_t size_original = routing_table_get_n_entries();
59 
60  // Currently not used here but used by the bitfeild stuff
61  bool failed_by_malloc = false;
62  // Currently not used here but used by the bitfeild stuff
63  bool stop_compressing = false;
64  if (run_compressor(header->compress_as_much_as_possible,
65  &failed_by_malloc, &stop_compressing)) {
66  // report size to the host for provenance aspects
67  log_info("Compressed the router table from %d to %d entries",
68  size_original, routing_table_get_n_entries());
69  } else {
70  log_info("Exiting as compressor reported failure");
71  // set the failed flag and exit
72  malloc_extras_terminate(EXIT_FAIL);
73 
74  }
75  // Try to load the routing table
76  log_debug("try loading tables");
77  if (load_routing_table(header->app_id)) {
78  cleanup_and_exit(header);
79  } else {
80  // Otherwise give up and exit with an error
81  log_error("Failed to minimise routing table to fit %u entries. "
82  "(Original table: %u after compression: %u).",
83  rtr_alloc_max(), size_original,
84  routing_table_get_n_entries());
85 
86  // Free the block of SDRAM used to load the routing table.
87  log_debug("free sdram blocks which held router tables");
88  FREE((void *) header);
