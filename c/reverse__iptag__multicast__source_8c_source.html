272

275  case EVENT_PADDING:

298 static inline uint16_t calculate_eieio_packet_event_size(
299  const eieio_msg_t eieio_msg_ptr) {
300  uint16_t data_hdr_value = eieio_msg_ptr[0];
301  bool pkt_apply_prefix = (bool) BITS(data_hdr_value, APPLY_PREFIX, 0x1);
302  bool pkt_payload_prefix_apply = (bool)
303  BITS(data_hdr_value, APPLY_PAYLOAD_PREFIX, 0x1);
304  uint8_t pkt_type = (uint8_t) BITS(data_hdr_value, PACKET_TYPE, 0x3);
305  uint8_t event_count = BITS(data_hdr_value, COUNT, 0xFF);
306  uint16_t event_size, total_size;
307  uint16_t header_size = 2;
308 
309  switch (pkt_type) {
310  case KEY_16_BIT:
311  event_size = 2;
312  break;
313  case KEY_PAYLOAD_16_BIT:
314  case KEY_32_BIT:
315  event_size = 4;
316  break;
317  case KEY_PAYLOAD_32_bIT:
318  event_size = 8;
319  break;
320  }
321 
322  if (pkt_apply_prefix) {
323  header_size += 2;
324  }
325  if (pkt_payload_prefix_apply) {
326  if (pkt_type == 0 || pkt_type == 1) {
327  header_size += 2;
328  } else {
329  header_size += 4;
330  }
331  }
332 
333  total_size = event_count * event_size + header_size;
334  return total_size;
335 }

440 static inline uint32_t extract_time_from_eieio_msg(
441  const eieio_msg_t eieio_msg_ptr) {
442  uint16_t data_hdr_value = eieio_msg_ptr[0];
443  bool pkt_has_timestamp = (bool)
444  BITS(data_hdr_value, PAYLOAD_IS_TIMESTAMP, 0x1);
445  bool pkt_apply_prefix = (bool) BITS(data_hdr_value, APPLY_PREFIX, 0x1);
446  bool pkt_class = (bool) BITS(data_hdr_value, PACKET_CLASS, 0x1);
447 
448  // If the packet is actually a command packet, return the current time
449  if (!pkt_apply_prefix && pkt_class) {
450  return time;
451  }
452 
453  // If the packet indicates that payloads are timestamps
454  if (pkt_has_timestamp) {
455  bool pkt_payload_prefix_apply = (bool)
456  BITS(data_hdr_value, APPLY_PAYLOAD_PREFIX, 0x1);
457  uint8_t pkt_type = (uint8_t) BITS(data_hdr_value, PACKET_TYPE, 0x3);
458  uint32_t payload_time = 0;
459  bool got_payload_time = false;
460  uint16_t *event_ptr = &eieio_msg_ptr[1];
461 
462  // If there is a payload prefix
463  if (pkt_payload_prefix_apply) {
464  // If there is a key prefix, the payload prefix is after that
465  if (pkt_apply_prefix) {
466  event_ptr++;
467  }
468 
469  if (pkt_type & 0x2) {
470  // 32 bit packet
471  payload_time = (event_ptr[1] << 16) | event_ptr[0];
472  event_ptr += 2;
473  } else {
474  // 16 bit packet
475  payload_time = event_ptr[0];
476  event_ptr++;
477  }
478  got_payload_time = true;
479  }
480 
481  // If the packets have a payload
482  if (pkt_type & 0x1) {
483  if (pkt_type & 0x2) {
484  // 32 bit packet
485  payload_time |= (event_ptr[1] << 16) | event_ptr[0];
486  } else {
487  // 16 bit packet
488  payload_time |= event_ptr[0];
489  }
490  got_payload_time = true;
491  }
492 
493  // If no actual time was found, return the current time
494  if (!got_payload_time) {
495  return time;
496  }
497  return payload_time;
498  }
499 
500  // This is not a timed packet, return the current time
501  return time;
502 }

509 static inline bool add_eieio_packet_to_sdram(
510  const eieio_msg_t eieio_msg_ptr, uint32_t length) {
511  uint8_t *msg_ptr = (uint8_t *) eieio_msg_ptr;
512 
513  log_debug("read_pointer = 0x%.8x, write_pointer= = 0x%.8x,"
514  "last_buffer_operation == read = %d, packet length = %d",
515  read_pointer, write_pointer,
516  last_buffer_operation == BUFFER_OPERATION_READ, length);
517  if ((read_pointer < write_pointer) ||
518  (read_pointer == write_pointer &&
519  last_buffer_operation == BUFFER_OPERATION_READ)) {
520  uint32_t final_space =
521  (uint32_t) end_of_buffer_region - (uint32_t) write_pointer;
522 
523  if (final_space >= length) {
524  log_debug("Packet fits in final space of %d", final_space);
525 
526  spin1_memcpy(write_pointer, msg_ptr, length);
527  write_pointer += length;
528  last_buffer_operation = BUFFER_OPERATION_WRITE;
529  if (write_pointer >= end_of_buffer_region) {
530  write_pointer = buffer_region;
531  }
532  return true;
533  } else {
534  uint32_t total_space = final_space +
535  ((uint32_t) read_pointer - (uint32_t) buffer_region);
536  if (total_space < length) {
537  log_debug("Not enough space (%d bytes)", total_space);
538  return false;
539  }
540 
541  log_debug("Copying first %d bytes to final space of %d",
542  length, final_space);
543  spin1_memcpy(write_pointer, msg_ptr, final_space);
544  write_pointer = buffer_region;
545  msg_ptr += final_space;
546 
547  uint32_t final_len = length - final_space;
548  log_debug("Copying remaining %d bytes", final_len);
549  spin1_memcpy(write_pointer, msg_ptr, final_len);
550  write_pointer += final_len;
551  last_buffer_operation = BUFFER_OPERATION_WRITE;
552  if (write_pointer == end_of_buffer_region) {
553  write_pointer = buffer_region;
554  }
555  return true;
556  }
557  } else if (write_pointer < read_pointer) {
558  uint32_t middle_space =
559  (uint32_t) read_pointer - (uint32_t) write_pointer;
560  if (middle_space < length) {
561  log_debug("Not enough space in middle (%d bytes)", middle_space);
562  return false;
563  }
564 
565  log_debug("Packet fits in middle space of %d", middle_space);
566  spin1_memcpy(write_pointer, msg_ptr, length);
567  write_pointer += length;
568  last_buffer_operation = BUFFER_OPERATION_WRITE;
569  if (write_pointer == end_of_buffer_region) {
570  write_pointer = buffer_region;
571  }
572  return true;
573  }
574 
575  log_debug("Buffer already full");
576  return false;
577 }

588 static inline void process_16_bit_packets(
589  const uint16_t* event_pointer, bool pkt_prefix_upper,
590  uint32_t pkt_count,
591  uint32_t pkt_key_prefix, uint32_t pkt_payload_prefix,
592  bool pkt_has_payload, bool pkt_payload_is_timestamp) {
593  log_debug("process_16_bit_packets");
594  log_debug("event_pointer: %08x", (uint32_t) event_pointer);
595  log_debug("count: %d", pkt_count);
596  log_debug("pkt_prefix: %08x", pkt_key_prefix);
597  log_debug("pkt_payload_prefix: %08x", pkt_payload_prefix);
598  log_debug("payload on: %d", pkt_has_payload);
599  log_debug("pkt_format: %d", pkt_prefix_upper);
600 
601  for (uint32_t i = 0; i < pkt_count; i++) {
602  uint32_t key = (uint32_t) event_pointer[0];
603  log_debug("Packet key = %d", key);
604  event_pointer++;
605  uint32_t payload = 0;
606  if (pkt_has_payload) {
607  payload = (uint32_t) event_pointer[0];
608  event_pointer++;
609  }
610 
611  if (!pkt_prefix_upper) {
612  key <<= 16;
613  }
614  key |= pkt_key_prefix;
615  payload |= pkt_payload_prefix;
616 
617  log_debug("check before send packet: check_key_in_space=%d, key=0x%08x,"
618  " mask=0x%08x, key_space=%d: %d",
619  check_key_in_space, key, mask, key_space,
620  (!check_key_in_space || (key & mask) == key_space));
621 
622  if (has_key) {
623  if (!check_key_in_space || (key & mask) == key_space) {
624  provenance.sent_packets++;
625  if (pkt_has_payload && !pkt_payload_is_timestamp) {
626  log_debug("mc packet 16-bit key=%d, payload=%d",
627  key, payload);
628  while (!spin1_send_mc_packet(key, payload, WITH_PAYLOAD)) {
629  spin1_delay_us(1);
630  }
631  } else {
632  log_debug("mc packet 16-bit key=%d", key);
633  while (!spin1_send_mc_packet(key, 0, NO_PAYLOAD)) {
634  spin1_delay_us(1);
635  }
636  }
637  } else {
638  provenance.incorrect_keys++;
639  }
640  }
641  }

953 static void fetch_and_process_packet(void) {
954  uint32_t last_len = 2;
955 
956  log_debug("in fetch_and_process_packet");
957  msg_from_sdram_in_use = false;
958 
959  // If we are not buffering, there is nothing to do
960  log_debug("buffer size is %d", buffer_region_size);
961  if (buffer_region_size == 0) {
962  return;
963  }
964 
965  log_debug("dealing with SDRAM is set to %d", msg_from_sdram_in_use);
966  log_debug("has_eieio_packet_in_buffer set to %d",
967  is_eieio_packet_in_buffer());
968  while (!msg_from_sdram_in_use && is_eieio_packet_in_buffer() &&
969  (last_len > 0)) {
970  // If there is padding, move on 2 bytes
971  uint16_t next_header = (uint16_t) *read_pointer;
972  if (next_header == 0x4002) {
973  read_pointer += 2;
974  if (read_pointer >= end_of_buffer_region) {
975  read_pointer = buffer_region;
976  }
977  } else {
978  uint8_t *src_ptr = (uint8_t *) read_pointer;
979  uint8_t *dst_ptr = (uint8_t *) msg_from_sdram;
980  uint32_t len = calculate_eieio_packet_size(
981  (eieio_msg_t) read_pointer);
982 
983  last_len = len;
984  if (len > MAX_PACKET_SIZE) {
985  log_error("Packet from SDRAM at 0x%08x of %u bytes is too big!",
986  src_ptr, len);
987  rt_error(RTE_SWERR);
988  }
989  uint32_t final_space = (end_of_buffer_region - read_pointer);
990 
991  log_debug("packet with length %d, from address: %08x", len,
992  read_pointer);
993 
994  if (len > final_space) {
995  // If the packet is split, get the bits
996  log_debug("split packet");
997  log_debug("1 - reading packet to %08x from %08x length: %d",
998  (uint32_t) dst_ptr, (uint32_t) src_ptr, final_space);
999  spin1_memcpy(dst_ptr, src_ptr, final_space);
1000 
1001  uint32_t remaining_len = len - final_space;
1002  dst_ptr += final_space;
1003  src_ptr = buffer_region;
1004  log_debug("2 - reading packet to %08x from %08x length: %d",
1005  (uint32_t) dst_ptr, (uint32_t) src_ptr, remaining_len);
1006 
1007  spin1_memcpy(dst_ptr, src_ptr, remaining_len);
1008  read_pointer = buffer_region + remaining_len;
1009  } else {
1010  // If the packet is whole, get the packet
1011  log_debug("full packet");
1012  log_debug("1 - reading packet to %08x from %08x length: %d",
1013  (uint32_t) dst_ptr, (uint32_t) src_ptr, len);
1014 
1015  spin1_memcpy(dst_ptr, src_ptr, len);
1016  read_pointer += len;
1017  if (read_pointer >= end_of_buffer_region) {
1018  read_pointer = buffer_region;
1019  }
1020  }
1021 
1022  last_buffer_operation = BUFFER_OPERATION_READ;
1023 
1024  print_packet_bytes(msg_from_sdram, len);
1025  next_buffer_time = extract_time_from_eieio_msg(msg_from_sdram);
1026  log_debug("packet time: %d, current time: %d",
1027  next_buffer_time, time);
1028 
1029  if (next_buffer_time <= time) {
1030  packet_handler_selector(msg_from_sdram, len);
1031  } else {
1032  msg_from_sdram_in_use = true;
1033  msg_from_sdram_length = len;
1034  }
1035  }
1036  }
