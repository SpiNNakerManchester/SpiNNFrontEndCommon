97 static inline void merge_add(merge_t *m, unsigned int i) {
98  // Add the entry to the bit set contained in the merge
99  if (bit_set_add(&m->entries, i)) {
100  entry_t *entry = routing_table_get_entry(i);
101 
102  // Get the key_mask
103  if (m->key_mask.key == FULL && m->key_mask.mask == EMPTY) {
104  // If this is the first entry in the merge then the merge key_mask
105  // is a copy of the entry key_mask.
106  m->key_mask = entry->key_mask;
107  } else {
108  // Otherwise update the key and mask associated with the merge
109  m->key_mask = key_mask_merge(m->key_mask, entry->key_mask);
110  }
111 
112  // Add the route
113  m->route |= entry->route;
114  m->source |= entry->source;
115  }
116 }

130 static inline void merge_remove(merge_t *m, unsigned int i) {
131  // Remove the entry from the bit_set contained in the merge
132  if (bit_set_remove(&m->entries, i)) {
133  // Rebuild the key and mask from scratch
134  m->route = INIT_ROUTE;
135  m->source = INIT_SOURCE;
136  m->key_mask.key = FULL;
137  m->key_mask.mask = EMPTY;
138  for (int j = 0; j < routing_table_get_n_entries(); j++) {
139  entry_t *e = routing_table_get_entry(j);
140 
141  if (bit_set_contains(&m->entries, j)) {
142  m->route |= e->route;
143  m->source |= e->source;
144  if (m->key_mask.key == FULL && m->key_mask.mask == EMPTY) {
145  // Initialise the key_mask
146  m->key_mask.key = e->key_mask.key;
147  m->key_mask.mask = e->key_mask.mask;
148  } else {
149  // Merge the key_mask
150  m->key_mask = key_mask_merge(m->key_mask, e->key_mask);
151  }
152  }
153  }
154  }
