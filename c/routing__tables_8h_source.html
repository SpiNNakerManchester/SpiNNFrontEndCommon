41entry_t* routing_tables_get_entry_marked(uint32_t entry_id_to_find, int marker) {
42 uint32_t table_id = entry_id_to_find >> TABLE_SHIFT;
43 if (table_id >= multi_table.n_sub_tables) {
44 log_error("Id %d to big for %d tables marker %d",
45 entry_id_to_find, multi_table.n_sub_tables, marker);
46 malloc_extras_terminate(RTE_SWERR);
47 }
48 uint32_t local_id = entry_id_to_find & LOCAL_ID_ADD;
49 if (local_id >= multi_table.sub_tables[table_id]->size) {
50 log_error("Id %d has local_id %d which is too big for "
51 "table of size %d marker %d",
52 entry_id_to_find, local_id,
53 multi_table.sub_tables[table_id]->size, marker);
54 malloc_extras_terminate(RTE_SWERR);
55 }
56 return &multi_table.sub_tables[table_id]->entries[local_id];
57}

65entry_t* routing_tables_append_get_entry(void) {
66 // check that we're not hitting the max entries supported by the table
67 if (multi_table.n_entries == (int) multi_table.max_entries) {
68 log_error(
69 "There is no more space out of %d entries in this multi-table"
70 "for this entry.", multi_table.max_entries);
71 malloc_extras_terminate(RTE_SWERR);
72 }
73
74 // locate right table index
75 uint32_t table_id = multi_table.n_entries >> TABLE_SHIFT;
76 if (table_id >= multi_table.n_sub_tables) {
77 log_error("Id %d to big for %d tables",
78 multi_table.n_entries, multi_table.n_sub_tables);
79 malloc_extras_terminate(RTE_SWERR);
80 }
81
82 // locate entry index
83 uint32_t local_id = multi_table.n_entries & LOCAL_ID_ADD;
84 if (local_id != multi_table.sub_tables[table_id]->size) {
85 log_error("Id %d has local_id %d which is big for %d table",
86 multi_table.n_entries, local_id,
87 multi_table.sub_tables[table_id]->size);
88 malloc_extras_terminate(RTE_SWERR);
89 }
90
91 // update trackers.
92 multi_table.n_entries++;
93 multi_table.sub_tables[table_id]->size++;
94 return &multi_table.sub_tables[table_id]->entries[local_id];
95}

133void routing_tables_init(multi_table_t* table) {
134 multi_table.sub_tables = table->sub_tables;
135 multi_table.n_sub_tables = table->n_sub_tables;
136 multi_table.n_entries = table->n_entries;
137 multi_table.max_entries = table->max_entries;
138 log_debug("init with n table %d entries %d",
139 multi_table.n_sub_tables, multi_table.n_entries);
140
141 for (uint32_t i = 0; i < multi_table.n_sub_tables; i++) {
142 log_debug("table %d size %d", i, multi_table.sub_tables[i]->size);
143 }
144}

157void routing_table_remove_from_size(int size_to_remove) {
158 if (size_to_remove > multi_table.n_entries) {
159 log_error("Remove %d large than n_entries %d",
160 size_to_remove, multi_table.n_entries);
161 malloc_extras_terminate(RTE_SWERR);
162 }
163 multi_table.n_entries -= size_to_remove;
164}
