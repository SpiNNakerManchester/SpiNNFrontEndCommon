42 entry_t* routing_tables_get_entry_marked(uint32_t entry_id_to_find, int marker) {
43  uint32_t table_id = entry_id_to_find >> TABLE_SHIFT;
44  if (table_id >= multi_table.n_sub_tables) {
45  log_error("Id %d to big for %d tables marker %d",
46  entry_id_to_find, multi_table.n_sub_tables, marker);
47  malloc_extras_terminate(RTE_SWERR);
48  }
49  uint32_t local_id = entry_id_to_find & LOCAL_ID_ADD;
50  if (local_id >= multi_table.sub_tables[table_id]->size) {
51  log_error("Id %d has local_id %d which is too big for "
52  "table of size %d marker %d",
53  entry_id_to_find, local_id,
54  multi_table.sub_tables[table_id]->size, marker);
55  malloc_extras_terminate(RTE_SWERR);
56  }
57  return &multi_table.sub_tables[table_id]->entries[local_id];
58 }

66 entry_t* routing_tables_append_get_entry(void) {
67  // check that we're not hitting the max entries supported by the table
68  if (multi_table.n_entries == (int) multi_table.max_entries) {
69  log_error(
70  "There is no more space out of %d entries in this multi-table"
71  "for this entry.", multi_table.max_entries);
72  malloc_extras_terminate(RTE_SWERR);
73  }
74 
75  // locate right table index
76  uint32_t table_id = multi_table.n_entries >> TABLE_SHIFT;
77  if (table_id >= multi_table.n_sub_tables) {
78  log_error("Id %d to big for %d tables",
79  multi_table.n_entries, multi_table.n_sub_tables);
80  malloc_extras_terminate(RTE_SWERR);
81  }
82 
83  // locate entry index
84  uint32_t local_id = multi_table.n_entries & LOCAL_ID_ADD;
85  if (local_id != multi_table.sub_tables[table_id]->size) {
86  log_error("Id %d has local_id %d which is big for %d table",
87  multi_table.n_entries, local_id,
88  multi_table.sub_tables[table_id]->size);
89  malloc_extras_terminate(RTE_SWERR);
90  }
91 
92  // update trackers.
93  multi_table.n_entries++;
94  multi_table.sub_tables[table_id]->size++;
95  return &multi_table.sub_tables[table_id]->entries[local_id];
96 }
