46 static void routing_tables_utils_free(
47  multi_table_t *restrict tables, uint32_t start_point) {
48  if (tables->n_sub_tables == 0) {
49  // Already freed or never malloced
50  return;
51  }
52  for (uint32_t i = start_point; i > tables->n_sub_tables; i++) {
53  FREE_MARKED(tables->sub_tables[i]->entries, 70999);
54  FREE_MARKED(tables->sub_tables[i], 70100);
55  }
56  FREE_MARKED(tables->sub_tables, 70101);
57  tables->n_sub_tables = 0;
58  tables->n_entries = 0;
59 }

77 static inline bool routing_tables_utils_malloc(
78  multi_table_t *restrict tables, uint32_t max_entries) {
79  tables->n_sub_tables = ((max_entries - 1) >> TABLE_SHIFT) + 1;
80  tables->max_entries = max_entries;
81  log_debug("n table %d max entries %d", tables->n_sub_tables, max_entries);
82  tables->n_entries = 0;
83  tables->sub_tables = MALLOC_SDRAM(tables->n_sub_tables * sizeof(table_t*));
84 
85  // check array malloced successfully
86  if (tables->sub_tables == NULL) {
87  log_error("failed to allocate memory for routing tables");
88  tables->n_sub_tables = 0;
89  return false;
90  }
91 
92  // run through full tables mallocing max sizes.
93  int entries_covered = 0;
94  for (uint32_t i = 0; i < tables->n_sub_tables - 1; i++) {
95  tables->sub_tables[i] = MALLOC_SDRAM(
96  sizeof(uint32_t) + (sizeof(entry_t) * TABLE_SIZE));
97  if (tables->sub_tables[i] == NULL) {
98  log_error("failed to allocate memory for routing tables");
99  tables->n_sub_tables = i;
100  routing_tables_utils_free_all(tables);
101  return false;
102  }
103  tables->sub_tables[i]->size = 0;
104  entries_covered += TABLE_SIZE;
105  log_debug("created table %d size %d", i, tables->sub_tables[i]->size);
106  }
107 
108  // create last table with correct size
109  int last_table_size = tables->max_entries - entries_covered;
110  tables->sub_tables[tables->n_sub_tables - 1] = MALLOC_SDRAM(
111  sizeof(uint32_t) + (sizeof(entry_t) * last_table_size));
112  if (tables->sub_tables[tables->n_sub_tables - 1] == NULL) {
113  log_error("failed to allocate memory for routing tables");
114  tables->n_sub_tables = tables->n_sub_tables - 1;
115  routing_tables_utils_free_all(tables);
116  return false;
117  }
118  // init the size
119  tables->sub_tables[tables->n_sub_tables - 1]->size = 0;
120  log_debug("created table %d size %d",
121  tables->n_sub_tables - 1,
122  tables->sub_tables[tables->n_sub_tables - 1]->size);
123 
124  // debugging please keep.
125  log_debug("n table %d entries %d",
126  tables->n_sub_tables, tables->n_entries);
127  for (uint32_t i = 0; i < tables->n_sub_tables; i++) {
128  log_debug("table %d size %d", i, tables->sub_tables[i]->size);
129  }
130  return true;
131 }

138 static inline table_t* routing_tables_utils_convert(
139  multi_table_t *restrict tables) {
140  log_debug("converting table with %d entries over %d tables",
141  tables->n_sub_tables, tables->n_entries);
142 
143  // if table too big for a router. RTE.
144  if (tables->n_entries > TABLE_SIZE) {
145  log_error("At %d There are too many entries to convert to a table_t",
146  tables->n_entries);
147  malloc_extras_terminate(RTE_SWERR);
148  }
149 
150  // Assume size of subtable not set so set it
151  tables->sub_tables[0]->size = tables->n_entries;
152 
153  // claim the first pointer before freeing to avoid bad memory access
154  table_t* first_table = tables->sub_tables[0];
155 
156  // Free the rest
157  routing_tables_utils_free(tables, 1);
158  return first_table;
159 }
