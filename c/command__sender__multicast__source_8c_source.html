137static void transmit_command(command *command_to_send) {
138 // check for repeats
139 if (command_to_send->repeats != 0) {
140 for (uint32_t repeat_count = 0;
141 repeat_count <= command_to_send->repeats;
142 repeat_count++) {
143 if (command_to_send->has_payload) {
144 log_debug("Sending %08x, %08x at time %u with %u repeats and "
145 "%u delay",
146 command_to_send->key, command_to_send->payload, time,
147 command_to_send->repeats, command_to_send->delay);
148 spin1_send_mc_packet(
149 command_to_send->key, command_to_send->payload,
150 WITH_PAYLOAD);
151 } else {
152 log_debug("Sending %08x at time %u with %u repeats and "
153 "%u delay",
154 command_to_send->key, time, command_to_send->repeats,
155 command_to_send->delay);
156 spin1_send_mc_packet(command_to_send->key, 0, NO_PAYLOAD);
157 }
158 n_commands_sent++;
159
160 // if the delay is 0, don't call delay
161 if (command_to_send->delay > 0) {
162 spin1_delay_us(command_to_send->delay);
163 }
164 }
165 } else {
166 if (command_to_send->has_payload) {
167 log_debug("Sending %08x, %08x at time %u",
168 command_to_send->key, command_to_send->payload, time);
169
170 //if no repeats, then just send the message
171 spin1_send_mc_packet(
172 command_to_send->key, command_to_send->payload,
173 WITH_PAYLOAD);
174 } else {
175 log_debug("Sending %08x at time %u", command_to_send->key, time);
176 spin1_send_mc_packet(command_to_send->key, 0, NO_PAYLOAD);
177 }
178 n_commands_sent++;
179 }
180}

204static bool read_scheduled_parameters(timed_command_list *sdram_timed_commands) {
205 n_timed_commands = sdram_timed_commands->size;
206 log_info("%d timed commands", n_timed_commands);
207
208 // if no data, do not read it in
209 if (n_timed_commands == 0) {
210 return true;
211 }
212
213 // Allocate the space for the scheduled_commands
214 timed_commands = spin1_malloc(n_timed_commands * sizeof(timed_command));
215
216 if (timed_commands == NULL) {
217 log_error("Could not allocate the scheduled commands");
218 return false;
219 }
220
221 spin1_memcpy(timed_commands, sdram_timed_commands->commands,
222 n_timed_commands * sizeof(timed_command));
223
224 log_info("Schedule commands starts at time %u",
225 timed_commands[FIRST_TIME].time);
226 return true;
227}

285static void timer_callback(UNUSED uint unused0, UNUSED uint unused1) {
286 time++;
287
288 if (resume) {
289 log_info("running first start resume commands");
290 run_start_resume_commands();
291 resume = false;
292 }
293
294 if (simulation_is_finished()) {
295 run_stop_pause_commands();
296
297 simulation_handle_pause_resume(NULL);
298
299 log_info("in pause resume mode");
300 resume = true;
301
302 // Subtract 1 from the time so this tick gets done again on the next
303 // run
304 time--;
305
306 simulation_ready_to_read();
307 return;
308 }
309
310 while ((next_timed_command < n_timed_commands) &&
311 (timed_commands[next_timed_command].time == time)) {
312 transmit_command(&timed_commands[next_timed_command].command);
313 ++next_timed_command;
314 }
315}

327static bool initialize(uint32_t *timer_period) {
328 // Get the address this core's DTCM data starts at from SRAM
329 data_specification_metadata_t *ds_regions =
330 data_specification_get_data_address();
331
332 // Read the header
333 if (!data_specification_read_header(ds_regions)) {
334 return false;
335 }
336
337 // Get the timing details and set up the simulation interface
338 if (!simulation_initialise(
339 data_specification_get_region(SYSTEM_REGION, ds_regions),
340 APPLICATION_NAME_HASH, timer_period, &simulation_ticks,
341 &infinite_run, &time, SDP, DMA)) {
342 return false;
343 }
344 simulation_set_provenance_function(
345 write_provenance,
346 data_specification_get_region(PROVENANCE_REGION, ds_regions));
347 simulation_set_exit_function(run_stop_pause_commands);
348
349 // Read the parameters
350 bool success;
351 success = read_scheduled_parameters(data_specification_get_region(
352 COMMANDS_WITH_ARBITRARY_TIMES, ds_regions));
353 success &= read_start_resume_commands(data_specification_get_region(
354 COMMANDS_AT_START_RESUME, ds_regions));
355 success &= read_pause_stop_commands(data_specification_get_region(
356 COMMANDS_AT_STOP_PAUSE, ds_regions));
357 return success;
358}

361void c_main(void) {
362 // Configure system
363 uint32_t timer_period = 0;
364 if (!initialize(&timer_period)) {
365 log_error("Error in initialisation - exiting!");
366 rt_error(RTE_SWERR);
367 }
368
369 // Set timer_callback
370 spin1_set_timer_tick(timer_period);
371
372 // Register callbacks
373 spin1_callback_on(TIMER_TICK, timer_callback, TIMER);
374
375 // Start the time at "-1" so that the first tick will be 0
376 time = UINT32_MAX;
377 simulation_run();
378}
