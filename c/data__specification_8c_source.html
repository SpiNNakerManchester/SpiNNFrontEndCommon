75 bool data_specification_read_header(
76  data_specification_metadata_t *ds_regions) {
77  // Check for the magic number
78  if (ds_regions->magic_number != DATA_SPECIFICATION_MAGIC_NUMBER) {
79  log_error("Magic number is incorrect: %08x", ds_regions->magic_number);
80  return false;
81  }
82 
83  if (ds_regions->version != DATA_SPECIFICATION_VERSION) {
84  log_error("Version number is incorrect: %08x", ds_regions->version);
85  return false;
86  }
87 
88  // Log what we have found
89  log_info("magic = %08x, version = %d.%d", ds_regions->magic_number,
90  ds_regions->version >> VERSION_SHIFT,
91  ds_regions->version & VERSION_MASK);
92 
93  // Check the data in the regions checksums
94  for (uint32_t region = 0; region < N_REGIONS; region++) {
95  uint32_t *data = ds_regions->regions[region].pointer;
96  uint32_t checksum = ds_regions->regions[region].checksum;
97  uint32_t n_words = ds_regions->regions[region].n_words;
98 
99  // If the region is not in use, skip
100  if (data == NULL) {
101  continue;
102  }
103 
104  // Do simple unsigned 32-bit checksum
105  uint32_t sum = 0;
106  for (uint32_t i = 0; i < n_words; i++) {
107  sum += data[i];
108  }
109  if (sum != checksum) {
110  log_error("Region %u with %u words starting at 0x%08x: "
111  "checksum %u does not match computed sum %u",
112  region, n_words, data, checksum, sum);
113  return false;
114  }
115  }
116 
117  return true;
118 }
