49data_specification_metadata_t *data_specification_get_data_address(void) {
50 // Get pointer to 1st virtual processor info struct in SRAM
51 vcpu_t *virtual_processor_table = (vcpu_t*) SV_VCPU;
52
53 // Get the address this core's DTCM data starts at from the user data
54 // member of the structure associated with this virtual processor
55 uint user0 = virtual_processor_table[spin1_get_core_id()].user0;
56
57 log_debug("SDRAM data begins at address: %08x", user0);
58
59 // Cast to the correct type
60 return (data_specification_metadata_t *) user0;
61}

73bool data_specification_read_header(
74 data_specification_metadata_t *ds_regions) {
75 // Check for the magic number
76 if (ds_regions->magic_number != DATA_SPECIFICATION_MAGIC_NUMBER) {
77 log_error("Magic number is incorrect: %08x", ds_regions->magic_number);
78 return false;
79 }
80
81 if (ds_regions->version != DATA_SPECIFICATION_VERSION) {
82 log_error("Version number is incorrect: %08x", ds_regions->version);
83 return false;
84 }
85
86 // Log what we have found
87 log_info("magic = %08x, version = %d.%d", ds_regions->magic_number,
88 ds_regions->version >> VERSION_SHIFT,
89 ds_regions->version & VERSION_MASK);
90 return true;
91}
