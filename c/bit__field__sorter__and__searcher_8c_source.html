121 static inline bool load_routing_table_into_router(void) {
122  // Try to allocate sufficient room for the routing table.
123  int start_entry = rtr_alloc_id(last_compressed_table->size, app_id);
124  if (start_entry == RTR_MC_FAILED) {
125  log_error("Unable to allocate routing table of size %d\n",
126  last_compressed_table->size);
127  return false;
128  }
129 
130  // Load entries into the table (provided the allocation succeeded).
131  // Note that although the allocation included the specified
132  // application ID we also need to include it as the most significant
133  // byte in the route (see `sark_hw.c`).
134  log_debug("loading %d entries into router", last_compressed_table->size);
135  for (uint32_t entry_id = 0; entry_id < last_compressed_table->size;
136  entry_id++) {
137  entry_t entry = last_compressed_table->entries[entry_id];
138  uint32_t route = entry.route | (app_id << ROUTE_APP_ID_BIT_SHIFT);
139  uint success = rtr_mc_set(
140  start_entry + entry_id,
141  entry.key_mask.key, entry.key_mask.mask, route);
142 
143  // chekc that the entry was set
144  if (success == RTR_MC_FAILED) {
145  log_error("failed to set a router table entry at index %d",
146  start_entry + entry_id);
147  return false;
148  }
149  }
150 
151  // Indicate we were able to allocate routing table entries.
152  return true;
153 }

204 static inline bool pass_instructions_to_compressor(
205  uint32_t processor_id, uint32_t mid_point, uint32_t table_size) {
206 
207  bool success = routing_tables_utils_malloc(
208  comms_sdram[processor_id].routing_tables, table_size);
209  if (!success) {
210  log_info("failed to create bitfield tables for midpoint %d",
211  mid_point);
212  return false;
213  }
214 
215  // set the midpoint for the given compressor processor.
216  comms_sdram[processor_id].mid_point = mid_point;
217 
218  if (comms_sdram[processor_id].mid_point == 0){
219  // Info stuff but local sorted_bit_fields as compressor not set yet
220  log_info("using processor %d with %d entries for %d bitfields out of %d",
221  processor_id, table_size,
222  comms_sdram[processor_id].mid_point,
223  sorted_bit_fields->n_bit_fields);
224  } else {
225  // Info stuff using compressor data
226  log_info("using processor %d with %d entries for %d bitfields out of %d",
227  processor_id, table_size,
228  comms_sdram[processor_id].mid_point,
229  comms_sdram[processor_id].sorted_bit_fields->n_bit_fields);
230  }
231 
232  comms_sdram[processor_id].sorter_instruction = RUN;
233  return true;

242 static inline void malloc_tables_and_set_off_bit_compressor(

246

254  bool success = pass_instructions_to_compressor(

256

263  if ((comms_sdram[p_id].sorter_instruction == PREPARE) ||
264  (comms_sdram[p_id].sorter_instruction == TO_BE_PREPARED)) {
265  comms_sdram[p_id].sorter_instruction = DO_NOT_USE;
266  }
267  }
268 
269  // Ok that midpoint did not work so need to try it again
270  bit_field_clear(tested_mid_points, mid_point);
271  }

301 static inline void set_n_merged_filters(void) {
302  uint32_t highest_key[MAX_PROCESSORS];
303  int highest_order[MAX_PROCESSORS];
304  log_info("best_success %d", best_success);
305 
306  // Initialize highest order to -1 ie None merged in
307  for (int index = 0; index < MAX_PROCESSORS; index++) {
308  highest_order[index] = -1;
309  }
310 
311  // Find the first key above the best midpoint for each processor
312  for (int sorted_index = 0; sorted_index < sorted_bit_fields->n_bit_fields;
313  sorted_index++) {
314  int test = sorted_bit_fields->sort_order[sorted_index];
315  if (test <= best_success) {
316  int processor_id = sorted_bit_fields->processor_ids[sorted_index];
317  if (test > highest_order[processor_id]) {
318  highest_order[processor_id] = test;
319  highest_key[processor_id] =
320  sorted_bit_fields->bit_fields[sorted_index]->key;
321  }
322  }
323  }
324 
325  // debug
326  for (int processor_id = 0; processor_id < MAX_PROCESSORS; processor_id++) {
327  log_debug("processor %d, first_key %d first_order %d", processor_id,
328  highest_key[processor_id], highest_order[processor_id]);
329  }
330 
331  // Set n_redundancy_filters
332  for (int r_id = 0; r_id < region_addresses->n_triples; r_id++) {
333  int processor_id = region_addresses->triples[r_id].processor;
334  filter_region_t *filter = region_addresses->triples[r_id].filter;
335  int index = filter->n_redundancy_filters - 1;
336 
337  // Find the index of highest one merged in
338  while ((index >= 0) &&
339  (filter->filters[index].key != highest_key[processor_id])) {
340  index--;
341  }
342  filter->n_merged_filters = index + 1;
343  log_info("core %d has %d bitfields of which %d have redundancy "
344  "of which %d merged in", processor_id, filter->n_filters,
345  filter->n_redundancy_filters, filter->n_merged_filters);
346  }
347 }

351 static inline int locate_next_mid_point(void) {
352  if (sorted_bit_fields->n_bit_fields == 0) {
353  return FAILED_TO_FIND;
354  }
355 
356  // if not tested yet / reset test all
357  if (!bit_field_test(tested_mid_points, sorted_bit_fields->n_bit_fields)) {
358  log_info("Retrying all which is mid_point %d",
359  sorted_bit_fields->n_bit_fields);
360  return sorted_bit_fields->n_bit_fields;
361  }
362 
363  // need to find a midpoint
364  log_debug("n_bf_addresses %d tested_mid_points %d",
365  sorted_bit_fields->n_bit_fields,
366  bit_field_test(tested_mid_points,
367  sorted_bit_fields->n_bit_fields));
368 
369  // the last point of the longest space
370  int best_end = FAILED_TO_FIND;
371 
372  // the length of the longest space to test
373  int best_length = 0;
374 
375  // the current length of the currently detected space
376  int current_length = 0;
377 
378  log_debug("best_success %d lowest_failure %d",
379  best_success, lowest_failure);
380 
381  // iterate over the range to binary search, looking for biggest block to
382  // explore, then take the middle of that block
383 
384  // NOTE: if there are no available bitfields, this will result in best end
385  // being still set to -1, as every bit is set, so there is no blocks with
386  // any best length, and so best end is never set and lengths will still be
387  // 0 at the end of the for loop. -1 is a special midpoint which higher
388  // code knows to recognise as no more exploration needed.
389  for (int index = best_success + 1; index <= lowest_failure; index++) {
390  log_debug("index: %d, value: %u current_length: %d",
391  index, bit_field_test(tested_mid_points, index),
392  current_length);
393 
394  // verify that the index has been used before
395  if (bit_field_test(tested_mid_points, index)) {
396  // if used before and is the end of the biggest block seen so far.
397  // Record and repeat.
398  if (current_length > best_length) {
399  best_length = current_length;
400  best_end = index - 1;
401  log_debug("found best_length: %d best_end %d",
402  best_length, best_end);
403  // if not the end of the biggest block, ignore (log for debugging)
404  } else {
405  log_debug("not best: %d best_end %d", best_length, best_end);
406  }
407  // if its seen a set we're at the end of a block. so reset the
408  // current block len, as we're about to start another block.
409  current_length = 0;
410  // not set, so still within a block, increase len.
411  } else {
412  current_length += 1;
413  }
414  }
415 
416  // use the best less half (shifted) of the best length
417  int new_mid_point = best_end - (best_length >> 1);
418  log_debug("returning mid point %d", new_mid_point);
419 
420  // set the mid point to be tested. (safe as we de-set if we fail later on)
421  if (new_mid_point >= 0) {
422  log_debug("setting new mid point %d", new_mid_point);
423 
424  // just a safety check, as this has caught us before.
425  if (bit_field_test(tested_mid_points, new_mid_point)) {
426  log_info("HOW THE HELL DID YOU GET HERE!");
427  malloc_extras_terminate(EXIT_SWERR);
428  }
429  }
430 
431  return new_mid_point;
432 }

466 
467  // Create the space for the routing table meta data
468  comms_sdram[processor_id].routing_tables =
469  MALLOC_SDRAM(sizeof(multi_table_t));
470  if (comms_sdram[processor_id].routing_tables == NULL) {
471  comms_sdram[processor_id].sorter_instruction = DO_NOT_USE;
472  log_error("Error mallocing routing bake pointer on %d", processor_id);
473  return false;
474  }

477  comms_sdram[processor_id].routing_tables->n_sub_tables = 0;
478  comms_sdram[processor_id].routing_tables->n_entries = 0;
479 
480  // Pass the fake heap stuff
481  comms_sdram[processor_id].fake_heap_data = malloc_extras_get_stolen_heap();
482  log_debug("fake_heap_data %u", comms_sdram[processor_id].fake_heap_data);
483 
484  // Check the processor is live
485  int count = 0;
486  while (comms_sdram[processor_id].compressor_state != PREPARED) {
487  // give chance for compressor to read
488  spin1_delay_us(SDRAM_POLL_DELAY);
489  if (++count > SDRAM_POLL_ATTEMPTS) {
490  comms_sdram[processor_id].sorter_instruction = DO_NOT_USE;
491  log_error("compressor failed to reply %d", processor_id);
492  return false;
493  }
494  }
495  return true;
496 }

502 int find_prepared_processor(void) {
503  // Look for a prepared one
504  for (int processor_id = 0; processor_id < MAX_PROCESSORS; processor_id++) {
505  if (comms_sdram[processor_id].sorter_instruction == PREPARE &&
506  comms_sdram[processor_id].compressor_state == PREPARED) {
507  log_debug("found prepared %d", processor_id);
508  return processor_id;
509  }
510  }
511 
512  // NOTE: This initialization component exists here due to a race condition
513  // with the compressors, where we dont know if they are reacting to
514  // "messages" before sync signal has been sent. We also have this here to
515  // save the 16 bytes per compressor core we dont end up using.
516 
517  // Look for a processor never used and prepare it
518  for (int processor_id = 0; processor_id < MAX_PROCESSORS; processor_id++) {
519  log_debug("processor_id %d status %d",
520  processor_id, comms_sdram[processor_id].sorter_instruction);
521  if (comms_sdram[processor_id].sorter_instruction == TO_BE_PREPARED) {
522  if (prepare_processor_first_time(processor_id)) {
523  log_debug("found to be prepared %d", processor_id);
524  return processor_id;
525  }
526  log_debug("first failed %d", processor_id);
527  }
528  }
529  log_debug("FAILED %d", FAILED_TO_FIND);
530  return FAILED_TO_FIND;
531 }

610 bool exit_carry_on_if_all_compressor_processors_done(void) {
611  for (int processor_id = 0; processor_id < MAX_PROCESSORS; processor_id++) {
612  if (comms_sdram[processor_id].sorter_instruction >= PREPARE) {
613  return false;
614  }
615  }
616 
617  // Check there is nothing left to do
618  int mid_point = locate_next_mid_point();
619  if (mid_point != FAILED_TO_FIND) {
620  log_error("Ran out of processors while still having mid_point %d to do",
621  mid_point);
622  malloc_extras_terminate(RTE_SWERR);
623  // Should never get here but break out of the loop
624  terminated = true;
625  return true;
626  }
627 
628  // Should never get here if above check worked but just in case
629  if (just_reduced_cores_due_to_malloc) {
630  log_error("Last result was a malloc fail! Use host");
631  malloc_extras_terminate(RTE_SWERR);
632  // Should never get here but break out of the loop
633  terminated = true;
634  return true;
635  }
636 
637  // Check there was actually a result
638  if (best_success == FAILED_TO_FIND) {
639  log_error("No usable result found! Use host");
640  malloc_extras_terminate(RTE_SWERR);
641  // Should never get here but break out of the loop
642  terminated = true;
643  return true;
644  }
645 
646  // Should never get here if above check failed but just in case
647  if (best_success < (int) threshold_in_bitfields) {
648  log_error("The threshold is %d bitfields. "
649  "Which is %d percent of the total of %d",
650  threshold_in_bitfields, region_addresses->threshold,
651  sorted_bit_fields->n_bit_fields);
652  log_error("Best result found was %d Which is below the threshold! "
653  "Use host", best_success);
654  malloc_extras_terminate(RTE_SWERR);
655  // Should never get here but break out of the loop
656  terminated = true;
657  return true;
658  }
659 
660  handle_best_cleanup();
661 
662  // Should never get here but break out of the loop
663  terminated = true;
664  return true;
665 }

668 void carry_on_binary_search(void) {
669  if (exit_carry_on_if_all_compressor_processors_done()) {
670  return; // Should never get here but just in case
671  }
672  if (all_compressor_processors_busy()) {
673  log_debug("all_compressor_processors_busy");
674  return; //Pass back to check_buffer_queue
675  }
676  log_debug("start carry_on_binary_search");
677 
678  int mid_point = locate_next_mid_point();
679  log_debug("available with midpoint %d", mid_point);
680 
681  if (mid_point == FAILED_TO_FIND) {
682  // OK, lets turn all ready processors off as done.
683  for (int p_id = 0; p_id < MAX_PROCESSORS; p_id++) {
684  if (comms_sdram[p_id].sorter_instruction == PREPARE) {
685  comms_sdram[p_id].sorter_instruction = DO_NOT_USE;
686  } else if (comms_sdram[p_id].sorter_instruction > PREPARE) {
687  log_debug(
688  "waiting for processor %d status %d doing midpoint %u",
689  p_id, comms_sdram[p_id].sorter_instruction,
690  comms_sdram[p_id].mid_point);
691  }
692  }
693  return;
694  }
695 
696  int processor_id = find_compressor_processor_and_set_tracker(mid_point);
697  log_debug("start create at time step: %u", time_steps);
698  malloc_tables_and_set_off_bit_compressor(mid_point, processor_id);
699  log_debug("end create at time step: %u", time_steps);
700 }

744  send_force_stop_message(processor_id);
745  }
746  }

777 void process_failed(int mid_point, int processor_id) {
778  // safety check to ensure we dont go on if the uncompressed failed
779  if (mid_point <= (int) threshold_in_bitfields) {
780  if (threshold_in_bitfields == NO_BIT_FIELDS) {
781  log_error("The no bitfields attempted failed! Giving up");
782  } else {
783  log_error("The threshold is %d, "
784  "which is %d percent of the total of %d",
785  threshold_in_bitfields, region_addresses->threshold,
786  sorted_bit_fields->n_bit_fields);
787  log_error("The attempt with %d bitfields failed. ! Giving up",
788  mid_point);
789  }
790  malloc_extras_terminate(EXIT_FAIL);
791  }
792  if (lowest_failure > mid_point) {
793  log_info("Changing lowest_failure from: %d to mid_point:%d",
794  lowest_failure, mid_point);
795  lowest_failure = mid_point;
796  } else {
797  log_info("lowest_failure: %d already lower than mid_point:%d",
798  lowest_failure, mid_point);
799  }
800  routing_tables_utils_free_all(comms_sdram[processor_id].routing_tables);
801 
802  // tell all compression processors trying midpoints above this one
803  // to stop, as its highly likely a waste of time.
804  for (int processor_id = 0; processor_id < MAX_PROCESSORS; processor_id++) {
805  if (comms_sdram[processor_id].mid_point > mid_point) {
806  send_force_stop_message(processor_id);
807  }
808  }
809 
810  // handler to say this message has changed the last to not be a malloc fail
811  just_reduced_cores_due_to_malloc = false;
812 }

817 void process_compressor_response(
818  int processor_id, compressor_states finished_state) {
819  // locate this responses midpoint
820  int mid_point = comms_sdram[processor_id].mid_point;
821  log_debug("received response %d from processor %d doing %d midpoint",
822  finished_state, processor_id, mid_point);
823 
824  // free the processor for future processing
825  send_prepare_message(processor_id);
826 
827  // process compressor response based off state.
828  switch (finished_state) {
829  case SUCCESSFUL_COMPRESSION:
830  // compressor was successful at compressing the tables.
831  log_info("successful from processor %d doing mid point %d "
832  "best so far was %d",
833  processor_id, mid_point, best_success);
834  process_success(mid_point, processor_id);
835  break;
836 
837  case FAILED_MALLOC:
838  // compressor failed as a malloc request failed.
839  log_info("failed by malloc from processor %d doing mid point %d",
840  processor_id, mid_point);
841  process_failed_malloc(mid_point, processor_id);
842  break;
843 
844  case FAILED_TO_COMPRESS:
845  // compressor failed to compress the tables as no more merge options.
846  log_info("failed to compress from processor %d doing mid point %d",
847  processor_id, mid_point);
848  process_failed(mid_point, processor_id);
849  break;
850 
851  case RAN_OUT_OF_TIME:
852  // compressor failed to compress as it ran out of time.
853  log_info("failed by time from processor %d doing mid point %d",
854  processor_id, mid_point);
855  process_failed(mid_point, processor_id);
856  break;
857 
858  case FORCED_BY_COMPRESSOR_CONTROL:
859  // compressor stopped at the request of the sorter.
860  log_debug("ack from forced from processor %d doing mid point %d",
861  processor_id, mid_point);
862  routing_tables_utils_free_all(comms_sdram[processor_id].routing_tables);
863  break;
864 
865  case UNUSED_CORE:
866  case PREPARED:
867  case COMPRESSING:
868  // states that shouldn't occur
869  log_error("no idea what to do with finished state %d, "
870  "from processor %d", finished_state, processor_id);
871  malloc_extras_terminate(RTE_SWERR);
872  }
873 }

878 void check_compressors(UNUSED uint unused0, UNUSED uint unused1) {
879  log_info("Entering the check_compressors loop");
880  // iterate over the compressors buffer until we have the finished state
881  while (!terminated) {
882  bool no_new_result = true;
883 
884  // iterate over processors looking for a new result
885  for (int processor_id = 0; processor_id < MAX_PROCESSORS; processor_id++) {
886  // Check each compressor asked to run or forced
887  compressor_states finished_state =
888  comms_sdram[processor_id].compressor_state;
889  if (finished_state > COMPRESSING) {
890  no_new_result = false;
891  process_compressor_response(processor_id, finished_state);
892  }
893  }
894  if (no_new_result) {
895  log_debug("no_new_result");
896  // Check if another processor could be started or even done
897  carry_on_binary_search();
898  } else {
899  log_debug("result");
900  }
901  }
902  // Safety code in case exit after setting best_found fails
903  log_info("exiting the interrupt, to allow the binary to finish");
904 }

908 void start_binary_search(void) {
909  // Find the number of available processors
910  uint32_t available = 0;
911  for (int processor_id = 0; processor_id < MAX_PROCESSORS; processor_id++) {
912  if (comms_sdram[processor_id].sorter_instruction == TO_BE_PREPARED) {
913  available++;
914  }
915  }
916 
917  // Set off the worse acceptable (note no bitfield would have been set off
918  // earlier)
919  if (threshold_in_bitfields > 0) {
920  int processor_id = find_compressor_processor_and_set_tracker(
921  threshold_in_bitfields);
922  malloc_tables_and_set_off_bit_compressor(
923  threshold_in_bitfields, processor_id);
924  }
925 
926  // create slices and set off each slice.
927  uint32_t mid_point = sorted_bit_fields->n_bit_fields;
928  while ((available > 0) && (mid_point > threshold_in_bitfields)) {
929  int processor_id = find_compressor_processor_and_set_tracker(mid_point);
930  // Check the processor replied and has not been turned of by previous
931  if (processor_id == FAILED_TO_FIND) {
932  log_error("No processor available in start_binary_search");
933  return;
934  }
935  malloc_tables_and_set_off_bit_compressor(mid_point, processor_id);
936 
937  // Find the next step which may change due to rounding
938  int step = ((mid_point - threshold_in_bitfields) / available);
939  if (step < 1) {
940  step = 1;
941  }
942  mid_point -= step;
943  available--;
944  }
945 }

950 void start_compression_process(UNUSED uint unused0, UNUSED uint unused1) {
951  // malloc the struct and populate n bit-fields. DOES NOT populate the rest.
952  sorted_bit_fields = bit_field_reader_initialise(region_addresses);
953  // check state to fail if not read in
954  if (sorted_bit_fields == NULL) {
955  log_error("failed to read in bitfields, quitting");
956  malloc_extras_terminate(EXIT_MALLOC);
957  }
958 
959  // Set the threshold
960  if (region_addresses->threshold == 0) {
961  threshold_in_bitfields = 0;
962  } else {
963  threshold_in_bitfields = (sorted_bit_fields->n_bit_fields *
964  region_addresses->threshold) / 100;
965  best_success = threshold_in_bitfields;
966  }
967  log_info("threshold_in_bitfields %d which is %d percent of %d",
968  threshold_in_bitfields, region_addresses->threshold,
969  sorted_bit_fields->n_bit_fields);
970 
971  // set up mid point trackers. NEEDED here as setup no bitfields attempt
972  // will use it during processor allocation.
973  set_up_tested_mid_points();
974 
975  // set off the first compression attempt (aka no bitfields).
976  bool success = setup_no_bitfields_attempt();
977  if (!success) {
978  log_error("failed to set up uncompressed attempt");
979  malloc_extras_terminate(EXIT_MALLOC);
980  }
981 
982  log_debug("populating sorted bitfields at time step: %d", time_steps);
983  bit_field_reader_read_in_bit_fields(region_addresses, sorted_bit_fields);
984 
985  // the first possible failure is all bitfields so set there.
986  lowest_failure = sorted_bit_fields->n_bit_fields;
987  log_debug("finished reading bitfields at time step: %d", time_steps);
988 
989  //TODO: safety code to be removed
990  for (int bit_field_index = 0;
991  bit_field_index < sorted_bit_fields->n_bit_fields;
992  bit_field_index++) {
993  // get key
994  filter_info_t *bf_pointer =
995  sorted_bit_fields->bit_fields[bit_field_index];
996  if (bf_pointer == NULL) {
997  log_info("failed at index %d", bit_field_index);
998  malloc_extras_terminate(RTE_SWERR);
999  return;
1000  }
1001  }
1002 
1003  // start the binary search by slicing the search space by the compressors.
1004  start_binary_search();
1005 
1006  // set off checker which in turn sets of the other compressor processors
1007  spin1_schedule_callback(
1008  check_compressors, 0, 0, COMPRESSION_START_PRIORITY);
1009 }
