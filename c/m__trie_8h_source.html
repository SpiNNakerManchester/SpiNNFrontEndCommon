122 static unsigned int m_trie_count(m_trie_t *node) {
123  if (node == NULL) {
124  return 0; // Not a node, so return 0
125  } else if (!node->bit) {
126  return 1; // Node is a leaf, so return a 1
127  }
128 
129  // Get the contribution from each child (if there are any)
130  return m_trie_count(node->child_0) +
131  m_trie_count(node->child_1) +
132  m_trie_count(node->child_X);
133 }

141 static m_trie_entry_t *_get_entries(
142  m_trie_t *node, m_trie_entry_t *table, uint32_t p_key,
143  uint32_t p_mask) {
144  if (node == NULL) {
145  // Do nothing as this isn't a valid node.
146  } else if (!node->bit) {
147  // If this is a leaf then add an entry to the table representing this
148  // entry and return a pointer to the next entry in the table.
149  table->key_mask.key = p_key;
150  table->key_mask.mask = p_mask;
151  table->source = node->source;
152 
153  // Point to the next table entry
154  table++;
155  } else {
156  // If this is not a leaf then get entries from any children we may have.
157  uint32_t b = node->bit; // Bit to set
158  table = _get_entries(node->child_0, table, p_key, p_mask | b);
159  table = _get_entries(node->child_1, table, p_key | b, p_mask | b);
160  table = _get_entries(node->child_X, table, p_key, p_mask);
161  }
162 
163  return table;
164 }

178 static inline m_trie_t **get_child(
179  m_trie_t *node, uint32_t key, uint32_t mask) {
180  if (mask & node->bit) { // Either a 0 or a 1
181  if (!(key & node->bit)) { // A 0 at this bit
182  return &node->child_0;
183  } else { // A 1 at this bit
184  return &node->child_1;
185  }
186  } else if (!(key & node->bit)) { // An X at this bit
187  return &node->child_X;
188  } else {
189  return NULL; // A `!' at this bit, abort
190  }
191 }

199 static m_trie_t* m_trie_traverse(
200  m_trie_t *node, uint32_t key, uint32_t mask, uint32_t source) {
201  while (node->bit) {
202  // See where to turn at this node
203  m_trie_t **child = get_child(node, key, mask);
204 
205  // If no child was returned then the given key and mask were invalid
206  if (!child) {
207  return NULL;
208  }
209 
210  // If the child is NULL then create a new child
211  if (!*child) {
212  *child = m_trie_new_node(node, node->bit >> 1);
213  }
214 
215  // Delegate the traversal to the child
216  node = *child;
217  }
218 
219  // If we are a leaf then update our source and return our parent
220  node->source |= source;
221  return node->parent;
222 }

229 static inline bool path_exists(m_trie_t *node, uint32_t key, uint32_t mask) {
230  while (node->bit) {
231  // See where to turn at this node
232  m_trie_t **child = get_child(node, key, mask);
233 
234  // If there is no child then the path does not exist or the given path
235  // was invalid.
236  if (!child || !*child) {
237  return false;
238  }
239 
240  // Delegate the traversal to the child
241  node = *child;
242  }
243 
244  // If we are a leaf then the path must exist
245  return true;
246 }

253 static bool m_trie_un_traverse(
254  m_trie_t *node, uint32_t key, uint32_t mask) {
255  if (!node->bit) {
256  // If a leaf then free ourselves and then return to true to indicate
257  // that this has occurred.
258  FREE(node);
259  return true;
260  }
261 
262  // Get the child to attempt to un traverse
263  m_trie_t **child = get_child(node, key, mask);
264 
265  if (m_trie_un_traverse(*child, key, mask)) {
266  // The child has been freed, so remove our reference to it
267  *child = NULL;
268  }
269 
270  // If we have no children left then free ourselves and return true;
271  // otherwise return false.
272  if (node->child_0 == NULL && node->child_1 == NULL &&
273  node->child_X == NULL) {
274  FREE(node);
275  return true;
276  } else{
277  return false;
278  }
279 }

347 static inline void m_trie_insert(
348  m_trie_t *root, uint32_t key, uint32_t mask, uint32_t source) {
349  // Traverse a path through the trie and keep a reference to the leaf we
350  // reach
351  m_trie_t *leaf = m_trie_traverse(root, key, mask, source);
352 
353  // Attempt to find overlapping paths, going up the tree
354  for (; leaf != NULL; leaf = leaf->parent) {
355  m_trie_t **child_0 = &leaf->child_0;
356  m_trie_t **child_1 = &leaf->child_1;
357  m_trie_t **child_X = &leaf->child_X;
358 
359  if (*child_0 != NULL && path_exists(*child_0, key, mask) &&
360  *child_1 != NULL && path_exists(*child_1, key, mask)) {
361  // Get the combined sources from the existing children
362  source = get_source_from_child(*child_0, key, mask) |
363  get_source_from_child(*child_1, key, mask);
364 
365  // Traverse the path in X and then un traverse in 0 and 1
366  if (*child_X == NULL) {
367  *child_X = m_trie_new_node(leaf, leaf->bit >> 1);
368  }
369  m_trie_traverse(*child_X, key, mask, source);
370 
371  // Un traverse in `0' and `1'
372  un_traverse_in_child(child_0, key, mask);
373  un_traverse_in_child(child_1, key, mask);
374 
375  // Update the key and mask
376  key &= ~leaf->bit;
377  mask &= ~leaf->bit;
378  } else if (*child_X != NULL && path_exists(*child_X, key, mask) &&
379  *child_0 != NULL && path_exists(*child_0, key, mask)) {
380  // Get the source for packets matching the `0'
381  source = get_source_from_child(*child_0, key, mask);
382 
383  // Un traverse in `0'
384  un_traverse_in_child(child_0, key, mask);
385 
386  // Add the sources to X
387  add_source_to_child(*child_X, key, mask, source);
388 
389  // Update the key and mask
390  key &= ~leaf->bit;
391  mask &= ~leaf->bit;
392  } else if (*child_X != NULL && path_exists(*child_X, key, mask) &&
393  *child_1 != NULL && path_exists(*child_1, key, mask)) {
394  // Get the source for packets matching the `1'
395  source = get_source_from_child(*child_1, key, mask);
396 
397  // Un traverse in `1'
398  un_traverse_in_child(child_1, key, mask);
399 
400  // Add the sources to X
401  add_source_to_child(*child_X, key, mask, source);
402 
403  // Update the key and mask
404  key &= ~leaf->bit;
405  mask &= ~leaf->bit;
406  }
407  }

435 static inline void sub_table_expand(sub_table_t *sb, table_t *table) {
436  // Keep a count of the new number of entries
437  table->size = 0;
438 
439  // For each sub table copy in the new entries
440  entry_t *next = table->entries;
441  for (; sb != NULL; sb = sb->next) {
442  // Expand the current sub table
443  for (unsigned int i = 0; i < sb->n_entries; i++) {
444  next->key_mask.key = sb->entries[i].key_mask.key;
445  next->key_mask.mask = sb->entries[i].key_mask.mask;
446  next->source = sb->entries[i].source;
447  next->route = sb->route;
448 
449  next++;
450  }
451 
452  // Increase the size of the output table
453  table->size += sb->n_entries;
454  }
455 }

477 static inline void m_trie_minimise(table_t *table) {
478  // For each set of unique routes in the table we construct an m-Trie to
479  // minimise the entries; we then write the minimised table back in on-top
480  // of the original table.
481 
482  // Keep a reference to entries we've already dealt with
483  bit_set_t visited;
484  bit_set_init(&visited, table->size);
485 
486  // Maintain a chain of partially minimised tables
487  sub_table_t *sub_tables = NULL;
488 
489  // For every entry in the table that hasn't been inspected create a new
490  // m-Trie, then add to it every entry with an equivalent route.
491  for (unsigned int i = 0; i < table->size; i++) {
492  if (bit_set_contains(&visited, i)) {
493  continue;
494  }
495 
496  // Create a new m-Trie
497  m_trie_t *trie = m_trie_new();
498  uint32_t route = table->entries[i].route;
499 
500  // Add all equivalent entries to the trie
501  for (unsigned int j = i; j < table->size; j++) {
502  if (table->entries[j].route == route) {
503  // Mark the entry as visited
504  bit_set_add(&visited, j);
505 
506  // Add to the trie
507  uint32_t key = table->entries[j].key_mask.key;
508  uint32_t mask = table->entries[j].key_mask.mask;
509  uint32_t source = table->entries[j].source;
510  m_trie_insert(trie, key, mask, source);
511  }
512  }
513 
514  // Read out all the minimised entries into a new sub table
515  sub_table_t *sb = sub_table_new(&sub_tables, m_trie_count(trie), route);
516  m_trie_get_entries(trie, sb->entries);
517 
518  // Delete the m-Trie
519  m_trie_delete(trie);
520  }
521 
522  // Overwrite the original routing table by copying entries back from the
523  // sub_tables.
524  sub_table_expand(sub_tables, table);
525 
526  // Clear up the sub_tables and bit_set.
527  sub_table_delete(sub_tables); sub_tables = NULL;
528  bit_set_delete(&visited);
529 }
