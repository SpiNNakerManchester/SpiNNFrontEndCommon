123 static unsigned int m_trie_count(m_trie_t *node) {
124  if (node == NULL) {
125  return 0; // Not a node, so return 0
126  } else if (!node->bit) {
127  return 1; // Node is a leaf, so return a 1
128  }
129 
130  // Get the contribution from each child (if there are any)
131  return m_trie_count(node->child_0) +
132  m_trie_count(node->child_1) +
133  m_trie_count(node->child_X);
134 }

142 static m_trie_entry_t *_get_entries(
143  m_trie_t *node, m_trie_entry_t *table, uint32_t p_key,
144  uint32_t p_mask) {
145  if (node == NULL) {
146  // Do nothing as this isn't a valid node.
147  } else if (!node->bit) {
148  // If this is a leaf then add an entry to the table representing this
149  // entry and return a pointer to the next entry in the table.
150  table->key_mask.key = p_key;
151  table->key_mask.mask = p_mask;
152  table->source = node->source;
153 
154  // Point to the next table entry
155  table++;
156  } else {
157  // If this is not a leaf then get entries from any children we may have.
158  uint32_t b = node->bit; // Bit to set
159  table = _get_entries(node->child_0, table, p_key, p_mask | b);
160  table = _get_entries(node->child_1, table, p_key | b, p_mask | b);
161  table = _get_entries(node->child_X, table, p_key, p_mask);
162  }
163 
164  return table;
165 }

179 static inline m_trie_t **get_child(
180  m_trie_t *node, uint32_t key, uint32_t mask) {
181  if (mask & node->bit) { // Either a 0 or a 1
182  if (!(key & node->bit)) { // A 0 at this bit
183  return &node->child_0;
184  } else { // A 1 at this bit
185  return &node->child_1;
186  }
187  } else if (!(key & node->bit)) { // An X at this bit
188  return &node->child_X;
189  } else {
190  return NULL; // A `!' at this bit, abort
191  }
192 }

200 static m_trie_t* m_trie_traverse(
201  m_trie_t *node, uint32_t key, uint32_t mask, uint32_t source) {
202  while (node->bit) {
203  // See where to turn at this node
204  m_trie_t **child = get_child(node, key, mask);
205 
206  // If no child was returned then the given key and mask were invalid
207  if (!child) {
208  return NULL;
209  }
210 
211  // If the child is NULL then create a new child
212  if (!*child) {
213  *child = m_trie_new_node(node, node->bit >> 1);
214  }
215 
216  // Delegate the traversal to the child
217  node = *child;
218  }
219 
220  // If we are a leaf then update our source and return our parent
221  node->source |= source;
222  return node->parent;
223 }

230 static inline bool path_exists(m_trie_t *node, uint32_t key, uint32_t mask) {
231  while (node->bit) {
232  // See where to turn at this node
233  m_trie_t **child = get_child(node, key, mask);
234 
235  // If there is no child then the path does not exist or the given path
236  // was invalid.
237  if (!child || !*child) {
238  return false;
239  }
240 
241  // Delegate the traversal to the child
242  node = *child;
243  }
244 
245  // If we are a leaf then the path must exist
246  return true;
247 }

254 static bool m_trie_un_traverse(
255  m_trie_t *node, uint32_t key, uint32_t mask) {
256  if (!node->bit) {
257  // If a leaf then free ourselves and then return to true to indicate
258  // that this has occurred.
259  FREE(node);
260  return true;
261  }
262 
263  // Get the child to attempt to un traverse
264  m_trie_t **child = get_child(node, key, mask);
265 
266  if (m_trie_un_traverse(*child, key, mask)) {
267  // The child has been freed, so remove our reference to it
268  *child = NULL;
269  }
270 
271  // If we have no children left then free ourselves and return true;
272  // otherwise return false.
273  if (node->child_0 == NULL && node->child_1 == NULL &&
274  node->child_X == NULL) {
275  FREE(node);
276  return true;
277  } else{
278  return false;
279  }
280 }

348 static inline void m_trie_insert(
349  m_trie_t *root, uint32_t key, uint32_t mask, uint32_t source) {
350  // Traverse a path through the trie and keep a reference to the leaf we
351  // reach
352  m_trie_t *leaf = m_trie_traverse(root, key, mask, source);
353 
354  // Attempt to find overlapping paths, going up the tree
355  for (; leaf != NULL; leaf = leaf->parent) {
356  m_trie_t **child_0 = &leaf->child_0;
357  m_trie_t **child_1 = &leaf->child_1;
358  m_trie_t **child_X = &leaf->child_X;
359 
360  if (*child_0 != NULL && path_exists(*child_0, key, mask) &&
361  *child_1 != NULL && path_exists(*child_1, key, mask)) {
362  // Get the combined sources from the existing children
363  source = get_source_from_child(*child_0, key, mask) |
364  get_source_from_child(*child_1, key, mask);
365 
366  // Traverse the path in X and then un traverse in 0 and 1
367  if (*child_X == NULL) {
368  *child_X = m_trie_new_node(leaf, leaf->bit >> 1);
369  }
370  m_trie_traverse(*child_X, key, mask, source);
371 
372  // Un traverse in `0' and `1'
373  un_traverse_in_child(child_0, key, mask);
374  un_traverse_in_child(child_1, key, mask);
375 
376  // Update the key and mask
377  key &= ~leaf->bit;
378  mask &= ~leaf->bit;
379  } else if (*child_X != NULL && path_exists(*child_X, key, mask) &&
380  *child_0 != NULL && path_exists(*child_0, key, mask)) {
381  // Get the source for packets matching the `0'
382  source = get_source_from_child(*child_0, key, mask);
383 
384  // Un traverse in `0'
385  un_traverse_in_child(child_0, key, mask);
386 
387  // Add the sources to X
388  add_source_to_child(*child_X, key, mask, source);
389 
390  // Update the key and mask
391  key &= ~leaf->bit;
392  mask &= ~leaf->bit;
393  } else if (*child_X != NULL && path_exists(*child_X, key, mask) &&
394  *child_1 != NULL && path_exists(*child_1, key, mask)) {
395  // Get the source for packets matching the `1'
396  source = get_source_from_child(*child_1, key, mask);
397 
398  // Un traverse in `1'
399  un_traverse_in_child(child_1, key, mask);
400 
401  // Add the sources to X
402  add_source_to_child(*child_X, key, mask, source);
403 
404  // Update the key and mask
405  key &= ~leaf->bit;
406  mask &= ~leaf->bit;
407  }
408  }
409 }

436 static inline void sub_table_expand(sub_table_t *sb, table_t *table) {
437  // Keep a count of the new number of entries
438  table->size = 0;
439 
440  // For each sub table copy in the new entries
441  entry_t *next = table->entries;
442  for (; sb != NULL; sb = sb->next) {
443  // Expand the current sub table
444  for (unsigned int i = 0; i < sb->n_entries; i++) {
445  next->key_mask.key = sb->entries[i].key_mask.key;
446  next->key_mask.mask = sb->entries[i].key_mask.mask;
447  next->source = sb->entries[i].source;
448  next->route = sb->route;
449 
450  next++;
451  }
452 
453  // Increase the size of the output table
454  table->size += sb->n_entries;
455  }
456 }

478 static inline void m_trie_minimise(table_t *table) {
479  // For each set of unique routes in the table we construct an m-Trie to
480  // minimise the entries; we then write the minimised table back in on-top
481  // of the original table.
482 
483  // Keep a reference to entries we've already dealt with
484  bit_set_t visited;
485  bit_set_init(&visited, table->size);
486 
487  // Maintain a chain of partially minimised tables
488  sub_table_t *sub_tables = NULL;
489 
490  // For every entry in the table that hasn't been inspected create a new
491  // m-Trie, then add to it every entry with an equivalent route.
492  for (unsigned int i = 0; i < table->size; i++) {
493  if (bit_set_contains(&visited, i)) {
494  continue;
495  }
496 
497  // Create a new m-Trie
498  m_trie_t *trie = m_trie_new();
499  uint32_t route = table->entries[i].route;
500 
501  // Add all equivalent entries to the trie
502  for (unsigned int j = i; j < table->size; j++) {
503  if (table->entries[j].route == route) {
504  // Mark the entry as visited
505  bit_set_add(&visited, j);
506 
507  // Add to the trie
508  uint32_t key = table->entries[j].key_mask.key;
509  uint32_t mask = table->entries[j].key_mask.mask;
510  uint32_t source = table->entries[j].source;
511  m_trie_insert(trie, key, mask, source);
512  }
513  }
514 
515  // Read out all the minimised entries into a new sub table
516  sub_table_t *sb = sub_table_new(&sub_tables, m_trie_count(trie), route);
517  m_trie_get_entries(trie, sb->entries);
518 
519  // Delete the m-Trie
520  m_trie_delete(trie);
521  }
522 
523  // Overwrite the original routing table by copying entries back from the
524  // sub_tables.
525  sub_table_expand(sub_tables, table);
526 
527  // Clear up the sub_tables and bit_set.
528  sub_table_delete(sub_tables); sub_tables = NULL;
529  bit_set_delete(&visited);
530 }
