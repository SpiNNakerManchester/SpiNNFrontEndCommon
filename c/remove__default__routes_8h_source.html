30 static bool remove_default_routes_minimise(
31  int *pointer_to_new_size, bool remove_elements) {
32  // Mark the entries to be removed from the table
33  bit_set_t remove;
34  bool success = bit_set_init(&remove, routing_table_get_n_entries());
35  if (!success) {
36  log_debug("failed to initialise the bit_set. shutting down");
37  return false;
38  }
39 
40  // Work up the table from the bottom, marking entries to remove
41  for (int i = routing_table_get_n_entries() - 1; i >= 0; i--) {
42 
43  // Get the current entry
44  entry_t *entry = routing_table_get_entry(i);
45 
46  // See if it can be removed
47  // only removed if Only one output direction which is a link. or
48  // Only one input direction which is a link. or
49  // Source is opposite to sink
50  //TODO NAME THESE MAGIC NUMBERS (MUNDY)
51  if (__builtin_popcount(entry->route) == 1 && (entry->route & 0x3f) &&
52  __builtin_popcount(entry->source) == 1 &&
53  (entry->source & 0x3f) &&
54  (entry->route >> 3) == (entry->source & 0x7) &&
55  (entry->source >> 3) == (entry->route & 0x7)) {
56 
57  // The entry can be removed iff. it doesn't intersect with any entry
58  // further down the table.
59  bool remove_entry = true;
60  for (int j = i + 1; j < routing_table_get_n_entries(); j++) {
61  // If entry we're comparing with is already going to be
62  // removed, ignore it.
63  if (bit_set_contains(&remove, j)) {
64  continue;
65  }
66 
67  key_mask_t a = entry->key_mask;
68 
69  // get next entry key mask
70 
71  entry_t *j_entry = routing_table_get_entry(j);
72  key_mask_t b = j_entry->key_mask;
73 
74  if (key_mask_intersect(a, b)) {
75  remove_entry = false;
76  break;
77  }
78  }
79 
80  if (remove_entry) {
81  // Mark this entry as being removed
82  bit_set_add(&remove, i);
83  }
84  }
85  }
86 
87  // Remove the selected entries from the table
88  if (remove_elements) {
89  for (int insert = 0, read = 0; read < routing_table_get_n_entries();
90  read++) {
91  // Grab the current entry before we potentially overwrite it
92  entry_t *current = routing_table_get_entry(read);
93 
94  // Insert the entry if it isn't being removed
95  if (!bit_set_contains(&remove, read)) {
96  entry_t* insert_entry = routing_table_get_entry(insert++);
97  insert_entry->key_mask.key = current->key_mask.key;
98  insert_entry->key_mask.mask = current->key_mask.mask;
99  insert_entry->route = current->route;
100  insert_entry->source = current->source;
101  }
102  }
103  // Update the table size
104  log_debug("remove redundant");
105  routing_table_remove_from_size(remove.count);
106  }
107 
108  *pointer_to_new_size -= remove.count;
109 
110  // Clear up
111  bit_set_delete(&remove);
112  return true;
113 }
