32 static bool remove_default_routes_minimise(
33  int *new_size, bool remove_elements) {
34  // Mark the entries to be removed from the table
35  bit_set_t remove;
36  bool success = bit_set_init(&remove, routing_table_get_n_entries());
37  if (!success) {
38  log_debug("failed to initialise the bit_set. shutting down");
39  return false;
40  }
41 
42  // Work up the table from the bottom, marking entries to remove
43  for (int i = routing_table_get_n_entries() - 1; i >= 0; i--) {
44 
45  // Get the current entry
46  entry_t *entry = routing_table_get_entry(i);
47 
48  // See if it can be removed
49  // only removed if Only one output direction which is a link. or
50  // Only one input direction which is a link. or
51  // Source is opposite to sink
52  //TODO NAME THESE MAGIC NUMBERS (MUNDY)
53  if (__builtin_popcount(entry->route) == 1 && (entry->route & 0x3f) &&
54  __builtin_popcount(entry->source) == 1 &&
55  (entry->source & 0x3f) &&
56  (entry->route >> 3) == (entry->source & 0x7) &&
57  (entry->source >> 3) == (entry->route & 0x7)) {
58 
59  // The entry can be removed iff. it doesn't intersect with any entry
60  // further down the table.
61  bool remove_entry = true;
62  for (int j = i + 1; j < routing_table_get_n_entries(); j++) {
63  // If entry we're comparing with is already going to be
64  // removed, ignore it.
65  if (bit_set_contains(&remove, j)) {
66  continue;
67  }
68 
69  key_mask_t a = entry->key_mask;
70 
71  // get next entry key mask
72 
73  entry_t *j_entry = routing_table_get_entry(j);
74  key_mask_t b = j_entry->key_mask;
75 
76  if (key_mask_intersect(a, b)) {
77  remove_entry = false;
78  break;
79  }
80  }
81 
82  if (remove_entry) {
83  // Mark this entry as being removed
84  bit_set_add(&remove, i);
85  }
86  }
87  }
88 
89  // Remove the selected entries from the table
90  if (remove_elements) {
91  for (int insert = 0, read = 0; read < routing_table_get_n_entries();
92  read++) {
93  // Grab the current entry before we potentially overwrite it
94  entry_t *current = routing_table_get_entry(read);
95 
96  // Insert the entry if it isn't being removed
97  if (!bit_set_contains(&remove, read)) {
98  entry_t* insert_entry = routing_table_get_entry(insert++);
99  insert_entry->key_mask.key = current->key_mask.key;
100  insert_entry->key_mask.mask = current->key_mask.mask;
101  insert_entry->route = current->route;
102  insert_entry->source = current->source;
103  }
104  }
105  // Update the table size
106  log_debug("remove redundant");
107  routing_table_remove_from_size(remove.count);
108  }
109 
110  *new_size -= remove.count;
111 
112  // Clear up
113  bit_set_delete(&remove);
114  return true;
115 }
