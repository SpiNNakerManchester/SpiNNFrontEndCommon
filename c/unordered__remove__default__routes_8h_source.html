32 static inline bool _just_a_link(uint32_t direction) {
33  return __builtin_popcount(direction) == 1 && // Only one direction...
34  (direction & LINK_MASK); // which is a link.
35 }

37 // Test if route's source is opposite to sink
38 static inline bool _opposite_links(entry_t *entry) {
39  uint32_t src = entry->source & LINK_MASK;
40  uint32_t dst = entry->route & LINK_MASK;
41  // Equivalent to a rotate of 6 bits by 3 places and single compare
42  return ((dst >> 3) == (src & 0x7) && (src >> 3) == (dst & 0x7));
43 }

47 static inline void remove_default_routes_minimise(table_t *table) {
48  // Work out if removing defaultable links is worthwhile
49  uint32_t after_size = table->size;
50  for (uint32_t i = 0; i < table->size; i++) {
51  // Get the current entry
52  entry_t entry = table->entries[i];
53 
54  // See if it can be removed
55  if (_just_a_link(entry.route) && // Only one output, a link
56  _just_a_link(entry.source) && // Only one input, a link
57  _opposite_links(&entry)) { // Source is opposite to sink
58  after_size--;
59  }
60  }
61 
62  // If we won't fit afterwards, no sense trying
63  if (after_size > rtr_alloc_max()) {
64  return;
65  }
66 
67  // Do the actual removal
68  for (uint32_t i = 0; i < table->size; i++) {
69  // Get the current entry
70  entry_t entry = table->entries[i];
71 
72  // See if it can be removed
73  if (_just_a_link(entry.route) && // Only one output, a link
74  _just_a_link(entry.source) && // Only one input, a link
75  _opposite_links(&entry)) { // Source is opposite to sink
76  uint32_t last = table->size - 1;
77  if (i < last) {
78  table->entries[i] = table->entries[last];
79  table->size--;
80  // Reuse same i in next pass so reduce before the ++
81  i--;
82  }
83  }
84  }
