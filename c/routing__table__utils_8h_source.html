47 static void routing_table_utils_free(
48  multi_table_t *restrict tables, uint32_t start_point) {
49  if (tables->n_sub_tables == 0) {
50  // Already freed or never malloced
51  return;
52  }
53  for (uint32_t i = start_point; i > tables->n_sub_tables; i++) {
54  FREE_MARKED(tables->sub_tables[i]->entries, 70999);
55  FREE_MARKED(tables->sub_tables[i], 70100);
56  }
57  FREE_MARKED(tables->sub_tables, 70101);
58  tables->n_sub_tables = 0;
59  tables->n_entries = 0;
60 }

78 static inline bool routing_table_utils_malloc(
79  multi_table_t *restrict tables, uint32_t max_entries) {
80  tables->n_sub_tables = ((max_entries - 1) >> TABLE_SHIFT) + 1;
81  tables->max_entries = max_entries;
82  log_debug("n table %d max entries %d", tables->n_sub_tables, max_entries);
83  tables->n_entries = 0;
84  tables->sub_tables = MALLOC_SDRAM(tables->n_sub_tables * sizeof(table_t*));
85 
86  // check array malloced successfully
87  if (tables->sub_tables == NULL) {
88  log_error("failed to allocate memory for routing tables");
89  tables->n_sub_tables = 0;
90  return false;
91  }
92 
93  // run through full tables mallocing max sizes.
94  int entries_covered = 0;
95  for (uint32_t i = 0; i < tables->n_sub_tables - 1; i++) {
96  tables->sub_tables[i] = MALLOC_SDRAM(
97  sizeof(uint32_t) + (sizeof(entry_t) * TABLE_SIZE));
98  if (tables->sub_tables[i] == NULL) {
99  log_error("failed to allocate memory for routing tables");
100  tables->n_sub_tables = i;
101  routing_table_utils_free_all(tables);
102  return false;
103  }
104  tables->sub_tables[i]->size = 0;
105  entries_covered += TABLE_SIZE;
106  log_debug("created table %d size %d", i, tables->sub_tables[i]->size);
107  }
108 
109  // create last table with correct size
110  int last_table_size = tables->max_entries - entries_covered;
111  tables->sub_tables[tables->n_sub_tables - 1] = MALLOC_SDRAM(
112  sizeof(uint32_t) + (sizeof(entry_t) * last_table_size));
113  if (tables->sub_tables[tables->n_sub_tables - 1] == NULL) {
114  log_error("failed to allocate memory for routing tables");
115  tables->n_sub_tables = tables->n_sub_tables - 1;
116  routing_table_utils_free_all(tables);
117  return false;
118  }
119  // init the size
120  tables->sub_tables[tables->n_sub_tables - 1]->size = 0;
121  log_debug("created table %d size %d",
122  tables->n_sub_tables - 1,
123  tables->sub_tables[tables->n_sub_tables - 1]->size);
124 
125  // debugging please keep.
126  log_debug("n table %d entries %d",
127  tables->n_sub_tables, tables->n_entries);
128  for (uint32_t i = 0; i < tables->n_sub_tables; i++) {
129  log_debug("table %d size %d", i, tables->sub_tables[i]->size);
130  }
131  return true;
132 }

139 static inline table_t* routing_table_utils_convert(
140  multi_table_t *restrict tables) {
141  log_debug("converting table with %d entries over %d tables",
142  tables->n_sub_tables, tables->n_entries);
143 
144  // if table too big for a router. RTE.
145  if (tables->n_entries > TABLE_SIZE) {
146  log_error("At %d There are too many entries to convert to a table_t",
147  tables->n_entries);
148  malloc_extras_terminate(RTE_SWERR);
149  }
150 
151  // Assume size of subtable not set so set it
152  tables->sub_tables[0]->size = tables->n_entries;
153 
154  // claim the first pointer before freeing to avoid bad memory access
155  table_t* first_table = tables->sub_tables[0];
156 
157  // Free the rest

159  return first_table;
