38 int count_unique_keys(
39  sorted_bit_fields_t *restrict sorted_bit_fields, int midpoint) {
40  // semantic sugar
41  filter_info_t **restrict bit_fields = sorted_bit_fields->bit_fields;
42  int *restrict sort_order = sorted_bit_fields->sort_order;
43  int n_bit_fields = sorted_bit_fields->n_bit_fields;
44 
45  // as the sorted bitfields are sorted by key. checking key changes when
46  // within the midpoint will work.
47  int count = 0;
48  uint32_t last_key = -1;
49  for (int i = 0; i < n_bit_fields; i++) {
50  if ((sort_order[i] < midpoint) && (last_key != bit_fields[i]->key)) {
51  count++;
52  last_key = bit_fields[i]->key;
53  }
54  }
55  return count;
56 }

64 void generate_table(
65  entry_t original_entry, filter_info_t **restrict filters,
66  uint32_t *restrict bit_field_processors, int bf_found) {
67  uint32_t n_atoms = filters[0]->n_atoms;
68 
69  uint32_t stripped_route = original_entry.route;
70  for (int i =0; i < bf_found; i++) {
71  // Safety code to be removed
72  if (!bit_field_test(&stripped_route,
73  bit_field_processors[i] + MAX_LINKS_PER_ROUTER)) {
74  log_error("WHAT THE F***!");
75  }
76  bit_field_clear(&stripped_route,
77  bit_field_processors[i] + MAX_LINKS_PER_ROUTER);
78  }
79 
80  // iterate though each atom and set the route when needed
81  for (uint32_t atom = 0; atom < n_atoms; atom++) {
82  // Assigning to a uint32 creates a copy
83  uint32_t new_route = stripped_route;
84 
85  // iterate through the bitfield processor's and see if they need this
86  // atom
87  for (int bf_index = 0; bf_index < bf_found; bf_index++) {
88  log_debug("data address is %x", filters[bf_index]->data);
89  if (bit_field_test(filters[bf_index]->data, atom)) {
90  log_debug(
91  "setting for atom %d from bitfield index %d so proc %d",
92  atom, bf_index, bit_field_processors[bf_index]);
93  bit_field_set(&new_route,
94  MAX_LINKS_PER_ROUTER + bit_field_processors[bf_index]);
95  }
96  }
97 
98  routing_tables_append_new_entry(
99  original_entry.key_mask.key + atom,
100  NEURON_LEVEL_MASK, new_route, original_entry.source);
101  }
102  log_debug("key %d atoms %d size %d",
103  original_entry.key_mask.key, n_atoms,
104  routing_table_get_n_entries());
105 }

113 static inline uint32_t bit_field_table_generator_max_size(
114  int mid_point, table_t *restrict uncompressed_table,
115  sorted_bit_fields_t *restrict sorted_bit_fields) {
116  // semantic sugar to avoid referencing
117  filter_info_t **restrict bit_fields = sorted_bit_fields->bit_fields;
118  int *restrict sort_order = sorted_bit_fields->sort_order;
119 
120  // Start with the size of the uncompressed table
121  uint32_t max_size = uncompressed_table->size;
122  log_debug("keys %d", max_size);
123 
124  // Check every bitfield to see if is to be used
125  // Only need each key once to track last used as tables is sorted by key
126  uint32_t used_key = FAILED_TO_FIND;
127  for (int bf_i = sorted_bit_fields->n_bit_fields -1; bf_i >= 0; bf_i--) {
128  if (sort_order[bf_i] < mid_point) {
129  if (used_key != bit_fields[bf_i]->key) {
130  used_key = bit_fields[bf_i]->key;
131 
132  // One entry per atom but we can remove the uncompressed one
133  max_size += bit_fields[bf_i]->n_atoms -1;
134  log_debug("key %d size %d",
135  used_key, bit_fields[bf_i]->n_atoms);
136  }
137  }
138  }
139  log_debug("Using mid_point %d, counted size of table is %d",
140  mid_point, max_size);
141  return max_size;
142 }

149 static inline void bit_field_table_generator_create_bit_field_router_tables(
150  int mid_point,
151  table_t *restrict uncompressed_table,
152  sorted_bit_fields_t *restrict sorted_bit_fields) {
153  // semantic sugar to avoid referencing
154  filter_info_t **restrict bit_fields = sorted_bit_fields->bit_fields;
155  int *restrict processor_ids = sorted_bit_fields->processor_ids;
156  int *restrict sort_order = sorted_bit_fields->sort_order;
157  entry_t *restrict original = uncompressed_table->entries;
158  uint32_t original_size = uncompressed_table->size;
159  int n_bit_fields = sorted_bit_fields->n_bit_fields;
160 
161  filter_info_t * filters[MAX_PROCESSORS];
162  uint32_t bit_field_processors[MAX_PROCESSORS];
163  int bf_i = 0;
164  log_debug("pre size %d", routing_table_get_n_entries());
165 
166  for (uint32_t rt_i = 0; rt_i < original_size; rt_i++) {
167  uint32_t key = original[rt_i].key_mask.key;
168  log_debug("key %d", key);
169  int bf_found = 0;
170 
171  while ((bf_i < n_bit_fields) && (bit_fields[bf_i]->key == key)) {
172  if (sort_order[bf_i] < mid_point) {
173  filters[bf_found] = bit_fields[bf_i];
174  bit_field_processors[bf_found] = processor_ids[bf_i];
175  bf_found++;
176  }
177  bf_i++;
178  }
179 
180  if (bf_found > 0) {
181  generate_table(original[rt_i], filters, bit_field_processors,
182  bf_found);
183  } else {
184  routing_tables_append_entry(original[rt_i]);
185  }
186  log_debug("key %d size %d",
187  original[rt_i].key_mask.key, routing_table_get_n_entries());
188  }
189 }

205 void sort_table_by_key(table_t *table) {
206  uint32_t size = table->size;
207  entry_t *entries = table->entries;
208 
209  for (uint32_t i = 0; i < size - 1; i++) {
210  for (uint32_t j = i + 1; j < size; j++) {
211  if (entries[i].key_mask.key > entries[j].key_mask.key) {
212  uint32_t temp = entries[i].key_mask.key;
213  entries[i].key_mask.key = entries[j].key_mask.key;
214  entries[j].key_mask.key = temp;
215 
216  temp = entries[i].key_mask.mask;
217  entries[i].key_mask.mask = entries[j].key_mask.mask;
218  entries[j].key_mask.mask = temp;
219 
220  temp = entries[i].route;
221  entries[i].route = entries[j].route;
222  entries[j].route = temp;
223 
224  temp = entries[i].source;
225  entries[i].source = entries[j].source;
226  entries[j].source = temp;
227  }
228  }
229  }
