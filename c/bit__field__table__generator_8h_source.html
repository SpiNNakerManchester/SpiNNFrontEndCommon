38int count_unique_keys(
39 sorted_bit_fields_t *restrict sorted_bit_fields, int midpoint) {
40 // semantic sugar
41 filter_info_t **restrict bit_fields = sorted_bit_fields->bit_fields;
42 int *restrict sort_order = sorted_bit_fields->sort_order;
43 int n_bit_fields = sorted_bit_fields->n_bit_fields;
44
45 // as the sorted bitfields are sorted by key. checking key changes when
46 // within the midpoint will work.
47 int count = 0;
48 uint32_t last_key = -1;
49 for (int i = 0; i < n_bit_fields; i++) {
50 if ((sort_order[i] < midpoint) && (last_key != bit_fields[i]->key)) {
51 count++;
52 last_key = bit_fields[i]->key;
53 }
54 }
55 return count;
56}

67bool generate_table(
68 entry_t original_entry, filter_info_t **restrict filters,
69 uint32_t *restrict bit_field_processors, int bf_found,
70 struct core_atom *core_atom) {
71
72 // Remove the processor bits from the route that match the bitfields
73 uint32_t stripped_route = original_entry.route;
74 for (int i = 0; i < bf_found; i++) {
75 bit_field_clear(&stripped_route,
76 bit_field_processors[i] + MAX_LINKS_PER_ROUTER);
77 }
78
79 // Go through the atoms, potentially starting where we left off
80 uint32_t first_atom = global_atom(filters[0], core_atom);
81 uint32_t n_atoms = filters[0]->n_atoms;
82 for (uint32_t atom = first_atom; atom < n_atoms; atom++) {
83
84 // Stop when the route no longer matches the key from the bit field,
85 // as this will resume with another entry later
86 uint32_t atom_key = get_bf_key(filters[0], core_atom);
87 if ((atom_key & original_entry.key_mask.mask)
88 != original_entry.key_mask.key) {
89 // We need to continue this later, so say yes!
90 return true;
91 }
92
93 // Start with a copy of the stripped route
94 uint32_t new_route = stripped_route;
95
96 // Add the processor for each bit field where the bit for the atom is set
97 for (int bf_index = 0; bf_index < bf_found; bf_index++) {
98 log_debug("data address is %x", filters[bf_index]->data);
99 if (bit_field_test(filters[bf_index]->data, atom)) {
100 log_debug(
101 "setting for atom %d from bitfield index %d so proc %d",
102 atom, bf_index, bit_field_processors[bf_index]);
103 bit_field_set(&new_route,
104 MAX_LINKS_PER_ROUTER + bit_field_processors[bf_index]);
105 }
106 }
107
108 // Add a new entry based on the bit fields
109 routing_tables_append_new_entry(
110 original_entry.key_mask.key + (atom - first_atom),
111 NEURON_LEVEL_MASK, new_route, original_entry.source);
112
113 // Get the next core atom for the next round
114 next_core_atom(filters[0], core_atom);
115 }
116
117 log_debug("key %d atoms %d size %d",
118 original_entry.key_mask.key, n_atoms,
119 routing_table_get_n_entries());
120 // We got through all atoms, so say no!
121 return false;
122}

130static inline uint32_t bit_field_table_generator_max_size(
131 int mid_point, table_t *restrict uncompressed_table,
132 sorted_bit_fields_t *restrict sorted_bit_fields) {
133 // semantic sugar to avoid referencing
134 filter_info_t **restrict bit_fields = sorted_bit_fields->bit_fields;
135 int *restrict sort_order = sorted_bit_fields->sort_order;
136
137 // Start with the size of the uncompressed table
138 uint32_t max_size = uncompressed_table->size;
139 log_debug("keys %d", max_size);
140
141 // Check every bitfield to see if is to be used
142 // Only need each key once to track last used as tables is sorted by key
143 uint32_t last_key = 0xFFFFFFFF;
144 bool is_last_key = false;
145 for (int bf_i = 0; bf_i < sorted_bit_fields->n_bit_fields; bf_i++) {
146 if (sort_order[bf_i] < mid_point) {
147 if (!is_last_key || last_key != bit_fields[bf_i]->key) {
148 last_key = bit_fields[bf_i]->key;
149 is_last_key = true;
150
151 // One entry per atom but we can remove the uncompressed one
152 max_size += bit_fields[bf_i]->n_atoms - 1;
153 log_debug("key %d size %d",
154 last_key, bit_fields[bf_i]->n_atoms);
155 }
156 }
157 }
158 log_debug("Using mid_point %d, counted size of table is %d",
159 mid_point, max_size);
160 return max_size;
161}

168static inline void bit_field_table_generator_create_bit_field_router_tables(
169 int mid_point,
170 table_t *restrict uncompressed_table,
171 sorted_bit_fields_t *restrict sorted_bit_fields) {
172 // semantic sugar to avoid referencing
173 filter_info_t **restrict bit_fields = sorted_bit_fields->bit_fields;
174 int *restrict processor_ids = sorted_bit_fields->processor_ids;
175 int *restrict sort_order = sorted_bit_fields->sort_order;
176 entry_t *restrict original = uncompressed_table->entries;
177 uint32_t original_size = uncompressed_table->size;
178 uint32_t n_bit_fields = sorted_bit_fields->n_bit_fields;
179
180 filter_info_t * filters[MAX_PROCESSORS];
181 uint32_t bit_field_processors[MAX_PROCESSORS];
182 log_debug("pre size %d", routing_table_get_n_entries());
183
184 // Go through key-sorted bit fields and routing entries in tandem.
185 // Note: there may be multiple routing entries for each bit field,
186 // but there must be only one bit field per processor per routing entry!
187 uint32_t rt_i = 0;
188 uint32_t bf_i = 0;
189 while (bf_i < n_bit_fields && rt_i < original_size) {
190 // Find a routing entry that starts at the current bit field (there
191 // must be one, because combined entries must be from the same source
192 // at this point).
193 while (rt_i < original_size &&
194 original[rt_i].key_mask.key != bit_fields[bf_i]->key) {
195 routing_tables_append_entry(original[rt_i++]);
196 }
197
198 // Get out while you still can!
199 if (rt_i >= original_size) {
200 break;
201 }
202
203 // Now find all bit fields with the same key, which will have the same
204 // remaining properties too (like atoms per core etc.) since they will
205 // be from the same source.
206 uint32_t key = original[rt_i].key_mask.key;
207 uint32_t bf_found = 0;
208 while (bf_i < n_bit_fields && bit_fields[bf_i]->key == key) {
209 if (sort_order[bf_i] < mid_point) {
210 filters[bf_found] = bit_fields[bf_i];
211 bit_field_processors[bf_found] = processor_ids[bf_i];
212 bf_found++;
213 }
214 bf_i++;
215 }
216
217 // If we found any bit fields that now match, create entries for each
218 // routing entry that continues to match the keys
219 if (bf_found > 0) {
220 // While the bit field is not finished from this entry, keep
221 // generating more
222 struct core_atom core_atom = {0, 0};
223 while (rt_i < original_size
224 && generate_table(original[rt_i], filters,
225 bit_field_processors, bf_found, &core_atom)) {
226 rt_i++;
227 }
228 // The last one will return false, so increment one more time
229 rt_i++;
230 }
231 }
232
233 // At this point, we might still not have finished the routing table;
234 // all remaining entries must be outside of the bit fields, so just copy
235 // them.
236 while (rt_i < original_size) {
237 routing_tables_append_entry(original[rt_i++]);
238 }
239}

264void sort_table_by_key(table_t *table) {
265 uint32_t size = table->size;
266 entry_t *entries = table->entries;
267
268 uint32_t i, j;
269 for (i = 1; i < size; i++) {
270 const entry_t temp = entries[i];
271 for (j = i; j > 0 && compare_entries(&entries[j - 1], &temp); j--) {
272 entries[j] = entries[j - 1];
273 }
274 entries[j] = temp;
275 }
276}
