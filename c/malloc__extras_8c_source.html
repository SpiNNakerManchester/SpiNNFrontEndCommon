96 bool malloc_extras_check(void *ptr) {
97  // only check if safety is turned on. else its not possible to check.
98  if (safety) {
99  int *int_pointer = (int *) ptr;
100  int_pointer = int_pointer - 1;
101  int words = int_pointer[0];
102 
103  for (int buffer_index = 0; buffer_index < BUFFER_WORDS;
104  buffer_index++) {
105  uint32_t flag = int_pointer[words + buffer_index];
106  if (flag != SAFETY_FLAG) {
107  bool found = false;
108  for (int index = 0; index < malloc_points_size; index ++) {
109  if ((malloc_points[index] != 0) &&
110  (malloc_points[index] == ptr)) {
111  found = true;
112  }
113  }
114  if (found) {
115  log_error("flag is actually %x for ptr %x", flag, ptr);
116  } else {
117  log_error("Unexpected ptr %x", ptr);
118  }
119  return false;
120  }
121  }
122  return true;
123  }
124  return true;
125 }

234 static inline bool add_heap_to_collection(
235  sdram_block *list_of_available_blocks) {
236  // go through true heap and allocate and add to end of list.
237  int position = 0;
238 
239  // loop over the true heap and add accordingly.
240  while (sv->sdram_heap->free != NULL) {
241  block_t *next_blk = sv->sdram_heap->free->next;
242 
243  // get next size minus the size it'll need to put in when alloc'ing
244  int size = ((uchar *) next_blk - (uchar *) sv->sdram_heap->free) -
245  sizeof(block_t);
246 
247  // make life easier by saying blocks have to be bigger than the heap.
248  // so all spaces can be used for heaps
249  uchar *b_address = sark_xalloc(sv->sdram_heap, size, 0, ALLOC_LOCK);
250  if (b_address == NULL) {
251  log_error("failed to allocate %d", size);
252  return false;
253  }
254  list_of_available_blocks[position].sdram_base_address = b_address;
255  list_of_available_blocks[position].size = size;
256  stolen_sdram_heap->free_bytes += size;
257  position++;
258  }
259  return true;
260 }

268 static inline void make_heap_structure(
269  available_sdram_blocks *sizes_region, int n_mallocs,
270  sdram_block *list_of_available_blocks) {
271  // generate position pointers
272  int stolen_current_index = 0;
273  int heap_current_index = 0;
274  bool first = true;
275  block_t *previous = NULL;
276  block_t *previous_free = NULL;
277 
278  // generate heap pointers
279  while (stolen_current_index < sizes_region->n_blocks ||
280  heap_current_index < n_mallocs) {
281  // build pointers to try to reduce code space
282  int *to_process;
283  sdram_block *to_process_blocks;
284 
285  // determine which tracker to utilise
286  uint top_stolen = (uint) sizes_region->blocks[
287  stolen_current_index].sdram_base_address;
288  uint top_true = (uint) list_of_available_blocks[
289  heap_current_index].sdram_base_address;
290 
291  // determine which one to utilise now
292  if ((stolen_current_index < sizes_region->n_blocks) &&
293  top_stolen < top_true) {
294  to_process = &stolen_current_index;
295  to_process_blocks = sizes_region->blocks;
296  } else {
297  to_process = &heap_current_index;
298  to_process_blocks = list_of_available_blocks;
299  }
300 
301  // if has not already set up the heap struct, set it up
302  if (first) {
303  // set flag to not need to do this again
304  first = false;
305 
306  // set up stuff we can
307  stolen_sdram_heap->free = (block_t *)
308  to_process_blocks[*to_process].sdram_base_address;
309 
310  stolen_sdram_heap->free->next = (block_t *) (
311  to_process_blocks[*to_process].sdram_base_address +
312  to_process_blocks[*to_process].size - sizeof(block_t));
313 
314  stolen_sdram_heap->free->free = NULL;
315  stolen_sdram_heap->first = stolen_sdram_heap->free;
316 
317  // previous block in chain
318  previous = stolen_sdram_heap->free->next;
319  previous_free = stolen_sdram_heap->free;
320  } else {
321  // set up block in block
322  block_t *free = (block_t *)
323  to_process_blocks[*to_process].sdram_base_address;
324  free->free = NULL;
325 
326  // update next block
327  free->next = (block_t *) (
328  to_process_blocks[*to_process].sdram_base_address +
329  to_process_blocks[*to_process].size - sizeof(block_t));
330  free->next->free = NULL;
331  free->next->next = NULL;
332 
333  // update previous links
334  previous->next = free;
335  previous->free = free;
336  previous_free->free = free;
337 
338  // update previous pointers
339  previous = free->next;
340  previous_free = free;
341  }
342  // update pointers
343  (*to_process)++;
344  }
345 
346  // update last
347  stolen_sdram_heap->last = previous;
348  stolen_sdram_heap->last->free = NULL;
349  stolen_sdram_heap->last->next = NULL;
350 }

390 bool malloc_extras_initialise_and_build_fake_heap(
391  available_sdram_blocks *sizes_region) {
392  // hard set stolen sdram heap to the default heap. in case no fake heap
393  stolen_sdram_heap = sv->sdram_heap;
394 
395  /* if planning to track all mallocs and frees to verify no
396  overwrites/corruption. build the initial malloc tracker*/
397  if (safety) {
398  build_malloc_tracker();
399  }
400 
401  // only build the fake heap if there's bits to build with
402  if (sizes_region == NULL) {
403  return true;
404  }
405 
406  // allocate blocks store for figuring out block order
407  uint n_mallocs = find_n_available_mallocs(sv->sdram_heap);
408  sdram_block *list_of_available_blocks = sark_alloc(
409  n_mallocs * sizeof(sdram_block), 1);
410 
411  // if fail to alloc dtcm blow up
412  if (list_of_available_blocks == NULL) {
413  return false;
414  }
415 
416  // alloc space for a heap object (stealing from steal store if not by
417  // normal means.
418  stolen_sdram_heap =
419  sark_xalloc(sv->sdram_heap, MIN_SIZE_HEAP, 0, ALLOC_LOCK);
420  if (stolen_sdram_heap == NULL) {
421  // check we can steal
422  if (sizes_region->n_blocks == 0) {
423  log_error("cant find space for the heap");
424  return false;
425  }
426 
427  // deallocate 32 bytes from first handed down to be the heap object
428  stolen_sdram_heap = (heap_t *)
429  sizes_region->blocks[0].sdram_base_address;
430  sizes_region->blocks[0].sdram_base_address += MIN_SIZE_HEAP;
431  sizes_region->blocks[0].size -= MIN_SIZE_HEAP;
432  }
433 
434  // determine how much spare space there is.
435  stolen_sdram_heap->free_bytes = find_free_space_available(sizes_region);
436 
437  // go through true heap and allocate and add to end of list.
438  bool success = add_heap_to_collection(list_of_available_blocks);
439  if (!success) {
440  log_error("failed to add heap");
441  return false;
442  }
443 
444  // build the heap struct if there is a heap structure.
445  make_heap_structure(sizes_region, n_mallocs, list_of_available_blocks);
446 
447  // free the allocated dtcm for the true alloc stuff.
448  sark_free(list_of_available_blocks);
449 
450  // printer for sanity purposes
451  if (to_print) {
452  print_free_sizes_in_heap();
453  }
454 
455  return true;
456 }

465 void malloc_extras_free_marked(void *ptr, int marker) {
466  // only print if its currently set to print (saves iobuf)
467  if (to_print) {
468  log_info("freeing %x", ptr);
469  }
470 
471  // track if the pointer has been corrupted before trying to free it.
472  // only possible if safety been turned on
473  int *int_pointer = (int *) ptr;
474  if (safety) {
475  if (!malloc_extras_check(ptr)) {
476  log_error("over ran whatever is being freed");
477  log_error("marker is %d", marker);
478  malloc_extras_terminate(DETECTED_MALLOC_FAILURE);
479  }
480 
481  bool found = false;
482  int index = 0;
483  while (!found && index < malloc_points_size) {
484  if (malloc_points[index] == ptr) {
485  found = true;
486  malloc_points[index] = 0;
487  } else {
488  index++;
489  }
490  }
491 
492  // if set to print and there was a free index, print it
493  if (found && to_print) {
494  log_info("freeing index %d", index);
495  }
496 
497  // shift pointer if in safety
498  int_pointer--;
499  }
500 
501  // if safe to free, free from the correct heap based off position.
502  if ((int) ptr >= DTCM_BASE && (int) ptr <= DTCM_TOP) {
503  sark_xfree(sark.heap, int_pointer, ALLOC_LOCK);
504  } else {
505  sark_xfree(stolen_sdram_heap, int_pointer, ALLOC_LOCK);
506  }
507 }

514 static inline void build_bigger_size(void) {
515  // make twice as big tracker
516  int new_malloc_points_size = malloc_points_size * 2;
517 
518  // make new tracker
519  void **temp_pointer = sark_xalloc(
520  stolen_sdram_heap, new_malloc_points_size * sizeof(void*), 0,
521  ALLOC_LOCK);
522 
523  // check for null
524  if (temp_pointer == NULL) {
525  log_error("failed to allocate space for next range.");
526  rt_error(RTE_SWERR);
527  }
528 
529  // init the new store
530  for (int index = 0; index < new_malloc_points_size; index ++) {
531  temp_pointer[index] = 0;
532  }
533 
534  // move from old to new
535  for (int index = 0; index < malloc_points_size; index ++) {
536  temp_pointer[index] = malloc_points[index];
537  }
538 
539  // free old and update pointers
540  sark_xfree(stolen_sdram_heap, malloc_points, ALLOC_LOCK);
541  malloc_points = temp_pointer;
542  malloc_points_size = new_malloc_points_size;
543 }

579 static void add_safety_len_and_padding(int *p, uint bytes) {
580  // add len
581  int n_words = (int) ((bytes - MINUS_POINT) / BYTE_TO_WORD);
582  p[0] = n_words;
583 
584  // fill in buffer at end of malloc.
585  for (int buffer_word = 0; buffer_word < BUFFER_WORDS; buffer_word++) {
586  p[n_words + buffer_word] = SAFETY_FLAG;
587  }
588 
589  // add malloc to the malloc tracker.
590  int malloc_point_index = find_free_malloc_index();
591  if (malloc_point_index == -1) {
592  log_error("cant track this malloc. failing");
593  rt_error(RTE_SWERR);
594  }
595  malloc_points[malloc_point_index] = (void *) &p[1];
596 
597  // only print if its currently set to print (saves iobuf)
598  if (to_print) {
599  log_info("index %d", malloc_point_index);
600  log_info("address is %x", &p[1]);
601  }
602 }

625 void *malloc_extras_malloc(uint bytes) {
626  if (safety) {
627  bytes = bytes + EXTRA_BYTES;
628  }
629 
630  // try DTCM if allowed (not safe if overused, due to stack overflows)
631  int *p = NULL;
632  if (use_dtcm) {
633  p = sark_alloc(bytes, 1);
634 
635  // if DTCM failed to malloc, go to SDRAM.
636  if (p == NULL) {
637  if (to_print) {
638  log_info("went to SDRAM");
639  }
640  p = safe_sdram_malloc(bytes);
641  }
642  // only use SDRAM. (safer to avoid stack overflows)
643  } else {
644  if (to_print) {
645  log_info("went to SDRAM without checking DTCM. as requested");
646  }
647  p = safe_sdram_malloc(bytes);
648  }
649 
650  // if safety, add the len and buffers and return location for app code.
651  if (safety) {
652  add_safety_len_and_padding(p, bytes);
653 
654  // return the point were user code can use from.
655  return (void *) &p[1];
656  }
657 
658  // if no safety, the point is the point used by the application code.
659  return (void *) p;
660 }
