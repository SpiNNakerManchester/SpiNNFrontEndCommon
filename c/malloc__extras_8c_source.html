93bool malloc_extras_check(void *ptr) {
94 // only check if safety is turned on. else its not possible to check.
95 if (safety) {
96 int *int_pointer = (int *) ptr;
97 int_pointer = int_pointer - 1;
98 int words = int_pointer[0];
99
100 for (int buffer_index = 0; buffer_index < BUFFER_WORDS;
101 buffer_index++) {
102 uint32_t flag = int_pointer[words + buffer_index];
103 if (flag != SAFETY_FLAG) {
104 bool found = false;
105 for (int index = 0; index < malloc_points_size; index ++) {
106 if ((malloc_points[index] != 0) &&
107 (malloc_points[index] == ptr)) {
108 found = true;
109 }
110 }
111 if (found) {
112 log_error("flag is actually %x for ptr %x", flag, ptr);
113 } else {
114 log_error("Unexpected ptr %x", ptr);
115 }
116 return false;
117 }
118 }
119 return true;
120 }
121 return true;
122}

159 // only check if safety turned on. else pointless.
160 if (safety) {

187static inline int find_n_available_mallocs(heap_t *sdram_heap) {
188 int n_available_true_malloc = 0;
189 block_t *free_blk = sdram_heap->free;
190
191 // traverse blocks till none more available
192 while (free_blk != NULL) {
193 free_blk = free_blk->free;
194 n_available_true_malloc += 1;
195 }
196 return n_available_true_malloc;

231static inline bool add_heap_to_collection(
232 sdram_block *list_of_available_blocks) {
233 // go through true heap and allocate and add to end of list.
234 int position = 0;
235
236 // loop over the true heap and add accordingly.
237 while (sv->sdram_heap->free != NULL) {
238 block_t *next_blk = sv->sdram_heap->free->next;
239
240 // get next size minus the size it'll need to put in when alloc'ing
241 int size = ((uchar *) next_blk - (uchar *) sv->sdram_heap->free) -
242 sizeof(block_t);
243
244 // make life easier by saying blocks have to be bigger than the heap.
245 // so all spaces can be used for heaps
246 uchar *b_address = sark_xalloc(sv->sdram_heap, size, 0, ALLOC_LOCK);
247 if (b_address == NULL) {
248 log_error("failed to allocate %d", size);
249 return false;
250 }
251 list_of_available_blocks[position].sdram_base_address = b_address;
252 list_of_available_blocks[position].size = size;
253 stolen_sdram_heap->free_bytes += size;
254 position++;
255 }
256 return true;
257}

265static inline void make_heap_structure(
266 available_sdram_blocks *sizes_region, int n_mallocs,
267 sdram_block *list_of_available_blocks) {
268 // generate position pointers
269 int stolen_current_index = 0;
270 int heap_current_index = 0;
271 bool first = true;
272 block_t *previous = NULL;
273 block_t *previous_free = NULL;
274
275 // generate heap pointers
276 while (stolen_current_index < sizes_region->n_blocks ||
277 heap_current_index < n_mallocs) {
278 // build pointers to try to reduce code space
279 int *to_process;
280 sdram_block *to_process_blocks;
281
282 // determine which tracker to utilise
283 uint top_stolen = (uint) sizes_region->blocks[
284 stolen_current_index].sdram_base_address;
285 uint top_true = (uint) list_of_available_blocks[
286 heap_current_index].sdram_base_address;
287
288 // determine which one to utilise now
289 if ((stolen_current_index < sizes_region->n_blocks) &&
290 top_stolen < top_true) {
291 to_process = &stolen_current_index;
292 to_process_blocks = sizes_region->blocks;
293 } else {
294 to_process = &heap_current_index;
295 to_process_blocks = list_of_available_blocks;
296 }
297
298 // if has not already set up the heap struct, set it up
299 if (first) {
300 // set flag to not need to do this again
301 first = false;
302
303 // set up stuff we can
304 stolen_sdram_heap->free = (block_t *)
305 to_process_blocks[*to_process].sdram_base_address;
306
307 stolen_sdram_heap->free->next = (block_t *) (
308 to_process_blocks[*to_process].sdram_base_address +
309 to_process_blocks[*to_process].size - sizeof(block_t));
310
311 stolen_sdram_heap->free->free = NULL;
312 stolen_sdram_heap->first = stolen_sdram_heap->free;
313
314 // previous block in chain
315 previous = stolen_sdram_heap->free->next;
316 previous_free = stolen_sdram_heap->free;
317 } else {
318 // set up block in block
319 block_t *free = (block_t *)
320 to_process_blocks[*to_process].sdram_base_address;
321 free->free = NULL;
322
323 // update next block
324 free->next = (block_t *) (
325 to_process_blocks[*to_process].sdram_base_address +
326 to_process_blocks[*to_process].size - sizeof(block_t));
327 free->next->free = NULL;
328 free->next->next = NULL;
329
330 // update previous links
331 previous->next = free;
332 previous->free = free;
333 previous_free->free = free;
334
335 // update previous pointers
336 previous = free->next;
337 previous_free = free;
338 }
339 // update pointers
340 (*to_process)++;
341 }
342
343 // update last
344 stolen_sdram_heap->last = previous;
345 stolen_sdram_heap->last->free = NULL;
346 stolen_sdram_heap->last->next = NULL;
347}

350static inline void print_free_sizes_in_heap(void) {
351 block_t *free_blk = stolen_sdram_heap->free;
352 uint total_size = 0;
353 uint index = 0;
354
355 // traverse blocks till none more available
356 while (free_blk) {
357 uint size = (uchar *) free_blk->next - (uchar *) free_blk;
358 log_info("free block %d has address %x and size of %d",
359 index, free_blk, size);
360
361 total_size += size;
362 free_blk = free_blk->free;
363 index++;
364 }
365
366 log_info("total free size is %d", total_size);
367}

371bool malloc_extras_initialise_with_fake_heap(
372 heap_t *heap_location) {
373 stolen_sdram_heap = heap_location;
374
375 // if no real stolen SDRAM heap. point at the original SDRAM heap.
376 if (stolen_sdram_heap == NULL) {
377 stolen_sdram_heap = sv->sdram_heap;

380 // only build tracker if not already built and its expected
381 if (malloc_points == NULL && safety) {
382 build_malloc_tracker();
383 }
384 return true;

387bool malloc_extras_initialise_and_build_fake_heap(
388 available_sdram_blocks *sizes_region) {
389 // hard set stolen sdram heap to the default heap. in case no fake heap
390 stolen_sdram_heap = sv->sdram_heap;
391
392 /* if planning to track all mallocs and frees to verify no
393 overwrites/corruption. build the initial malloc tracker*/
394 if (safety) {
395 build_malloc_tracker();
396 }
397
398 // only build the fake heap if there's bits to build with
399 if (sizes_region == NULL) {
400 return true;
401 }
402
403 // allocate blocks store for figuring out block order
404 uint n_mallocs = find_n_available_mallocs(sv->sdram_heap);
405 sdram_block *list_of_available_blocks = sark_alloc(
406 n_mallocs * sizeof(sdram_block), 1);
407
408 // if fail to alloc dtcm blow up
409 if (list_of_available_blocks == NULL) {
410 return false;
411 }
412
413 // alloc space for a heap object (stealing from steal store if not by
414 // normal means.
415 stolen_sdram_heap =
416 sark_xalloc(sv->sdram_heap, MIN_SIZE_HEAP, 0, ALLOC_LOCK);
417 if (stolen_sdram_heap == NULL) {
418 // check we can steal
419 if (sizes_region->n_blocks == 0) {
420 log_error("cant find space for the heap");
421 return false;
422 }
423
424 // deallocate 32 bytes from first handed down to be the heap object
425 stolen_sdram_heap = (heap_t *)
426 sizes_region->blocks[0].sdram_base_address;
427 sizes_region->blocks[0].sdram_base_address += MIN_SIZE_HEAP;
428 sizes_region->blocks[0].size -= MIN_SIZE_HEAP;
429 }
430
431 // determine how much spare space there is.
432 stolen_sdram_heap->free_bytes = find_free_space_available(sizes_region);
433
434 // go through true heap and allocate and add to end of list.
435 bool success = add_heap_to_collection(list_of_available_blocks);
436 if (!success) {
437 log_error("failed to add heap");
438 return false;
439 }
440
441 // build the heap struct if there is a heap structure.
442 make_heap_structure(sizes_region, n_mallocs, list_of_available_blocks);
443
444 // free the allocated dtcm for the true alloc stuff.
445 sark_free(list_of_available_blocks);
446
447 // printer for sanity purposes
448 if (to_print) {
449 print_free_sizes_in_heap();
450 }
451
452 return true;
453}

462void malloc_extras_free_marked(void *ptr, int marker) {
463 // only print if its currently set to print (saves iobuf)
464 if (to_print) {
465 log_info("freeing %x", ptr);
466 }
467
468 // track if the pointer has been corrupted before trying to free it.
469 // only possible if safety been turned on
470 int *int_pointer = (int *) ptr;
471 if (safety) {
472 if (!malloc_extras_check(ptr)) {
473 log_error("over ran whatever is being freed");
474 log_error("marker is %d", marker);
475 malloc_extras_terminate(DETECTED_MALLOC_FAILURE);
476 }
477
478 bool found = false;
479 int index = 0;
480 while (!found && index < malloc_points_size) {
481 if (malloc_points[index] == ptr) {
482 found = true;
483 malloc_points[index] = 0;
484 } else {
485 index++;
486 }
487 }
488
489 // if set to print and there was a free index, print it
490 if (found && to_print) {
491 log_info("freeing index %d", index);
492 }
493
494 // shift pointer if in safety
495 int_pointer--;
496 }
497
498 // if safe to free, free from the correct heap based off position.
499 if ((int) ptr >= DTCM_BASE && (int) ptr <= DTCM_TOP) {
500 sark_xfree(sark.heap, int_pointer, ALLOC_LOCK);
501 } else {
502 sark_xfree(stolen_sdram_heap, int_pointer, ALLOC_LOCK);
503 }
504}

511static inline void build_bigger_size(void) {
512 // make twice as big tracker
513 int new_malloc_points_size = malloc_points_size * 2;
514
515 // make new tracker
516 void **temp_pointer = sark_xalloc(
517 stolen_sdram_heap, new_malloc_points_size * sizeof(void*), 0,
518 ALLOC_LOCK);
519
520 // check for null
521 if (temp_pointer == NULL) {
522 log_error("failed to allocate space for next range.");
523 rt_error(RTE_SWERR);
524 }
525
526 // init the new store
527 for (int index = 0; index < new_malloc_points_size; index ++) {
528 temp_pointer[index] = 0;
529 }
530
531 // move from old to new
532 for (int index = 0; index < malloc_points_size; index ++) {
533 temp_pointer[index] = malloc_points[index];
534 }
535
536 // free old and update pointers
537 sark_xfree(stolen_sdram_heap, malloc_points, ALLOC_LOCK);
538 malloc_points = temp_pointer;
539 malloc_points_size = new_malloc_points_size;
540}

576static void add_safety_len_and_padding(int *p, uint bytes) {
577 // add len
578 int n_words = (int) ((bytes - MINUS_POINT) / BYTE_TO_WORD);
579 p[0] = n_words;
580
581 // fill in buffer at end of malloc.
582 for (int buffer_word = 0; buffer_word < BUFFER_WORDS; buffer_word++) {
583 p[n_words + buffer_word] = SAFETY_FLAG;
584 }
585
586 // add malloc to the malloc tracker.
587 int malloc_point_index = find_free_malloc_index();
588 if (malloc_point_index == -1) {
589 log_error("cant track this malloc. failing");
590 rt_error(RTE_SWERR);
591 }
592 malloc_points[malloc_point_index] = (void *) &p[1];
593
594 // only print if its currently set to print (saves iobuf)
595 if (to_print) {
596 log_info("index %d", malloc_point_index);
597 log_info("address is %x", &p[1]);
598 }
599}

601void *malloc_extras_sdram_malloc_wrapper(uint bytes) {
602 // if using safety. add the extra bytes needed for buffer and len.
603 if (safety) {
604 bytes = bytes + EXTRA_BYTES;
605 }
606
607 // malloc from SDRAM heap.
608 int * p = safe_sdram_malloc(bytes);
609
610 // if safety, add the len and buffers and return location for app code.
611 if (safety) {
612 add_safety_len_and_padding(p, bytes);
613
614 // return the point were user code can use from.
615 return (void *) &p[1];
616 }
617
618 // if no safety, the point is the point used by the application code.
619 return (void *) p;
620}

622void *malloc_extras_malloc(uint bytes) {
623 if (safety) {
624 bytes = bytes + EXTRA_BYTES;
625 }
626
627 // try DTCM if allowed (not safe if overused, due to stack overflows)
628 int *p = NULL;
629 if (use_dtcm) {
630 p = sark_alloc(bytes, 1);
631
632 // if DTCM failed to malloc, go to SDRAM.
633 if (p == NULL) {
634 if (to_print) {
635 log_info("went to SDRAM");
636 }
637 p = safe_sdram_malloc(bytes);
638 }
639 // only use SDRAM. (safer to avoid stack overflows)
640 } else {
641 if (to_print) {
642 log_info("went to SDRAM without checking DTCM. as requested");
643 }
644 p = safe_sdram_malloc(bytes);
645 }
646
647 // if safety, add the len and buffers and return location for app code.
648 if (safety) {
649 add_safety_len_and_padding(p, bytes);
650
651 // return the point were user code can use from.
652 return (void *) &p[1];
653 }
654
655 // if no safety, the point is the point used by the application code.
656 return (void *) p;
