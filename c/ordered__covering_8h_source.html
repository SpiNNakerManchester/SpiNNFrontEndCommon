40 static unsigned int oc_get_insertion_point(
41  const unsigned int generality) {
42  // Perform a binary search of the table to find entries of generality - 1
43  const unsigned int g_m_1 = generality - 1;
44  int bottom = 0;
45  int top = routing_table_get_n_entries();
46  int pos = top / 2;
47 
48  // get first entry
49  entry_t* entry = routing_table_get_entry(pos);
50  unsigned int count_xs = key_mask_count_xs(entry->key_mask);
51 
52  // iterate till found something
53  while ((bottom < pos) && (pos < top) && (count_xs != g_m_1)) {
54  if (count_xs < g_m_1) {
55  bottom = pos;
56  } else {
57  top = pos;
58  }
59 
60  // Update the position
61  pos = bottom + (top - bottom) / 2;
62 
63  // update entry and count
64  entry = routing_table_get_entry(pos);
65  count_xs = key_mask_count_xs(entry->key_mask);
66  }
67 
68  // Iterate through the table until either the next generality or the end of
69  // the table is found.
70  while (pos < routing_table_get_n_entries() &&
71  (count_xs < generality)) {
72  pos++;
73  if (pos < routing_table_get_n_entries()) {
74  entry = routing_table_get_entry(pos);
75  count_xs = key_mask_count_xs(entry->key_mask);
76  } else {
77  // FIX MUNDY LOOKING PAST END OF TABLE!
78  count_xs = 0;
79  }
80  }
81 
82  return pos;
83 }

95 static inline bool oc_up_check(
96  merge_t *merge, int min_goodness,
97  volatile bool *stop_compressing, bool *changed) {
98  min_goodness = (min_goodness > 0) ? min_goodness : 0;
99 
100  // Get the point where the merge will be inserted into the table.
101  unsigned int generality = key_mask_count_xs(merge->key_mask);
102  unsigned int insertion_index = oc_get_insertion_point(generality);
103 
104  // For every entry in the merge check that the entry would not be covered by
105  // any existing entries if it were to be merged.
106 
107  for (unsigned int _i = routing_table_get_n_entries(),
108  i = routing_table_get_n_entries() - 1;
109  (_i > 0) && (merge_goodness(merge) > min_goodness);
110  _i--, i--) {
111  // safety check for timing limits
112  if (*stop_compressing) {
113  return false;
114  }
115 
116  if (!merge_contains(merge, i)) {
117  // If this entry is not contained within the merge skip it
118  continue;
119  }
120 
121  // Get the key_mask for this entry
122  key_mask_t km = routing_table_get_entry(i)->key_mask;
123 
124  // Otherwise look through the table from the insertion point to the
125  // current entry position to ensure that nothing covers the merge.
126  for (unsigned int j = i + 1; j < insertion_index; j++) {
127  key_mask_t other_km = routing_table_get_entry(j)->key_mask;
128 
129  // If the key masks intersect then remove this entry from the merge
130  // and recalculate the insertion index.
131  if (key_mask_intersect(km, other_km)) {
132  // Indicate the the merge has changed
133  *changed = true;
134 
135  // Remove from the merge
136  merge_remove(merge, i);
137  generality = key_mask_count_xs(merge->key_mask);
138  insertion_index = oc_get_insertion_point(generality);
139  }
140  }
141  }
142 
143  // Completely empty the merge if its goodness drops below the minimum
144  // specified
145  if (merge_goodness(merge) <= min_goodness) {
146  *changed = true;
147  merge_clear(merge);
148  }
149 
150  return true;
151 }

159 static void _get_settable(
160  key_mask_t merge_km, key_mask_t covered_km, unsigned int *stringency,
161  uint32_t *set_to_zero, uint32_t *set_to_one) {
162  // We can "set" any bit where the merge contains an X and the covered
163  // entry doesn't.
164  uint32_t setable = ~key_mask_get_xs(covered_km) & key_mask_get_xs(merge_km);
165  unsigned int new_stringency = __builtin_popcount(setable);
166 
167  uint32_t this_set_to_zero = setable & covered_km.key;
168  uint32_t this_set_to_one = setable & ~covered_km.key;
169 
170  // The stringency indicates how many bits *could* be set to avoid the cover.
171  // If this new stringency is lower than the existing stringency then we
172  // reset which bits may be set.
173  if (new_stringency < *stringency) {
174  // Update the stringency count
175  *stringency = new_stringency;
176  *set_to_zero = this_set_to_zero;
177  *set_to_one = this_set_to_one;
178  } else if (new_stringency == *stringency) {
179  *set_to_zero |= this_set_to_zero;
180  *set_to_one |= this_set_to_one;
181  }
182 }

190 static inline __sets_t _get_removables(
191  merge_t *m, uint32_t settable, bool to_one, __sets_t sets) {
192  // For each bit which we are trying to set while the best set doesn't
193  // contain only one entry.
194  for (uint32_t bit = (1 << 31); bit > 0 && sets.best->count != 1;
195  bit >>= 1) {
196 
197  // If this bit cannot be set we ignore it
198  if (!(bit & settable)) {
199  continue;
200  }
201 
202  // Loop through the table adding to the working set any entries with
203  // either a X or a 0 or 1 (as specified by `to_one`) to the working set
204  // of entries to remove.
205  int entry = 0;
206  for (int i = 0; i < routing_table_get_n_entries(); i++) {
207 
208  // Skip if this isn't an entry
209  if (!merge_contains(m, i)) {
210  continue;
211  }
212 
213  // See if this entry should be removed
214  key_mask_t km = routing_table_get_entry(i)->key_mask;
215 
216  // check entry has x or 1 or 0 in this position.
217  if ((bit & ~km.mask) || (!to_one && (bit & km.key)) ||
218  (to_one && (bit & ~km.key))) {
219 
220  // NOTE: Indexing by position in merge!
221  bit_set_add(sets.working, entry);
222  }
223 
224  // Increment the index into the merge set
225  entry++;
226  }
227 
228  // If `working` contains fewer entries than `best` or `best` is empty
229  // swap `working and best`. Otherwise just empty the working set.
230  if (sets.best->count == 0 || sets.working->count < sets.best->count) {
231  // Perform the swap
232  bit_set_t *c = sets.best;
233  sets.best = sets.working;
234  sets.working = c;
235  }
236 
237  // Clear the working merge
238  bit_set_clear(sets.working);
239  }
240 
241  return sets;
242 }

256 static bool oc_down_check(
257  merge_t *merge, int min_goodness, aliases_t *aliases,
258  bool *failed_by_malloc,
259  volatile bool *stop_compressing) {
260  min_goodness = (min_goodness > 0) ? min_goodness : 0;
261 
262  while (merge_goodness(merge) > min_goodness) {
263  // safety check for timing limits
264  if (*stop_compressing) {
265  log_error("failed due to timing");
266  return false;
267  }
268 
269  // Record if there were any covered entries
270  bool covered_entries = false;
271  // Not at all stringent
272  unsigned int stringency = 33;
273  // Mask of which bits could be set to zero
274  uint32_t set_to_zero = 0x0;
275  // Mask of which bits could be set to one
276  uint32_t set_to_one = 0x0;
277 
278  // Look at every entry between the insertion index and the end of the
279  // table to see if there are any entries which could be covered by the
280  // entry resulting from the merge.
281  int insertion_point =
282  oc_get_insertion_point(key_mask_count_xs(merge->key_mask));
283 
284  for (int i = insertion_point;
285  i < routing_table_get_n_entries() && stringency > 0;
286  i++) {
287  // safety check for timing limits
288  if (*stop_compressing) {
289  log_error("failed due to timing");
290  return false;
291  }
292 
293  key_mask_t km = routing_table_get_entry(i)->key_mask;
294  if (key_mask_intersect(km, merge->key_mask)) {
295  if (!aliases_contains(aliases, km)) {
296  // The entry doesn't contain any aliases so we need to
297  // avoid hitting the key that has just been identified.
298  covered_entries = true;
299  _get_settable(
300  merge->key_mask, km, &stringency, &set_to_zero,
301  &set_to_one);
302  } else {
303  // We need to avoid any key_masks contained within the
304  // alias table.
305  alias_list_t *the_alias_list = aliases_find(aliases, km);
306  while (the_alias_list != NULL) {
307  // safety check for timing limits
308  if (*stop_compressing) {
309  log_error("failed due to timing");
310  return false;
311  }
312  for (unsigned int j = 0; j < the_alias_list->n_elements;
313  j++) {
314 
315  // safety check for timing limits
316  if (*stop_compressing) {
317  log_error("failed due to timing");
318  return false;
319  }
320 
321  km = alias_list_get(the_alias_list, j).key_mask;
322 
323  if (key_mask_intersect(km, merge->key_mask)) {
324  covered_entries = true;
325  _get_settable(
326  merge->key_mask, km, &stringency,
327  &set_to_zero, &set_to_one);
328  }
329  }
330 
331  // Progress through the alias list
332  the_alias_list = the_alias_list->next;
333  }
334  }
335  }
336  }
337 
338  if (!covered_entries) {
339  // If there were no covered entries then we needn't do anything
340  return true;
341  }
342 
343  if (stringency == 0) {
344  // We can't avoid a covered entry at all so we need to empty the
345  // merge entirely.
346  merge_clear(merge);
347  return true;
348  }
349 
350  // Determine which entries could be removed from the merge and then
351  // pick the smallest number of entries to remove.
352  __sets_t sets;
353 
354  // allocate and free accordingly
355  sets.best = MALLOC(sizeof(bit_set_t));
356  if (sets.best == NULL) {
357  log_error("failed to alloc sets.best");
358  *failed_by_malloc = true;
359  return false;
360  }
361 
362  sets.working = MALLOC(sizeof(bit_set_t));
363  if (sets.working == NULL) {
364  log_error("failed to alloc sets.working");
365  *failed_by_malloc = true;
366 
367  // free stuff already malloc
368  bit_set_delete(sets.best);
369  FREE(sets.best);
370  return false;
371  }
372 
373  bool success = bit_set_init(sets.best, merge->entries.count);
374  if (!success) {
375  log_error("failed to init the bitfield best");
376  *failed_by_malloc = true;
377 
378  // free stuff already malloc
379  bit_set_delete(sets.best);
380  bit_set_delete(sets.working);
381  FREE(sets.best);
382  FREE(sets.working);
383  return false;
384  }
385 
386  success = bit_set_init(sets.working, merge->entries.count);
387  if (!success) {
388  log_error("failed to init the bitfield working.");
389  *failed_by_malloc = true;
390 
391  // free stuff already malloc
392  bit_set_delete(sets.best);
393  bit_set_delete(sets.working);
394  FREE(sets.best);
395  FREE(sets.working);
396  return false;
397  }
398 
399  // Get the entries that can be removed because of the filtering we have
400  // computed above
401  sets = _get_removables(merge, set_to_zero, false, sets);
402  sets = _get_removables(merge, set_to_one, true, sets);
403 
404  // Remove the specified entries
405  int entry = 0;
406  for (int i = 0; i <routing_table_get_n_entries(); i++) {
407  // safety check for timing limits
408  if (*stop_compressing) {
409  log_error("failed due to timing");
410  bit_set_delete(sets.best);
411  bit_set_delete(sets.working);
412  FREE(sets.best);
413  FREE(sets.working);
414  return false;
415  }
416 
417  if (merge_contains(merge, i)) {
418  if (bit_set_contains(sets.best, entry)) {
419  // Remove this entry from the merge
420  merge_remove(merge, i);
421  }
422  entry++;
423  }
424  }
425 
426  // Tidy up
427  bit_set_delete(sets.best);
428  FREE(sets.best);
429  sets.best=NULL;
430  bit_set_delete(sets.working);
431  FREE(sets.working);
432  sets.working=NULL;
433 
434  // If the merge only contains 1 entry empty it entirely
435  if (merge->entries.count == 1) {
436  log_debug("final merge clear");
437  merge_clear(merge);
438  }
439  }
440  log_debug("returning from down check");
441  return true;
442 }

452 static inline bool oc_get_best_merge(
453  aliases_t *aliases, merge_t *best, bool *failed_by_malloc,
454  volatile bool *stop_compressing) {
455  // Keep track of which entries have been considered as part of merges.
456  bit_set_t considered;
457  bool success = bit_set_init(
458  &considered, routing_table_get_n_entries());
459  if (!success) {
460  log_warning("failed to initialise the bit_set. throwing response malloc");
461  *failed_by_malloc = true;
462  return false;
463  }
464 
465  // Keep track of the current best merge and also provide a working merge
466  merge_t working;
467 
468  success = merge_init(best, routing_table_get_n_entries());
469  if (!success) {
470  log_warning("failed to init the merge best. throw response malloc");
471  *failed_by_malloc = true;
472 
473  // free bits we already done
474  bit_set_delete(&considered);
475 
476  // return false
477  return false;
478  }
479 
480  success = merge_init(&working, routing_table_get_n_entries());
481  if (!success) {
482  log_warning("failed to init the merge working. throw response malloc");
483  *failed_by_malloc = true;
484 
485  // free bits we already done
486  merge_delete(best);
487  bit_set_delete(&considered);
488 
489  // return false
490  return false;
491  }
492 
493  // For every entry in the table see with which other entries it could be
494  // merged.
495  log_debug("starting search for merge entry");
496  for (int i = 0; i < routing_table_get_n_entries(); i++) {
497  // safety check for timing limits
498  if (*stop_compressing) {
499  log_info("closed due to stop request");
500  // free bits we already done
501  merge_delete(best);
502  bit_set_delete(&considered);
503  return false;
504  }
505 
506  // If this entry has already been considered then skip to the next
507  if (bit_set_contains(&considered, i)) {
508  continue;
509  }
510 
511  // Otherwise try to build a merge
512  // Clear the working merge
513  merge_clear(&working);
514 
515  // Add to the merge
516  merge_add(&working, i);
517 
518  // Mark this entry as considered
519  bit_set_add(&considered, i);
520 
521  // Get the entry
522  entry_t *entry = routing_table_get_entry(i);
523 
524  // Try to merge with other entries
525  log_debug("starting second search at index %d", i);
526  for (int j = i+1; j < routing_table_get_n_entries(); j++) {
527  // safety check for timing limits
528  if (*stop_compressing) {
529  log_info("closed due to stop request");
530  // free bits we already done
531  merge_delete(best);
532  bit_set_delete(&considered);
533  return false;
534  }
535 
536  // Get the other entry
537  entry_t *other = routing_table_get_entry(j);
538 
539  // check if merge-able
540  if (entry->route == other->route) {
541  // If the routes are the same then the entries may be merged
542  // Add to the merge
543  merge_add(&working, j);
544 
545  // Mark the other entry as considered
546  bit_set_add(&considered, j);
547  }
548  }
549 
550  if (merge_goodness(&working) <= merge_goodness(best)) {
551  continue;
552  }
553 
554  // Perform the first down check
555  success = oc_down_check(
556  &working, merge_goodness(best), aliases, failed_by_malloc,
557  stop_compressing);
558  if (!success) {
559  log_error("failed to down check.");
560 
561  // free bits we already done
562  merge_delete(best);
563  bit_set_delete(&considered);
564 
565  return false;
566  }
567 
568  if (merge_goodness(&working) <= merge_goodness(best)) {
569  continue;
570  }
571 
572  // Perform the up check, seeing if this actually makes a change to the
573  // size of the merge.
574  bool changed = false;
575  success = oc_up_check(
576  &working, merge_goodness(best), stop_compressing, &changed);
577  if (!success) {
578  log_info("failed to upcheck");
579  // free bits we already done
580  merge_delete(best);
581  bit_set_delete(&considered);
582  return false;
583  }
584 
585  // if changed the merge
586  if (changed) {
587  if (merge_goodness(&working) <= merge_goodness(best)) {
588  continue;
589  }
590 
591  // If the up check did make a change then the down check needs to
592  // be run again.
593  log_debug("down check");
594  success = oc_down_check(
595  &working, merge_goodness(best), aliases, failed_by_malloc,
596  stop_compressing);
597  if (!success) {
598  log_error("failed to down check. ");
599 
600  // free bits we already done
601  merge_delete(best);
602  bit_set_delete(&considered);
603 
604  return false;
605  }
606  }
607 
608  // If the merge is still better than the current best merge we swap the
609  // current and best merges to record the new best merge.
610  if (merge_goodness(best) < merge_goodness(&working)) {
611  merge_t other = working;
612  working = *best;
613  *best = other;
614  }
615  }
616 
617  // Tidy up
618  merge_delete(&working);
619  bit_set_delete(&considered);
620  log_debug("n entries is %d", routing_table_get_n_entries());
621  // found the best merge. return true
622  return true;
623 }

631 static inline bool oc_merge_apply(
632  merge_t *merge, aliases_t *aliases, bool *failed_by_malloc) {
633  // Get the new entry
634  entry_t new_entry;
635  new_entry.key_mask = merge->key_mask;
636  new_entry.route = merge->route;
637  new_entry.source = merge->source;
638 
639  log_debug("new entry k %x mask %x route %x source %x x mergable is %d",
640  new_entry.key_mask.key, new_entry.key_mask.mask, new_entry.route,
641  new_entry.source, merge->entries.count);
642 
643  // Get the insertion point for the new entry
644  int insertion_point =
645  oc_get_insertion_point(key_mask_count_xs(merge->key_mask));
646  log_debug("the insertion point is %d", insertion_point);
647 
648  // Keep track of the amount of reduction of the finished table
649  int reduced_size = 0;
650 
651  // Create a new aliases list with sufficient space for the key_masks of all
652  // of the entries in the merge. NOTE: IS THIS REALLY REQUIRED!?
653  log_debug("new alias");
654  alias_list_t *new_aliases = alias_list_new(merge->entries.count);
655  if (new_aliases == NULL) {
656  log_error("failed to malloc new alias list");
657  *failed_by_malloc = true;
658  return false;
659  }
660 
661  log_debug("alias insert");
662  bool malloc_success =
663  aliases_insert(aliases, new_entry.key_mask, new_aliases);
664  if (!malloc_success) {
665  log_error("failed to malloc new alias list during insert");
666  *failed_by_malloc = true;
667  return false;
668  }
669 
670  // Use two iterators to move through the table copying entries from one
671  // position to the other as required.
672  int insert = 0;
673 
674  log_debug("routing table entries = %d", routing_table_get_n_entries());
675 
676  for (int remove = 0; remove < routing_table_get_n_entries(); remove++) {
677  // Grab the current entry before we possibly overwrite it
678  entry_t* current = routing_table_get_entry(remove);
679  log_debug("bacon");
680  log_debug("entry %d being processed has key %x or %d, mask %x "
681  "route %x source %x", remove, current->key_mask.key,
682  current->key_mask.key, current->key_mask.mask, current->route,
683  current->source);
684  // Insert the new entry if this is the correct position at which to do
685  // so
686  if (remove == insertion_point) {
687  log_debug("remove == insert at index %d and insert %d at "
688  "insert point", remove, insert);
689 
690  entry_t *insert_entry = routing_table_get_entry(insert);
691 
692  // move data between entries
693  insert_entry->key_mask.key = new_entry.key_mask.key;
694  insert_entry->key_mask.mask = new_entry.key_mask.mask;
695  insert_entry->route = new_entry.route;
696  insert_entry->source = new_entry.source;
697  insert++;
698  }
699 
700  // If this entry is not contained within the merge then copy it from its
701  // current position to its new position.
702  if (!merge_contains(merge, remove)) {
703  log_debug("not contains at index %d and insert %d at insert point",
704  remove, insert);
705 
706  entry_t *insert_entry = routing_table_get_entry(insert);
707 
708  // move data between entries
709  insert_entry->key_mask.key = current->key_mask.key;
710  insert_entry->key_mask.mask = current->key_mask.mask;
711  insert_entry->route = current->route;
712  insert_entry->source = current->source;
713  insert++;
714 
715  } else {
716  log_debug("merge contains at index %d and insert %d at "
717  "insert point", remove, insert);
718 
719  // Otherwise update the aliases table to account for the entry
720  // which is being merged.
721  key_mask_t km = current->key_mask;
722  uint32_t source = current->source;
723 
724  if (aliases_contains(aliases, km)) {
725  // Join the old list of aliases with the new
726  alias_list_join(new_aliases, aliases_find(aliases, km));
727 
728  // Remove the old aliases entry
729  aliases_remove(aliases, km);
730  } else {
731  // Include the key_mask in the new list of aliases
732  alias_list_append(new_aliases, km, source);
733  }
734 
735  // Decrement the final table size to account for this entry being
736  // removed.
737  reduced_size++;
738 
739  log_debug("at index %d and insert %d not at merge point",
740  remove, insert);
741  }
742  }
743 
744  log_debug("alias delete");
745  alias_list_delete(new_aliases);
746  log_debug("alias delete end");
747 
748  // If inserting beyond the old end of the table then perform the insertion
749  // at the new end of the table.
750  if (insertion_point == routing_table_get_n_entries()) {
751  log_debug("insert point was at end of table, new insert point is %d",
752  insert);
753  entry_t *insert_entry = routing_table_get_entry(insert);

774 bool minimise_run(
775  int target_length, bool *failed_by_malloc,
776  volatile bool *stop_compressing) {
777  // check if any compression actually needed
778  log_debug("n entries before compression is %d",
779  routing_table_get_n_entries());
780 
781  if (routing_table_get_n_entries() < target_length) {
782  log_info("does not need compression.");
783  return true;
784  }
785 
786  // Some Mundy black magic
787  aliases_t aliases = aliases_init();
788 
789  // start the merger process
790  log_debug("n entries is %d", routing_table_get_n_entries());
791  int attempts = 0;
792 
793  while ((routing_table_get_n_entries() > target_length) &&
794  !*stop_compressing) {
795  log_debug("n entries is %d", routing_table_get_n_entries());
796 
797  // Get the best possible merge, if this merge is empty then break out
798  // of the loop.
799  merge_t merge;
800  bool success = oc_get_best_merge(
801  &aliases, &merge, failed_by_malloc, stop_compressing);
802  if (!success) {
803  log_debug("failed to do get best merge. "
804  "the number of merge cycles were %d", attempts);
805  aliases_clear(&aliases);
806  return false;
807  }
808 
809  log_debug("best merge end");
810  unsigned int count = merge.entries.count;
811 
812  if (count > 1) {
813  // Apply the merge to the table if it would result in merging
814  // actually occurring.
815  //routing_tables_print_out_table_sizes();
816  log_debug("merge apply");
817  bool malloc_success = oc_merge_apply(
818  &merge, &aliases, failed_by_malloc);
819 
820  if (!malloc_success) {
821  log_error("failed to malloc");
822  aliases_clear(&aliases);
823  return false;
824  }
825  log_debug("merge apply end");
826  //routing_tables_print_out_table_sizes();
827  }
828 
829  // Free any memory used by the merge
830  log_debug("merge delete");
831  merge_delete(&merge);
832  log_debug("merge delete end");
833 
834  // Break out of the loop if no merge could be performed (indicating
835  // that no more minimisation is possible).
836  if (count < 2) {
837  break;
838  }
839  attempts += 1;
840  }
841 
842  // shut down timer. as passed the compression
843  spin1_pause();
844 
845  // if it failed due to timing, report and then reset and fail.
846  if (*stop_compressing) {
847  log_info("Asked to stop. reached %d entries over %d attempts",
848  routing_table_get_n_entries(), attempts);
849  spin1_pause();
850  aliases_clear(&aliases);
851  return false;
852  }
853 
854  log_info("entries after compressed = %d, target_length = %d, stop = %d, "
855  " the number of merge cycles were %d",
856  routing_table_get_n_entries(), target_length, *stop_compressing,
857  attempts);
858 
859  log_debug("compressed!!!");
860  log_debug("produced table with %d entries", routing_table_get_n_entries());
861  aliases_clear(&aliases);
862  return true;
863 }
