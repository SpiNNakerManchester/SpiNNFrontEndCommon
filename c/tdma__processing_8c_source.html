65 void tdma_processing_send_packet(
66  uint32_t transmission_key, uint32_t payload,
67  uint32_t with_payload, uint32_t timer_count) {
68 
69  uint32_t timer_value = tc[T1_COUNT];
70 
71  // Find the next valid phase to send in; might run out of phases, at
72  // which point we will sent immediately. We also should just send
73  // if the timer has already expired completely as then we are really late!
74  while ((ticks == timer_count) && (timer_value < expected_time)
75  && (expected_time > params.min_expected_time)) {
76  expected_time -= params.time_between_sends;
77  }
78 
79  // Wait until the expected time to send; might already have passed in
80  // which case we just skip this
81  while ((ticks == timer_count) && (tc[T1_COUNT] > expected_time)) {
82  // Do Nothing
83  }
84 
85  // Send the spike
86  log_debug("sending spike %d", transmission_key);
87  while (!spin1_send_mc_packet(transmission_key, payload, with_payload)) {
88  spin1_delay_us(1);
89  }

91 }
