61 static inline void next_core_atom(filter_info_t *filter, struct core_atom *core_atom) {
62  core_atom->atom++;
63  // If n_atoms_per_core is 0 (i.e. disabled) this will never match, so the
64  // atom will be global
65  if (core_atom->atom == filter->n_atoms_per_core) {
66  core_atom->core++;
67  core_atom->atom = 0;
68  }
69 }
