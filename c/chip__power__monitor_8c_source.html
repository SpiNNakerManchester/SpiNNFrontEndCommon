123 static void resume_callback(void) {
124  // change simulation ticks to be a number related to sampling frequency
125  if (time == UINT32_MAX) {
126  log_info("resume_skipped as time still zero");
127  } else {
128  simulation_ticks = (simulation_ticks * timer) / sample_frequency;
129  log_info("total_sim_ticks = %d", simulation_ticks);
130  recording_reset();
131  log_info("resume_callback");
132  }

139 static inline void count_core_states(void) {
140  uint32_t sample = get_sample();
141 
142  for (uint32_t i = 0, j = 1 ; i < NUM_CPUS ; i++, j <<= 1) {
143  if (!(sample & j)) {
144  core_counters[i]++;
145  }
146  }

152 static void sample_in_slot(UNUSED uint unused0, UNUSED uint unused1) {
153  time++;
154 
155  // handle the situation when the first time update is sent
156  if (time == 0) {
157  simulation_ticks = (simulation_ticks * timer) / sample_frequency;
158  log_info("total_sim_ticks = %d", simulation_ticks);
159  }
160  // check if the simulation has run to completion
161  if (simulation_is_finished()) {
162  simulation_handle_pause_resume(resume_callback);
163 
164  recording_finalise();
165 
166  // Subtract 1 from the time so this tick gets done again on the next
167  // run
168  time--;
169 
170  simulation_ready_to_read();
171  }
172 
173  uint32_t count = ++sample_count;
174  uint32_t offset = get_random_busy();
175  while (offset --> 0) {
176  // Do nothing
177  }
178 
179  count_core_states();
180  if (count >= sample_count_limit) {
181  record_aggregate_sample();
182  reset_core_counters();
183  }
184 
185  recording_do_timestep_update(time);
186 }

202 static bool initialize(void) {
203  data_specification_metadata_t *ds_regions =
204  data_specification_get_data_address();
205  if (!data_specification_read_header(ds_regions)) {
206  return false;
207  }
208  if (!simulation_initialise(
209  data_specification_get_region(SYSTEM, ds_regions),
210  APPLICATION_NAME_HASH, &timer, &simulation_ticks,
211  &infinite_run, &time, SDP, DMA)) {
212  return false;
213  }
214  if (!read_parameters(
215  data_specification_get_region(CONFIG, ds_regions))) {
216  return false;
217  }
218 
219  // change simulation ticks to be a number related to sampling frequency
220  simulation_ticks = (simulation_ticks * timer) / sample_frequency;
221  log_info("total_sim_ticks = %d", simulation_ticks);
222 
223  void *recording_region =
224  data_specification_get_region(RECORDING, ds_regions);
225  return recording_initialize(&recording_region, &recording_flags);
226 }
