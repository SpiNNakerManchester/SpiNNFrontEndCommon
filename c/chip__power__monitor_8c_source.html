122static inline void rescale_sim_ticks(void) {
123 simulation_ticks = (simulation_ticks * timer) / sample_frequency;

125 simulation_ticks, timer, sample_frequency, time);

129static void resume_callback(void) {
130 // change simulation ticks to be a number related to sampling frequency
131 if (time == UINT32_MAX) {
132 log_debug("resume_skipped as time still zero");
133 } else {
134 rescale_sim_ticks();
135 // Also rescale time appropriately
136 reset_core_counters();
137 time++;
138 time = (time * timer) / sample_frequency;
139 // Subtract 1 again now so that this starts at the "correct" value on the next tick
140 time--;
141 log_info("resume total_sim_ticks = %d timer %d sample_frequency %d time %d",
142 simulation_ticks, timer, sample_frequency, time);
143 recording_reset();
144 log_debug("resume_callback");
145 }
146}

152static inline void count_core_states(void) {
153 uint32_t sample = get_sample();
154
155 for (uint32_t i = 0, j = 1 ; i < NUM_CPUS ; i++, j <<= 1) {
156 if (!(sample & j)) {
157 core_counters[i]++;
158 }
159 }
160}

165static void sample_in_slot(UNUSED uint unused0, UNUSED uint unused1) {
166 time++;
167
168 // handle the situation when the first time update is sent
169 if (time == 0) {
170 rescale_sim_ticks();
171 }
172 // check if the simulation has run to completion
173 if (simulation_is_finished()) {
174 simulation_handle_pause_resume(resume_callback);
175
176 recording_finalise();
177
178 // Subtract 1 from the time so this tick gets done again on the next
179 // run
180 time--;
181
182 simulation_ready_to_read();
183
184 return;
185 }
186
187 uint32_t count = ++sample_count;
188 uint32_t offset = get_random_busy();
189 while (offset --> 0) {
190 // Do nothing
191 }
192
193 count_core_states();
194 if (count >= sample_count_limit) {
195 record_aggregate_sample();
196 reset_core_counters();
197 }
198}

214static bool initialize(void) {
215 data_specification_metadata_t *ds_regions =
216 data_specification_get_data_address();
217 if (!data_specification_read_header(ds_regions)) {
218 return false;
219 }
220 if (!simulation_initialise(
221 data_specification_get_region(SYSTEM, ds_regions),
222 APPLICATION_NAME_HASH, &timer, &simulation_ticks,
223 &infinite_run, &time, SDP, DMA)) {
224 return false;
225 }
226 if (!read_parameters(
227 data_specification_get_region(CONFIG, ds_regions))) {
228 return false;
229 }
230
231 void *recording_region =
232 data_specification_get_region(RECORDING, ds_regions);
233 return recording_initialize(&recording_region, &recording_flags);
234}

240void c_main(void) {
241 if (!initialize()) {
242 log_error("failed to initialise");
243 rt_error(RTE_SWERR);
244 }
245
246 reset_core_counters();
247
248 spin1_set_timer_tick(sample_frequency);
249 spin1_callback_on(TIMER_TICK, sample_in_slot, TIMER);
250 time = UINT32_MAX;
251 simulation_run();
252}
